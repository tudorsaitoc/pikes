25  public class EverythingHandler extends AbstractHandler {
26  
27  	private static final Logger LOGGER = LoggerFactory.getLogger(EverythingHandler.class);
28  
29  	public EverythingHandler(AnnotationPipeline pipeline) {
30  		super(pipeline);
31  	}
32  
33  	@Override
34  	public void service(Request request, Response response) throws Exception {
35  
36  		super.service(request, response);
37  
38  		String host = request.getHeader("x-forwarded-for");
39  
40  		String referer = request.getHeader("referer");
41  		String okReferer = pipeline.getConfig().getProperty("back_referer");
42  
43  		boolean backLink = false;
44  		if (referer != null && okReferer != null && referer.equals(okReferer)) {
45  			backLink = true;
46  		}
47  
48  		String text = request.getParameter("text");
49  
50  		// Log for stats
51  		LOGGER.info("[SENTENCE]");
52  		LOGGER.info("Host: {}", host);
53  		LOGGER.info("Text: {}", text);
54  
55  		KAFDocument doc = text2naf(text, meta);
56  
57  		doc = pipeline.parseFromString(doc.toString(), annotators);
58  
59  		String viewString;
60  		try {
61  
62  			HashMap<String, Object> demoProperties = new HashMap<>();
63  			demoProperties.put("renderer.template.title", "PIKES demo");
64  			if (backLink) {
65  				demoProperties.put("renderer.template.backlink", "javascript:history.back();");
66  			}
67  			else {
68  				demoProperties.put("renderer.template.backlink", pipeline.getConfig().getProperty("back_alt_link"));
69  				demoProperties.put("renderer.template.backlabel", pipeline.getConfig().getProperty("back_alt_text"));
70  			}
71  
72  			boolean fusion = request.getParameter("rdf_fusion") != null;
73  			boolean normalization = request.getParameter("rdf_compaction") != null;
74  
75  			demoProperties.put("generator.fusion", fusion);
76  			demoProperties.put("generator.normalization", normalization);
77  
78  			NAFFilter filter = NAFFilter.builder().withProperties(pipeline.getConfig(), "filter").build();
79  			RDFGenerator generator = RDFGenerator.builder().withProperties(demoProperties, "generator").build();
80  			Renderer renderer = Renderer.builder().withProperties(demoProperties, "renderer").build();
81  
82  			filter.filter(doc);
83  			final Model model = generator.generate(doc, null);
84  			StringWriter writer = new StringWriter();
85  			renderer.renderAll(writer, doc, model, null, null);
86  			viewString = writer.toString();
87  
88  		} catch (Exception e) {
89  			e.printStackTrace();
90  			viewString = "Unable to show graph. <br /><br />\n<pre>" + doc.toString().replace("<", "&lt;").replace(">", "&gt;") + "</pre>";
91  		}
92  
93  		super.writeOutput(response, "text/html", viewString);
94  	}
95  }
