100     public static void normalize(final KAFDocument document) {
101 
102         // Convert SST, synset and BBN attributes to external refs
103         for (final Term term : document.getTerms()) {
104             boolean hasBBN = false;
105             boolean hasSynset = false;
106             boolean hasSST = false;
107             for (final ExternalRef ref : term.getExternalRefs()) {
108                 hasBBN |= RESOURCE_BBN.equalsIgnoreCase(ref.getResource());
109                 hasSynset |= RESOURCE_WN_SYNSET.equalsIgnoreCase(ref.getResource());
110                 hasSST |= RESOURCE_WN_SST.equalsIgnoreCase(ref.getResource());
111             }
112             if (!hasBBN && term.getBBNTag() != null) {
113                 term.addExternalRef(document.newExternalRef(RESOURCE_BBN, term.getBBNTag()));
114             }
115             if (!hasSynset && term.getWordnetSense() != null) {
116                 term.addExternalRef(document.newExternalRef(RESOURCE_WN_SYNSET,
117                         term.getWordnetSense()));
118             }
119             if (!hasSST && term.getSupersenseTag() != null) {
120                 term.addExternalRef(document.newExternalRef(RESOURCE_WN_SST,
121                         term.getSupersenseTag()));
122             }
123             term.setBBNTag(null);
124             term.setWordnetSense(null);
125             term.setSupersenseTag(null);
126         }
127 
128         // Remove duplicate external refs
129         for (final Predicate predicate : document.getPredicates()) {
130             normalizeRefs(getRefs(predicate));
131             for (final Role role : predicate.getRoles()) {
132                 normalizeRefs(getRefs(role));
133             }
134         }
135     }

137     public static List<Term> filterTerms(final Iterable<Term> terms) {
138         final List<Term> result = Lists.newArrayList();
139         boolean atBeginning = true;
140         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
141             final char pos = Character.toUpperCase(term.getPos().charAt(0));
142             if (atBeginning && (pos == 'D' || pos == 'P')) {
143                 continue;
144             }
145             for (final WF word : term.getWFs()) {
146                 final String text = word.getForm();
147                 if (SYMBOLS.contains(text) || !WF_EXCLUSION_PATTERN.matcher(text).matches()) {
148                     result.add(term);
149                     atBeginning = false;
150                     break;
151                 }
152             }
153         }
154         return result;
155     }

157     public static String getText(final Iterable<Term> terms) {
158         final StringBuilder builder = new StringBuilder();
159         boolean atBeginning = true;
160         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
161             final boolean properNoun = term.getMorphofeat().startsWith("NNP");
162             for (final WF word : term.getWFs()) {
163                 builder.append(atBeginning ? "" : " ");
164                 builder.append(properNoun ? word.getForm() : word.getForm().toLowerCase());
165                 atBeginning = false;
166             }
167         }
168         return builder.toString();
169     }

183     public static Set<Term> extractHeads(final KAFDocument document,
184             @Nullable final Iterable<Term> ancestors, @Nullable final Iterable<Term> span,
185             @Nullable final java.util.function.Predicate<Term> predicate) {
186 
187         Set<Term> ancestorSet;
188         if (ancestors != null) {
189             ancestorSet = ImmutableSet.copyOf(ancestors);
190         } else {
191             ancestorSet = Sets.newHashSet();
192             final Set<Term> termSet = Sets.newHashSet(span);
193             for (final Term term : termSet) {
194                 final Dep dep = document.getDepToTerm(term);
195                 if (dep == null || !termSet.contains(dep.getFrom())) {
196                     ancestorSet.add(term);
197                 }
198             }
199         }
200 
201         final Set<Term> result = Sets.newHashSet();
202         for (final Term ancestor : ancestorSet) {
203             extractHeadsHelper(document, ancestor, predicate, result);
204         }
205         if (span != null) {
206             result.retainAll(ImmutableSet.copyOf(span));
207         }
208         // System.err.println(document.getPublic().uri + " -> " + termFilter + " / " + ancestors
209         // + " -> " + result);
210         return result;
211     }

213     private static boolean extractHeadsHelper(final KAFDocument document, final Term term,
214             final java.util.function.Predicate<Term> predicate, final Collection<Term> result) {
215         final String pos = extendedPos(document, term);
216         boolean accepted = false;
217         if (pos.startsWith("V")) {
218             final Term srlHead = syntacticToSRLHead(document, term);
219             if (!term.equals(srlHead)) {
220                 accepted = extractHeadsHelper(document, srlHead, predicate, result);
221             }
222         }
223         if (!accepted && (predicate == null || predicate.test(term))) {
224             result.add(term);
225             accepted = true;
226         }
227         if (accepted) {
228             for (final Dep dep : document.getDepsFromTerm(term)) {
229                 if (dep.getRfunc().toUpperCase().contains("COORD")) {
230                     extractHeadsHelper(document, dep.getTo(), predicate, result);
231                 }
232             }
233         } else {
234             for (final Dep dep : document.getDepsFromTerm(term)) {
235                 extractHeadsHelper(document, dep.getTo(), predicate, result);
236             }
237         }
238         return accepted;
239     }

241     public static boolean hasHead(final KAFDocument document, final Object annotation,
242             final Term head) {
243         List<Span<Term>> spans;
244         if (annotation instanceof Coref) {
245             spans = ((Coref) annotation).getSpans();
246         } else if (annotation instanceof Entity) {
247             spans = ((Entity) annotation).getSpans();
248         } else if (annotation instanceof Timex3) {
249             spans = ImmutableList.of(KAFDocument.newTermSpan(document
250                     .getTermsByWFs(((Timex3) annotation).getSpan().getTargets())));
251         } else if (annotation instanceof Predicate) {
252             spans = ImmutableList.of(((Predicate) annotation).getSpan());
253         } else if (annotation instanceof Role) {
254             spans = ImmutableList.of(((Role) annotation).getSpan());
255         } else {
256             throw new IllegalArgumentException("Unsupported annotation: " + annotation);
257         }
258         for (final Span<Term> span : spans) {
259             if (head == extractHead(document, span)) {
260                 return true;
261             }
262         }
263         return false;
264     }

266     public static Span<Term> getNominalSpan(final KAFDocument document, final Term term,
267             final boolean includeCoord, final boolean includeModifiers) {
268 
269         // Start from the supplied term
270         final Set<Term> terms = Sets.newHashSet(term);
271 
272         // Identify head and terms of all NE and TIMEX markables containing supplied term
273         final Map<Term, List<Term>> markables = Maps.newHashMap();
274         for (final Entity entity : document.getEntitiesByTerm(term)) {
275             markables.put(document.getTermsHead(entity.getTerms()), entity.getTerms());
276         }
277         for (final WF wf : term.getWFs()) {
278             for (final Timex3 timex : document.getTimeExsByWF(wf)) {
279                 final List<Term> span = document.getTermsByWFs(timex.getSpan().getTargets());
280                 markables.put(document.getTermsHead(span), span);
281             }
282         }
283 
284         // Add the terms of the smallest markable 'matching' the term (i.e., whose head matches
285         // the term or a term ancestor in the dependency tree)
286         if (!markables.isEmpty()) {
287             Term t = term;
288             while (true) {
289                 final List<Term> parent = markables.get(t);
290                 if (parent != null) {
291                     terms.addAll(parent);
292                     break;
293                 }
294                 final Dep dep = document.getDepToTerm(t);
295                 if (dep == null) {
296                     break;
297                 }
298                 t = dep.getFrom();
299             }
300         }
301 
302         // Identify head
303         final Term head = document.getTermsHead(terms);
304 
305         // Add all terms reachable from the head using a regex
306         final String regex = includeCoord ? includeModifiers ? "(COORD CONJ?)* ((NAME|NMOD|AMOD|TMP) .*)?"
307                 : "(COORD CONJ?)* NAME"
308                 : includeModifiers ? "((NAME|NMOD|AMOD|TMP) .*)?" : "NAME";
309         terms.addAll(document.getTermsByDepAncestors(Collections.singleton(head), regex));
310 
311         // Sort obtained terms by offset and return resulting list
312         return KAFDocument.newTermSpan(Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms),
313                 head);
314     }

339             return null;

345         if (role == null) {
346             return null;
347         }

349         while (index > 0 && Character.isDigit(role.charAt(index - 1))) {

361     public static int getEnd(final Term term) {
362         final List<WF> wfs = term.getWFs();
363         final WF wf = wfs.get(wfs.size() - 1);
364         final String str = wf.getForm();
365         if (str.equals("-LSB-") || str.equals("-RSB-") || str.equals("''")) {
366             return wf.getOffset() + 1;
367         }
368         return wf.getOffset() + wf.getLength();
369     }

376         final String res = predicate.getTerms().get(0).getPos().equalsIgnoreCase("V") ? RESOURCE_PROPBANK
377                 : RESOURCE_NOMBANK;
378         String roleset = null;

382                     roleset = ref.getReference();
383                     break;
384                 } else if (roleset == null) {
385                     roleset = ref.getReference();
386                 }
387             }
388         }
389         return roleset;

396             @Nullable final String resource, @Nullable final String reference) {
397         ExternalRef result = null;
398         for (final ExternalRef ref : getRefs(annotation)) {
399             if (matchRef(ref, resource, reference)) {

402                             + resource + ", reference " + reference + ": " + ref.getReference()
403                             + ", " + result.getReference());
404                 }
405                 result = ref;
406             }
407         }
408         return result;
409     }

518     public static List<Range> termRangesFor(final KAFDocument document, final Iterable<Term> terms) {
519         final List<Range> ranges = Lists.newArrayList();
520         int startIndex = -1;
521         int lastIndex = -2;
522         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
523             final int termIndex = document.getTerms().indexOf(term);
524             if (termIndex - lastIndex > 1) {
525                 if (startIndex >= 0) {
526                     ranges.add(Range.create(startIndex, lastIndex + 1));
527                 }
528                 startIndex = termIndex;
529             }
530             lastIndex = termIndex;
531         }
532         if (startIndex != -1 && lastIndex >= startIndex) {
533             ranges.add(Range.create(startIndex, lastIndex + 1));
534         }
535         return ranges;
536     }

538     public static List<Range> rangesFor(final KAFDocument document, final Iterable<Term> terms) {
539         final List<Range> ranges = Lists.newArrayList();
540         int startOffset = -1;
541         int endOffset = -1;
542         int termIndex = -2;
543         for (final Term term : Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms)) {
544             final int lastTermIndex = termIndex;
545             termIndex = document.getTerms().indexOf(term);
546             if (termIndex - lastTermIndex > 1) {
547                 if (startOffset != -1) {
548                     ranges.add(Range.create(startOffset, endOffset));
549                 }
550                 startOffset = term.getOffset();
551             }
552             endOffset = NAFUtils.getEnd(term);
553         }
554         if (startOffset != -1 && endOffset > startOffset) {
555             ranges.add(Range.create(startOffset, endOffset));
556         }
557         return ranges;
558     }

574     @Nullable
575     public static Span<Term> trimSpan(@Nullable final Span<Term> span, final int sentenceID) {
576         if (span == null || span.isEmpty()) {
577             return null;
578         }
579         boolean sameSentence = true;
580         for (final Term term : span.getTargets()) {
581             if (term.getSent() != sentenceID) {
582                 sameSentence = false;
583                 break;
584             }
585         }
586         if (sameSentence) {
587             return span;
588         }
589         final List<Term> filteredTerms = Lists.newArrayList();
590         for (final Term term : span.getTargets()) {
591             if (term.getSent() == sentenceID) {
592                 filteredTerms.add(term);
593             }
594         }
595         final Span<Term> result = KAFDocument.newTermSpan(filteredTerms);
596         for (final Term head : span.getHeads()) {
597             if (head.getSent() == sentenceID) {
598                 result.getHeads().add(head);
599             }
600         }
601         return result;
602     }

606     public static Span<Term> normalizeSpan(final KAFDocument document,
607             @Nullable final Span<Term> span) {
608 
609         // Handle null and empty span
610         if (span == null || Iterables.isEmpty(span.getTargets())) {
611             return KAFDocument.newTermSpan();
612         }
613 
614         // Identify all the 'root' terms in the span whose dep tree parent is outside the span
615         final Set<Term> roots = Sets.newHashSet();
616         final Set<Term> terms = ImmutableSet.copyOf(span.getTargets());
617         for (final Term term : terms) {
618             final Dep dep = document.getDepToTerm(term);
619             if (dep == null || !terms.contains(dep.getFrom())) {
620                 roots.add(term);
621             }
622         }
623 
624         // If only one 'root', return the normalized span having that root as the head
625         if (roots.size() == 1) {
626             return KAFDocument.newTermSpan(span.getTargets(), roots.iterator().next());
627         }
628 
629         // Otherwise, look for the closest head outside the span. First compute all the paths from
630         // the dep tree roots to the 'root' terms identified before
631         final List<List<Term>> paths = Lists.newArrayList();
632         for (final Term root : roots) {
633             final List<Term> path = Lists.newArrayList(root);
634             for (Dep dep = document.getDepToTerm(root); dep != null; dep = document
635                     .getDepToTerm(dep.getFrom())) {
636                 path.add(dep.getFrom());
637             }
638             Collections.reverse(path);
639             paths.add(path);
640         }
641 
642         // Then look for the deepest node common to all those paths
643         int depth = 0;
644         Term externalHead = null;
645         outer: for (; depth < paths.get(0).size(); ++depth) {
646             final Term t = paths.get(0).get(depth);
647             for (int i = 1; i < paths.size(); ++i) {
648                 final List<Term> path = paths.get(i);
649                 if (depth >= path.size() || !path.get(depth).equals(t)) {
650                     break outer;
651                 }
652             }
653             externalHead = t;
654         }
655 
656         // If found, compute the terms for the external span
657         Set<Term> externalTerms = null;
658         if (externalHead != null) {
659             externalTerms = Sets.newHashSet(terms);
660             externalTerms.add(externalHead);
661             for (final List<Term> path : paths) {
662                 externalTerms.addAll(path.subList(depth, path.size()));
663             }
664         }
665 
666         // Now look for the internal head that covers the most part terms of the span. Start by
667         // associating to each candidate internal head the terms it would cover
668         final Multimap<Term, Term> map = HashMultimap.create();
669         for (final Term term : terms) {
670             Dep dep = document.getDepToTerm(term);
671             if (dep == null) {
672                 map.put(term, term);
673             } else {
674                 for (; dep != null; dep = document.getDepToTerm(dep.getFrom())) {
675                     if (!terms.contains(dep.getFrom())) {
676                         map.put(dep.getTo(), term);
677                         break;
678                     }
679                 }
680             }
681         }
682 
683         // Then identify the best internal head
684         Term internalHead = null;
685         Collection<Term> internalTerms = null;
686         for (final Map.Entry<Term, Collection<Term>> entry : map.asMap().entrySet()) {
687             if (internalHead == null || entry.getValue().size() >= internalTerms.size()) {
688                 internalTerms = entry.getValue();
689                 internalHead = entry.getKey();
690             }
691         }
692 
693         // Return either the external span (if defined) or the internal one, based on which one is
694         // most similar in size to the original span (if equal, prefer external one).
695         if (externalTerms != null
696                 && externalTerms.size() - terms.size() <= terms.size() - internalTerms.size()) {
697             return KAFDocument.newTermSpan(
698                     Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(externalTerms), externalHead);
699         } else {
700             return KAFDocument.newTermSpan(
701                     Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(internalTerms), internalHead);
702         }
703     }

705     public static List<Span<Term>> mergeSpans(final KAFDocument document,
706             final Iterable<Span<Term>> spans, final boolean canAddTerms) {
707 
708         // Build a map associating to each span head the other heads it is coordinated with
709         final Map<Term, List<Term>> extents = Maps.newHashMap();
710         final Map<Term, Set<Term>> clusters = Maps.newHashMap();
711         for (final Span<Term> span : spans) {
712             final Term head = extractHead(document, span);
713             clusters.put(head, Sets.newHashSet(head));
714             extents.put(head, span.getTargets());
715         }
716         for (final Term head : clusters.keySet()) {
717             for (Dep dep = document.getDepToTerm(head); dep != null
718                     && ("CONJ".equals(dep.getRfunc()) || "COORD".equals(dep.getRfunc())); dep = document
719                     .getDepToTerm(dep.getFrom())) {
720                 if (clusters.keySet().contains(dep.getFrom())) {
721                     clusters.get(head).add(dep.getFrom());
722                     clusters.get(dep.getFrom()).add(head);
723                 } else if ("CO".indexOf(dep.getFrom().getPos()) < 0) {
724                     break; // don't include intermediate terms that are not conjunctions or commas
725                 }
726             }
727         }
728 
729         // Create a span for each cluster of heads, including intermediate conjunctions
730         final List<Span<Term>> result = Lists.newArrayList();
731         while (!clusters.isEmpty()) {
732             final Set<Term> heads = clusters.values().iterator().next();
733             final Set<Term> terms = Sets.newHashSet();
734             Term spanHead = heads.iterator().next();
735             for (final Term head : heads) {
736                 clusters.remove(head);
737                 terms.addAll(extents.get(head));
738                 final List<Term> path = Lists.newArrayList();
739                 for (Dep dep = document.getDepToTerm(head); dep != null; dep = document
740                         .getDepToTerm(dep.getFrom())) {
741                     final Term term = dep.getFrom();
742                     path.add(term);
743                     if (heads.contains(term)) {
744                         terms.addAll(path);
745                         path.clear();
746                         spanHead = term;
747                     }
748                 }
749             }
750             List<Term> spanTerms = Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(terms);
751             if (canAddTerms) {
752                 final List<Term> docTerms = document.getTerms();
753                 spanTerms = Lists.newArrayList(docTerms.subList(
754                         docTerms.indexOf(spanTerms.get(0)),
755                         docTerms.indexOf(spanTerms.get(spanTerms.size() - 1)) + 1));
756             }
757             result.add(KAFDocument.newTermSpan(spanTerms, spanHead));
758         }
759         return result;
760     }

762     public static final List<Span<Term>> splitSpans(final KAFDocument document,
763             final Iterable<Span<Term>> spans) {
764 
765         // Identify all the heads taking coordination into consideration
766         final Set<Term> heads = Sets.newHashSet();
767         final Set<Term> terms = Sets.newHashSet();
768         for (final Span<Term> span : spans) {
769             final Term head = extractHead(document, span);
770             heads.add(head);
771             terms.addAll(span.getTargets());
772 
773             final List<Term> queue = Lists.newLinkedList();
774             queue.add(head);
775             while (!queue.isEmpty()) {
776                 final Term term = queue.remove(0);
777                 for (final Dep dep : document.getDepsFromTerm(term)) {
778                     final String func = dep.getRfunc();
779                     if ("COORD".equals(func) || "CONJ".equals(func)) {
780                         final Term t = dep.getTo();
781                         queue.add(t);
782                         if ("CC".equals(t.getMorphofeat())
783                                 || !Character.isLetter(t.getMorphofeat().charAt(0))) {
784                             heads.add(term);
785                         }
786                     }
787                 }
788             }
789         }
790 
791         // Build and return a span for each head
792         final Set<Term> excluded = document.getTermsByDepDescendants(heads);
793         final List<Span<Term>> result = Lists.newArrayList();
794         for (final Term head : heads) {
795             final Set<Term> extent = document.getTermsByDepAncestors(ImmutableSet.of(head));
796             extent.removeAll(excluded);
797             extent.add(head);
798             extent.retainAll(terms);
799             if (!extent.isEmpty()) {
800                 result.add(KAFDocument.newTermSpan(Ordering.from(Term.OFFSET_COMPARATOR)
801                         .sortedCopy(extent), head));
802             }
803         }
804         return result;
805     }

807     public static final List<Span<Term>> splitSpan(final KAFDocument document,
808             final Span<Term> span, final Iterable<Term> heads) {
809 
810         final Set<Term> excludedTerms = document.getTermsByDepDescendants(heads);
811         final List<Span<Term>> spans = Lists.newArrayList();
812         for (final Term head : heads) {
813             final Set<Term> terms = document.getTermsByDepAncestors(ImmutableSet.of(head));
814             terms.removeAll(excludedTerms);
815             terms.add(head);
816             terms.retainAll(span.getTargets());
817             if (!terms.isEmpty()) {
818                 spans.add(KAFDocument.newTermSpan(Ordering.from(Term.OFFSET_COMPARATOR)
819                         .sortedCopy(terms), head));
820             }
821         }
822         return spans;
823     }

875     public static String extendedPos(final KAFDocument document, final Term term) {
876         final String pos = term.getMorphofeat();
877         final String lemma = term.getLemma().toLowerCase();
878         if ("some".equals(lemma) || "many".equals(lemma) || "all".equals(lemma)
879                 || "few".equals(lemma) || "this".equals(lemma) || "these".equals(lemma)
880                 || "that".equals(lemma) || "those".equals(lemma)) {
881             final Dep dep = document.getDepToTerm(term);
882             if (dep == null || !"NMOD".equals(dep.getRfunc())) {
883                 return pos + "P"; // determiner (DT) or adj (JJ) used as demonstrative pronoun
884             }
885         }
886         return pos;
887     }

889     public static Boolean isActiveForm(final KAFDocument document, final Term term) {
890         final String word = term.getStr().toLowerCase();
891         final String pos = term.getMorphofeat();
892         if (!pos.startsWith("V")) {
893             return null;
894         }
895         if (word.equals("been") || !pos.equals("VBN")) {
896             return Boolean.TRUE;
897         }
898         return isActiveFormHelper(document, term);
899     }

901     private static Boolean isActiveFormHelper(final KAFDocument document, final Term term) {
902         final Dep dep = document.getDepToTerm(term);
903         if (dep == null) {
904             return Boolean.FALSE;
905         }
906         final Term parent = dep.getFrom();
907         final String word = parent.getStr().toLowerCase();
908         final String pos = parent.getMorphofeat();
909         if (pos.startsWith("NN")) {
910             return Boolean.FALSE;
911         }
912         if (word.matches("am|are|is|was|were|be|been|being")) {
913             return Boolean.FALSE;
914         }
915         if (word.matches("ha(ve|s|d|ving)")) {
916             return Boolean.TRUE;
917         }
918 
919         if (pos.matches("VBZ|VBD|VBP|MD")) {
920             return Boolean.FALSE;
921         }
922         return isActiveFormHelper(document, parent);
923     }

925     public static java.util.function.Predicate<Term> matchExtendedPos(final KAFDocument document,
926             final String... posPrefixes) {
927         return new java.util.function.Predicate<Term>() {
928 
929             @Override
930             public boolean test(final Term term) {
931                 final String pos = extendedPos(document, term);
932                 for (final String prefix : posPrefixes) {
933                     if (pos.startsWith(prefix)) {
934                         return true;
935                     }
936                 }
937                 return false;
938             }
939 
940         };
941     }

943     // extracts descendents that are consecutive with the supplied head
944     public static Set<Term> getTermsByDepAncestor(final KAFDocument document, final Term head,
945             final boolean consecutive) {
946         final Set<Term> descendants = document.getTermsByDepAncestors(ImmutableSet.of(head));
947         if (consecutive) {
948             final List<Term> sortedTerms = Ordering.from(Term.OFFSET_COMPARATOR).sortedCopy(
949                     descendants);
950             final int[] indexes = new int[sortedTerms.size()];
951             for (int i = 0; i < sortedTerms.size(); ++i) {
952                 indexes[i] = document.getTerms().indexOf(sortedTerms.get(i));
953             }
954             final int h = sortedTerms.indexOf(head);
955             boolean filtered = false;
956             for (int i = h + 1; i < indexes.length; ++i) {
957                 filtered |= indexes[i] > indexes[i - 1] + 1;
958                 if (filtered) {
959                     descendants.remove(sortedTerms.get(i));
960                 }
961             }
962             filtered = false;
963             for (int i = h - 1; i >= 0; --i) {
964                 filtered |= indexes[i] < indexes[i + 1] - 1;
965                 if (filtered) {
966                     descendants.remove(sortedTerms.get(i));
967                 }
968             }
969         }
970         return descendants;
971     }
