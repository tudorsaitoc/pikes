40  /**
41   * A digital representation of a resource.
42   * <p>
43   * A {@code Representation} object provides access to the binary or character representation of a
44   * resource, including associated (mutable) representation metadata ({@link #getMetadata()}) and
45   * the resource ID ( {@link #getResourceID()}).
46   * </p>
47   * <p>
48   * Representation data can be consumed either as a stream of bytes or as a stream of characters.
49   * Conversion from one stream to another is performed if necessary using the charset encoded by
50   * metadata attribute {@link NIE#MIME_TYPE}.
51   * </p>
52   * <p>
53   * A representation encapsulates an open {@code InputStream} or {@code Reader} that provide access
54   * to the representation data (see methods {@link #getInputStream()} and {@link #getReader()}),
55   * hence it is a {@code Closeable} object that MUST be closed after use. In addition to these
56   * methods, a number of {@code writeToXXX()} helper methods allow to consume the representation
57   * data in different ways:
58   * </p>
59   * <ul>
60   * <li>{@link #writeToByteArray()} returns a byte array with all the representation data;</li>
61   * <li>{@link #writeToString()} returns a string with all the representation characater data.</li>
62   * <li>{@link #writeTo(OutputStream)} writes all the binary representation data to a supplied
63   * {@code OutputStream};</li>
64   * <li>{@link #writeTo(Appendable)} writes the character representation data to a supplied
65   * {@code Appendable};</li>
66   * </ul>
67   * <p>
68   * Note that these methods exhaust the {@code InputStream} / {@code Reader} associated to the
69   * representation. Moreover, in case some data has already been read, it will not be written by
70   * those methods.
71   * </p>
72   * <p>
73   * Representation objects are created via {@code create()} factory methods that take care of
74   * acquiring both an {@code InputStream} / {@code Reader} over the data and the associated
75   * metadata starting from a number of sources:
76   * </p>
77   * <ul>
78   * <li>{@link #create(URI, InputStream)} builds a representation out of an {@code InputStream};</li>
79   * <li>{@link #create(URI, Reader)} builds a representation out of a {@code Reader} ;</li>
80   * <li>{@link #create(URI, byte[])} builds a representation out of a byte array, including
81   * metadata about the representation length;</li>
82   * <li>{@link #create(URI, CharSequence)} builds a representation out of a {@code CharSequence}
83   * (e.g., a {@code String});</li>
84   * <li>{@link #create(URI, File)} builds a representation out of a file, including metadata about
85   * the file name, size, mime type (from the extension) and last modified time;</li>
86   * <li>{@link #create(URI, URL)} builds a representation out of a resolvable URL, including
87   * metadata about the file name, file size, mime type (from the extension), MD5 hash, last
88   * modified time.</li>
89   * </ul>
90   * <p>
91   * Representation objects are mutable but thread safe. Object equality is used for
92   * {@code equals()} and {@code hashCode()}.
93   * </p>
94   */
95  public final class Representation implements Closeable {

108     private Charset getCharset() {
109         final String mimeType = this.metadata.getUnique(NIE.MIME_TYPE, String.class);
110         if (mimeType == null) {
111             return Charsets.UTF_8;
112         }
113         try {
114             return MediaType.parse(mimeType).charset().or(Charsets.UTF_8);
115         } catch (final Throwable ex) {
116             throw new IllegalArgumentException("Invalid mime type in metadata: " + mimeType, ex);
117         }
118     }

158     /**
159      * Creates a representation based on the {@code File} specified. The file length, size,
160      * creation time and MIME type will be reflected in the returned representation metadata
161      * (respectively, properties {@link NFO#FILE_SIZE}, {@link NFO#FILE_NAME},
162      * {@link NFO#FILE_LAST_MODIFIED}, {@link NIE#MIME_TYPE}). Note that this method causes the
163      * file to be opened for reading.
164      *
165      * @param file
166      *            the file containing the binary data of the representation
167      * @param autoDecompress
168      *            automatically decompress the file, if compressed with gzip, bzip2, xz, 7z or lz4
169      * @return the created representation
170      * @throws IllegalArgumentException
171      *             in case the file does not exist
172      */
173     public static Representation create(final File file, final boolean autoDecompress)
174             throws IllegalArgumentException {
175         try {
176             String name = file.getName();
177             final Representation representation;
178             if (autoDecompress) {
179                 representation = new Representation(IO.buffer(IO.read(file.getAbsolutePath())));
180                 if (name.endsWith(".gz") || name.endsWith(".xz") || name.endsWith(".7z")) {
181                     name = name.substring(0, name.length() - 3);
182                 } else if (name.endsWith(".bz2") || name.endsWith(".lz4")) {
183                     name = name.substring(0, name.length() - 4);
184                 }
185             } else {
186                 representation = new Representation(IO.buffer(new FileInputStream(file)));
187             }
188             representation.metadata.set(NFO.FILE_SIZE, file.length());
189             representation.metadata.set(NFO.FILE_NAME, name);
190             representation.metadata.set(NFO.FILE_LAST_MODIFIED, new Date(file.lastModified()));
191             representation.metadata.set(NIE.MIME_TYPE, Data.extensionToMimeType(name));
192             return representation;
193         } catch (final FileNotFoundException ex) {
194             throw new IllegalArgumentException("Not a file: " + file.getAbsolutePath());
195         } catch (final IOException e) {
196             throw new IllegalArgumentException("IOException on file: " + file.getAbsolutePath());
197         }
198     }

200     /**
201      * Creates a representation based on the resolvable URL specified. This method has the effect
202      * of acquiring a connection to the supplied URL, from which the representation stream and a
203      * number of metadata attributes are extracted. These attributes include the last modified
204      * timestamp ({@link NFO#FILE_LAST_MODIFIED}), the MIME type ({@link NIE#MIME_TYPE}), the file
205      * size ({@link NFO#FILE_SIZE}), the file name ({@link NFO#FILE_NAME}) and the MD5 hash (
206      * {@link NFO#HAS_HASH}); all of these attributes are optional and are extracted only if
207      * available.
208      *
209      * @param url
210      *            the URL that, resolved, will produced the binary data of the representation
211      * @return the created representation
212      * @throws IllegalArgumentException
213      *             in case acquiring a connection to the supplied URL fails
214      */
215     public static Representation create(final URL url) throws IllegalArgumentException {
216 
217         // Acquire a connection and open an InputStream over its entity content.
218         URLConnection connection;
219         InputStream stream;
220         try {
221             connection = url.openConnection();
222             connection.connect();
223             stream = connection.getInputStream();
224         } catch (final IOException ex) {
225             throw new IllegalArgumentException("Cannot acquire a connection to URL " + url, ex);
226         }
227 
228         // Wrap the stream in a Representation object.
229         final Representation representation = new Representation(stream);
230 
231         try {
232             // Extract last modified.
233             final long lastModified = connection.getLastModified();
234             if (lastModified != 0) {
235                 representation.metadata.set(NFO.FILE_LAST_MODIFIED, new Date(lastModified));
236             }
237 
238             // Extract MIME type from "Content-Type".
239             String mimeType = connection.getContentType();
240             if (mimeType == null) {
241                 mimeType = Data.extensionToMimeType(url.getFile());
242             }
243             representation.metadata.set(NIE.MIME_TYPE, mimeType);
244 
245             // Extract length from "Content-Length";
246             final int length = connection.getContentLength();
247             if (length >= 0) {
248                 representation.metadata.set(NFO.FILE_SIZE, length);
249             }
250 
251             // Extract the filename either from "Content-Disposition" header or from URL.
252             String filename = null;
253             final String disposition = connection.getHeaderField("Content-Disposition");
254             if (disposition != null && disposition.contains("filename")) {
255                 final int start = Math.max(disposition.indexOf('\"'), disposition.indexOf('\''));
256                 if (start > 0) {
257                     final int end = Math.max(disposition.lastIndexOf('\"'),
258                             disposition.lastIndexOf('\''));
259                     if (end > 0) {
260                         filename = disposition.substring(start + 1, end);
261                     }
262                 }
263             }
264             if (filename == null) {
265                 final String path = url.getPath();
266                 final int index = path.lastIndexOf('/');
267                 if (index >= 0) {
268                     filename = path.substring(index + 1);
269                 }
270 
271             }
272             representation.metadata.set(NFO.FILE_NAME, filename);
273 
274             // Extract the MD5 hash from "Content-MD5".
275             final String md5 = connection.getHeaderField("Content-MD5");
276             if (md5 != null) {
277                 final Record hash = Record.create();
278                 hash.set(NFO.HASH_ALGORITHM, "MD5");
279                 hash.set(NFO.HASH_VALUE, md5);
280                 representation.metadata.set(NFO.HAS_HASH, hash);
281             }
282 
283             // Return the representation built.
284             return representation;
285 
286         } catch (final Throwable ex) {
287             // Ensure to close the connection if something goes wrong.
288             try {
289                 connection.getInputStream().close();
290             } catch (final Throwable ex2) {
291                 // ignore
292             }
293             throw Throwables.propagate(ex);
294         }
295     }

467     @Override
468     public String toString() {
469         final String file = this.metadata.getUnique(NFO.FILE_NAME, String.class, "unnamed file");
470         final String type = this.metadata.getUnique(NIE.MIME_TYPE, String.class, "unknown type");
471         final long size = this.metadata.getUnique(NFO.FILE_SIZE, Long.class, -1L);
472         return file + ", " + type + ", " + (size >= 0 ? size + " bytes" : "unknown size");
473     }

475     // Source: org.apache.commons.io.input.ReaderInputStream
476     private class ReaderInputStream extends InputStream {
477 
478         private static final int BUFFER_SIZE = 1024;
479 
480         private final Reader reader;
481 
482         private final CharsetEncoder enc;
483 
484         private final CharBuffer encIn;
485 
486         private final ByteBuffer encOut;
487 
488         private CoderResult lastCoderResult;
489 
490         private boolean eof;
491 
492         ReaderInputStream(final Reader reader, final Charset charset) {
493 
494             this.reader = reader;
495             this.enc = charset.newEncoder().onMalformedInput(CodingErrorAction.REPLACE)
496                     .onUnmappableCharacter(CodingErrorAction.REPLACE);
497             this.encIn = CharBuffer.allocate(BUFFER_SIZE);
498             this.encIn.flip();
499             this.encOut = ByteBuffer.allocate(128);
500             this.encOut.flip();
501         }
502 
503         private void fillBuffer() throws IOException {
504 
505             if (!this.eof && (this.lastCoderResult == null || this.lastCoderResult.isUnderflow())) {
506                 this.encIn.compact();
507                 final int p = this.encIn.position();
508                 final int c = this.reader.read(this.encIn.array(), p, this.encIn.remaining());
509                 if (c == -1) {
510                     this.eof = true;
511                 } else {
512                     this.encIn.position(p + c);
513                 }
514                 this.encIn.flip();
515             }
516 
517             this.encOut.compact();
518             this.lastCoderResult = this.enc.encode(this.encIn, this.encOut, this.eof);
519             this.encOut.flip();
520         }
521 
522         @Override
523         public int read(final byte[] b, final int offset, final int length) throws IOException {
524 
525             Preconditions.checkNotNull(b);
526             Preconditions.checkPositionIndex(offset, b.length);
527             Preconditions.checkPositionIndex(offset + length, b.length);
528 
529             int read = 0;
530             int o = offset;
531             int l = length;
532 
533             while (l > 0) {
534                 if (this.encOut.hasRemaining()) {
535                     final int c = Math.min(this.encOut.remaining(), l);
536                     this.encOut.get(b, o, c);
537                     o += c;
538                     l -= c;
539                     read += c;
540                 } else {
541                     fillBuffer();
542                     if (this.eof && !this.encOut.hasRemaining()) {
543                         break;
544                     }
545                 }
546             }
547 
548             return read > 0 || !this.eof ? read : l == 0 ? 0 : -1;
549         }
550 
551         @Override
552         public int read() throws IOException {
553 
554             for (;;) {
555                 if (this.encOut.hasRemaining()) {
556                     return this.encOut.get() & 0xFF;
557                 } else {
558                     fillBuffer();
559                     if (this.eof && !this.encOut.hasRemaining()) {
560                         return -1;
561                     }
562                 }
563             }
564         }
565 
566         @Override
567         public void close() throws IOException {
568             this.reader.close();
569         }
