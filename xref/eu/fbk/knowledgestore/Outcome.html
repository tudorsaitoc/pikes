21  /**
22   * The outcome of the invocation of a KnowledgeStore operation.
23   * <p>
24   * An {@code Outcome} instance encodes the outcome of an {@code Operation} invocation. It stores:
25   * </p>
26   * <ul>
27   * <li>the {@link Status} code specifying whether and how the invocation was successful or failed
28   * (see {@link #getStatus()});</li>
29   * <li>the invocation ID, which is generated by the system and may link to additional information
30   * logged on client/server sides (see {@link #getInvocationID()});</li>
31   * <li>an optional object ID, in case the invocation was applied to a specific object (see
32   * {@link #getObjectID()});</li>
33   * <li>an optional message, providing additional information about the outcome (see
34   * {@link #getMessage()})</li>
35   * </ul>
36   * <p>
37   * {@code Outcome} instances can be created using one of the static factory methods
38   * {@link #create(Status, URI)}, {@link #create(Status, URI, URI)},
39   * {@link #create(Status, URI, URI, String)}. Equality and comparison are performed based on the
40   * invocation ID. This class is immutable and thus thread safe.
41   * </p>
42   */
43  public final class Outcome implements Comparable<Outcome>, Serializable {
44  
45      private static final long serialVersionUID = 1L;
46  
47      private final Status status;
48  
49      private final URI invocationID;
50  
51      @Nullable
52      private final URI objectID;
53  
54      @Nullable
55      private final String message;
56  
57      private Outcome(final Status status, final URI invocationID, @Nullable final URI objectID,
58              @Nullable final String message) {
59          this.status = Preconditions.checkNotNull(status);
60          this.invocationID = Preconditions.checkNotNull(invocationID);
61          this.objectID = objectID;
62          this.message = message;
63      }

65      /**

72       * @return the created {@code Outcome}
73       */

75          return create(status, invocationID, null, null);
76      }

86       * @param objectID
87       *            the optional object ID, possibly null
88       * @return the created {@code Outcome}

91              @Nullable final URI objectID) {
92          return create(status, invocationID, objectID, null);
93      }

95      /**
96       * Creates a new {@code Outcome} with the status code, invocation ID, optional object ID and
97       * optional message supplied.
98       * 
99       * @param status
100      *            the status code, not null
101      * @param invocationID
102      *            the invocation ID, not null
103      * @param objectID
104      *            the optional object ID, possibly null
105      * @param message
106      *            the optional message, possibly null
107      * @return the created {@code Outcome}
108      */
109     public static Outcome create(final Status status, final URI invocationID,
110             @Nullable final URI objectID, @Nullable final String message) {
111         return new Outcome(status, invocationID, objectID, message);
112     }

114     /**
115      * Creates a new {@code Outcome} starting from the {@code Record} supplied. The record ID is
116      * used as the invocation ID, while properties {@link KSR#STATUS}, {@link KSR#OBJECT} and
117      * {@link KSR#MESSAGE} are read, respectively, to recover the status code, the optional object
118      * ID and the optional message.
119      * 
120      * @param record
121      *            the record
122      * @return the created {@code Outcome}
123      */
124     public static Outcome create(final Record record) {
125         final URI invocationID = record.getID();
126         final Status status = Status.valueOf(record.getUnique(KSR.STATUS, URI.class));
127         final URI objectID = record.getUnique(KSR.OBJECT, URI.class);
128         final String message = record.getUnique(KSR.MESSAGE, String.class);
129         return create(status, invocationID, objectID, message);
130     }

138      */

172     /**
173      * {@inheritDoc} Comparison is done on the invocation ID.
174 	 * @param other
175      */
176     @Override
177     public int compareTo(final Outcome other) {
178         return Data.getTotalComparator().compare(this.invocationID, other.invocationID);

181     /**
182      * {@inheritDoc} Two {@code Outcome} instances are equal if the have the same invocation ID.
183      */
184     @Override
185     public boolean equals(@Nullable final Object object) {
186         if (object == this) {
187             return true;
188         }
189         if (!(object instanceof Outcome)) {
190             return false;
191         }
192         final Outcome other = (Outcome) object;
193         return this.invocationID.equals(other.invocationID);
194     }

196     /**
197      * {@inheritDoc} The returned code depends on the invocation ID.
198      */
199     @Override
200     public int hashCode() {
201         return this.invocationID.hashCode();
202     }

204     /**
205      * Return a {@code Record} version of this {@code Outcome} object. The returned record is
206      * identified by this invocation ID; it is associated to the status URI via property
207      * {@link KSR#STATUS}, to the optional object ID via property {@link KSR#OBJECT} and to the
208      * optional message via property {@link KSR#MESSAGE}.
209      * 
210      * @return the corresponding record
211      */
212     public Record toRecord() {
213         final Record record = Record.create(this.invocationID);
214         record.add(KSR.STATUS, this.status.getURI());
215         if (this.objectID != null) {
216             record.add(KSR.OBJECT, this.objectID);
217         }
218         if (this.message != null) {
219             record.add(KSR.MESSAGE, Data.getValueFactory().createLiteral(this.message));
220         }
221         return record;
222     }
223 
224     /**
225      * {@inheritDoc} The method returns a string of the form
226      * {@code status invocationID (objectID) message}.
227      */
228     @Override
229     public String toString() {
230         final StringBuilder builder = new StringBuilder();
231         builder.append(this.status);
232         builder.append(' ');
233         builder.append(Data.toString(this.invocationID, Data.getNamespaceMap()));
234         if (this.objectID != null) {
235             builder.append(' ');
236             builder.append('(');
237             builder.append(Data.toString(this.objectID, Data.getNamespaceMap()));
238             builder.append(')');
239         }
240         if (this.message != null) {
241             builder.append(' ');
242             builder.append(this.message);
243         }
244         return builder.toString();
245     }

247     /**
248      * Performs outcome-to-RDF encoding by converting a stream of outcomes in a stream of RDF
249      * statements.
250      * 
251      * @param stream
252      *            the stream of outcomes to encode.
253      * @return the resulting stream of statements
254      */
255     public static Stream<Statement> encode(final Stream<? extends Outcome> stream) {
256         Preconditions.checkNotNull(stream);
257         return Record.encode(stream.transform(new Function<Outcome, Record>() {
258 
259             @Override
260             public Record apply(final Outcome outcome) {
261                 return outcome.toRecord();
262             }
263 
264         }, 0), ImmutableSet.of(KSR.INVOCATION));
265     }
266 
267     /**
268      * Performs RDF-to-outcome decoding by converting a stream of RDF statements in a stream of
269      * outcomes. Parameter {@code chunked} specifies whether the input statement stream is
270      * chunked, i.e., organized as a sequence of statement chunks with each chunk containing the
271      * statements for an outcome. Chunked RDF streams noticeably speed up decoding, and are always
272      * produced by the KnowledgeStore API. Chunking information may be set to null (e.g., because
273      * unknown at the time the method is called): in this case, it will be read from metadata
274      * attribute {@code "chunked"} attached to the stream; reading will happen just before
275      * decoding will take place, i.e., when a terminal stream operation will be called.
276      * 
277      * @param stream
278      *            the stream of statements to decode
279      * @param chunked
280      *            true if the input statement stream is chunked, null if to be read from stream
281      *            metadata
282      * @return the resulting stream of outcomes
283      */
284     public static Stream<Outcome> decode(final Stream<Statement> stream,
285             @Nullable final Boolean chunked) {
286         Preconditions.checkNotNull(stream);
287         return Record.decode(stream, ImmutableSet.of(KSR.INVOCATION), chunked).transform(
288                 new Function<Record, Outcome>() {
289 
290                     @Override
291                     public Outcome apply(final Record record) {
292                         return Outcome.create(record);
293                     }
294 
295                 }, 0);
296     }

298     /**
299      * Enumeration of {@code Outcome} status codes.
300      * <p>
301      * This enumeration lists the possible status codes for {@code Outcome} objects. A status code
302      * is identified by an URI (see {@link #getURI()}), is briefly explained through a comment
303      * string (see {@link #getComment()}) and may be mapped to an HTTP status code (see
304      * {@link #getHTTPStatus()}). Methods {@link #isOK()}, {@link #isError()} determine,
305      * respectively, if the status code denotes a success or error situation. Lookup of status
306      * codes based on URI is supported by method {@link #valueOf(URI)}.
307      * </p>
308      */
309     public enum Status {
310 
311         /**
312          * Success status specifying successful completion of a bulk operation invocation for all
313          * the objects involved.
314          * 
315          * @see KSR#OK_BULK
316          */
317         OK_BULK(KSR.OK_BULK, "Bulk operation succeeded for all affected objects", 200, true),
318 
319         /**
320          * Success status specifying that an object has been created.
321          * 
322          * @see KSR#OK_CREATED
323          */
324         OK_CREATED(KSR.OK_CREATED, "Object created", 201, true),
325 
326         /**
327          * Success status specifying that an object has been modified.
328          * 
329          * @see KSR#OK_MODIFIED
330          */
331         OK_MODIFIED(KSR.OK_MODIFIED, "Object modified", 200, true),
332 
333         /**
334          * Success status specifying that it was not necessary to modify an object.
335          * 
336          * @see KSR#OK_UNMODIFIED
337          */
338         OK_UNMODIFIED(KSR.OK_UNMODIFIED, "Object not modified", 200, true),
339 
340         /**
341          * Success status specifying that an object has been deleted.
342          * 
343          * @see KSR#OK_DELETED
344          */
345         OK_DELETED(KSR.OK_DELETED, "Object deleted", 200, true),
346 
347         /**
348          * Error status specifying that a bulk operation invocation failed for one or more of the
349          * involved objects.
350          * 
351          * @see KSR#ERROR_BULK
352          */
353         ERROR_BULK(KSR.ERROR_BULK, "Bulk operation failed for at least one affected object", 200,
354                 false),
355 
356         /**
357          * Error status specifying that an operation invocation failed as its result would not be
358          * acceptable to the client.
359          * 
360          * @see KSR#ERROR_PRECONDITION_FAILED
361          */
362         ERROR_NOT_ACCEPTABLE(KSR.ERROR_NOT_ACCEPTABLE,
363                 "Operation failed as result would not be acceptable to client", 406, false),
364 
365         /**
366          * Error status specifying that the referenced object does not exist.
367          * 
368          * @see KSR#ERROR_OBJECT_NOT_FOUND
369          */
370         ERROR_OBJECT_NOT_FOUND(KSR.ERROR_OBJECT_NOT_FOUND,
371                 "Operation failed as target object does not exist", 404, false),
372 
373         /**
374          * Error status specifying that the referenced object already exists.
375          * 
376          * @see KSR#ERROR_OBJECT_ALREADY_EXISTS
377          */
378         ERROR_OBJECT_ALREADY_EXISTS(KSR.ERROR_OBJECT_ALREADY_EXISTS,
379                 "Operation failed as target object already exists", 409, false),
380 
381         /**
382          * Error status specifying that an object indirectly referenced by the operation
383          * invocation does not exist.
384          * 
385          * @see KSR#ERROR_DEPENDENCY_NOT_FOUND
386          */
387         ERROR_DEPENDENCY_NOT_FOUND(KSR.ERROR_DEPENDENCY_NOT_FOUND,
388                 "Operation failed as object it depends on does not exist", 400, false),
389 
390         /**
391          * Error status specifying that input arguments are missing or wrong.
392          * 
393          * @see KSR#ERROR_INVALID_INPUT
394          */
395         ERROR_INVALID_INPUT(KSR.ERROR_INVALID_INPUT,
396                 "Operation failed as required input arguments are missing or wrong", 400, false),
397 
398         /**
399          * Error status specifying that the invoked operation has not been executed because the
400          * requester has not enough privileges.
401          * 
402          * @see KSR#ERROR_FORBIDDEN
403          */
404         ERROR_FORBIDDEN(KSR.ERROR_FORBIDDEN, "Operation forbidden", 403, false),
405 
406         /**
407          * Error status specifying that the invoked operation was forcedly interrupted by the
408          * client, server or due to a connectivity problem.
409          * 
410          * @see KSR#ERROR_INTERRUPTED
411          */
412         ERROR_INTERRUPTED(KSR.ERROR_INTERRUPTED, "Operation interrupted", 503, false),
413 
414         /**
415          * Error status specifying that the invocation failed due to an unexpected error.
416          * 
417          * @see KSR#ERROR_UNEXPECTED
418          */
419         ERROR_UNEXPECTED(KSR.ERROR_UNEXPECTED, "Unexpected error", 500, false),
420 
421         /**
422          * Error status specifying that the outcome of the invoked operation is unknown.
423          * 
424          * @see KSR#UNKNOWN
425          */
426         UNKNOWN(KSR.UNKNOWN, "Unknown outcome", 503, false);

441         private Status(final URI uri, final String comment, final int httpStatus,
442                 @Nullable final boolean okOrError) {

486         }

493         public boolean isError() {

495         }

497         /**
498          * Lookups the status code with the URI specified.
499          * 
500          * @param uri
501          *            the URI of the status code to lookup
502          * @return the corresponding status code, on success
503          * @throws IllegalArgumentException
504          *             if there is no status for the URI specified
505          */
506         public static Status valueOf(final URI uri) throws IllegalArgumentException {
507             if (uriToStatusMap == null) {
508                 final ImmutableMap.Builder<URI, Status> builder = ImmutableMap.builder();
509                 for (final Status status : Status.values()) {
510                     builder.put(status.uri, status);
511                 }
512                 uriToStatusMap = builder.build();
513             }
514             final Status status = uriToStatusMap.get(uri);
515             if (status == null) {
516                 throw new IllegalArgumentException("Invalid status URI: " + uri);
517             }
518             return status;
519         }
520 
521         /**
522          * Return the {@code Status} that more closely matches the HTTP status code specified.
523          * Note that only part of the statuses of this enuemration are returned, as the mapping
524          * from {@code Status} to HTTP statuses is N:1.
525          * 
526          * @param httpStatus
527          *            the HTTP status
528          * @return the corresponding {@code Status}, defaulting to {@link #ERROR_UNEXPECTED}
529          */
530         public static Status valueOf(final int httpStatus) {
531             if (httpStatus == 400) {
532                 return Status.ERROR_INVALID_INPUT;
533             } else if (httpStatus == 401 || httpStatus == 403) {
534                 return Status.ERROR_FORBIDDEN;
535             } else if (httpStatus == 404) {
536                 return Status.ERROR_OBJECT_NOT_FOUND;
537             } else if (httpStatus == 406) {
538                 return Status.ERROR_NOT_ACCEPTABLE;
539             }
540             return Outcome.Status.ERROR_UNEXPECTED;
541         }
