138     public static String pathFor(final URI type) {
139         if (KS.RESOURCE.equals(type)) {
140             return PATH_RESOURCES;
141         } else if (KS.MENTION.equals(type)) {
142             return PATH_MENTIONS;
143         } else if (KS.ENTITY.equals(type)) {
144             return PATH_ENTITIES;
145         } else if (KS.AXIOM.equals(type)) {
146             return PATH_AXIOMS;
147         } else if (KS.REPRESENTATION.equals(type)) {
148             return PATH_REPRESENTATIONS;
149         }
150         throw new IllegalArgumentException("Invalid type " + type);
151     }
152 
153     public static URI typeFor(final String path) {
154         if (path.equals(PATH_RESOURCES)) {
155             return KS.RESOURCE;
156         } else if (path.equals(PATH_MENTIONS)) {
157             return KS.MENTION;
158         } else if (path.equals(PATH_ENTITIES)) {
159             return KS.ENTITY;
160         } else if (path.equals(PATH_AXIOMS)) {
161             return KS.AXIOM;
162         } else if (path.equals(PATH_REPRESENTATIONS)) {
163             return KS.REPRESENTATION;
164         }
165         throw new IllegalArgumentException("Invalid path " + path);
166     }
