            }

        linkTrainer.add(document, expressionHead, heads);

        // Add samples for span labelling (one for each argument)
        for (int i = 0; i < heads.size(); ++i) {
            final List<Term> excludedTerms = Lists.newArrayList(heads);

                        url.toString());
