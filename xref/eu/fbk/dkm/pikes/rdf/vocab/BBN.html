79      // UTILITY METHODS
80  
81      @Nullable
82      public static URI resolve(@Nullable final String bbnString) {
83          if (bbnString == null) {
84              return null;
85          }
86          return INDEX.get(bbnString.toLowerCase().replaceAll("[^a-z]+", ""));
87      }
88  
89      private static URI createURI(final String localName) {
90          return ValueFactoryImpl.getInstance().createURI(NAMESPACE, localName);
91      }
