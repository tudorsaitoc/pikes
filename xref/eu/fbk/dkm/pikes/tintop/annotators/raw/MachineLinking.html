19  public class MachineLinking extends Linking {
20  
21      private static String LABEL = "ml-annotate";
22      private Double minWeight;
23  
24      public MachineLinking(Properties properties) {
25          super(properties, properties.getProperty("address"));
26          minWeight = Defaults.getDouble(properties.getProperty("min_confidence"), Defaults.ML_CONFIDENCE);
27      }
28  
29      @Override
30      public List<LinkingTag> tag(String text) throws Exception {
31  
32          ArrayList<LinkingTag> ret = new ArrayList<>();
33          Map<String, String> pars;
34  
35          pars = new HashMap<>();
36          pars.put("min_weight", minWeight.toString());
37          pars.put("disambiguation", "1");
38          pars.put("topic", "1");
39          pars.put("include_text", "0");
40          pars.put("image", "0");
41          pars.put("class", "1");
42          pars.put("app_id", "0");
43          pars.put("app_key", "0");
44          pars.put("text", text);
45  
46          LOGGER.debug("Text length: {}", text.length());
47          LOGGER.debug("Pars: {}", pars);
48  
49          Map<String, Object> userData;
50          String output = request(pars);
51  
52          ObjectMapper mapper = new ObjectMapper();
53          userData = mapper.readValue(output, Map.class);
54  
55          LinkedHashMap annotation = (LinkedHashMap) userData.get(new String("annotation"));
56          if (annotation != null) {
57              ArrayList<LinkedHashMap> keywords = (ArrayList<LinkedHashMap>) annotation.get(new String("keyword"));
58              if (keywords != null) {
59                  for (LinkedHashMap keyword : keywords) {
60                      LinkedHashMap sense = (LinkedHashMap) keyword.get("sense");
61                      ArrayList dbpClass = (ArrayList) keyword.get("class");
62                      ArrayList<LinkedHashMap> ngrams = (ArrayList<LinkedHashMap>) keyword.get("ngram");
63                      for (LinkedHashMap ngram : ngrams) {
64                          String originalText = (String) ngram.get("form");
65                          LinkedHashMap span = (LinkedHashMap) ngram.get("span");
66  
67                          Integer start = (Integer) span.get("start");
68                          Integer end = (Integer) span.get("end");
69  
70                          LinkingTag tag = new LinkingTag(
71                                  start,
72                                  String.format("http://dbpedia.org/resource/%s", (String) sense.get("page")),
73                                  Double.parseDouble(keyword.get("rel").toString()),
74                                  originalText,
75                                  end - start,
76                                  LABEL
77                          );
78                          if (extractTypes) {
79                              tag.addTypesFromML(dbpClass);
80                          }
81                          ret.add(tag);
82                      }
83                  }
84              }
85          }
86  
87          return ret;
88      }
89  
90      public static void main(String[] args) {
91          Properties properties = new Properties();
92          properties.setProperty("address", "http://ml.apnetwork.it/annotate");
93          properties.setProperty("min_confidence", "0.5");
94          properties.setProperty("timeout", "2000");
95  
96          String fileName = args[0];
97  
98          MachineLinking s = new MachineLinking(properties);
99          try {
100             String text = Files.toString(new File(fileName), Charsets.UTF_8);
101             List<LinkingTag> tags = s.tag(text);
102             for (LinkingTag tag : tags) {
103                 System.out.println(tag);
104             }
105         } catch (Exception e) {
106             e.printStackTrace();
107             LOGGER.error(e.getMessage());
108         }
109     }
