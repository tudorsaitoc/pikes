1   package eu.fbk.dkm.pikes.tintop.annotators.models;
2   
3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   import se.lth.cs.srl.SemanticRoleLabeler;
6   import se.lth.cs.srl.languages.Language;
7   import se.lth.cs.srl.pipeline.Pipeline;
8   
9   import java.io.File;
10  import java.util.zip.ZipFile;
11  
12  /**
13   * Created by alessio on 25/05/15.
14   */
15  
16  public class MateSrlModel {
17  
18  	private static MateSrlModel instance;
19  	private SemanticRoleLabeler labeler;
20  	private static final Logger LOGGER = LoggerFactory.getLogger(MateSrlModel.class);
21  
22  	private MateSrlModel(File mateModel) {
23  		LOGGER.info("Loading model for Mate Tools");
24  
25  		try {
26  			ZipFile zipFile;
27  			zipFile = new ZipFile(mateModel);
28  			labeler = Pipeline.fromZipFile(zipFile);
29  			zipFile.close();
30  		} catch (Exception e) {
31  			LOGGER.error(e.getMessage());
32  		}
33  
34  		Language.setLanguage(Language.L.valueOf("eng"));
35  	}
36  
37  	public static MateSrlModel getInstance(File posModel) {
38  		if (instance == null) {
39  			instance = new MateSrlModel(posModel);
40  		}
41  
42  		return instance;
43  	}
44  
45  	public SemanticRoleLabeler getLabeler() {
46  		return labeler;
47  	}
48  }
