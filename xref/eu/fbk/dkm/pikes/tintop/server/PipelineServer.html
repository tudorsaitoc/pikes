25  public class PipelineServer {
26  
27      private static final org.slf4j.Logger logger = LoggerFactory.getLogger(PipelineServer.class);
28  
29      public static final String DEFAULT_HOST = "localhost";
30      public static final Integer DEFAULT_PORT = 8011;
31  
32      public PipelineServer(String host, Integer port) {
33          this(host, port, null, null);
34      }
35  
36      public PipelineServer(String host, Integer port, @Nullable File configFile) {
37          this(host, port, configFile, null);
38      }
39  
40      public PipelineServer(String host, Integer port, @Nullable File configFile,
41              @Nullable Properties additionalProperties) {
42          logger.info("starting " + host + "\t" + port + " (" + new Date() + ")...");
43  
44          AnnotationPipeline pipeline = null;
45          try {
46              pipeline = new AnnotationPipeline(configFile, additionalProperties);
47              pipeline.loadModels();
48          } catch (Exception e) {
49              e.printStackTrace();
50              logger.error(e.getMessage());
51              System.exit(1);
52          }
53  
54          int timeoutInSeconds = -1;
55          try {
56              if (pipeline.getDefaultConfig().getProperty("timeout") != null) {
57                  timeoutInSeconds = Integer.parseInt(pipeline.getDefaultConfig().getProperty("timeout"));
58                  logger.info("Timeout set to: " + timeoutInSeconds);
59              }
60          } catch (Exception e) {
61              e.printStackTrace();
62          }
63  
64          HttpServer httpServer = HttpServer.createSimpleServer(host, port);
65          httpServer.getServerConfiguration().setSessionTimeoutSeconds(timeoutInSeconds);
66          httpServer.getServerConfiguration().setMaxPostSize(4194304);
67          httpServer.getServerConfiguration().addHttpHandler(new NafHandler(pipeline), "/naf");
68          httpServer.getServerConfiguration().addHttpHandler(new NafVisualizeHandler(pipeline), "/view");
69          httpServer.getServerConfiguration().addHttpHandler(new NafGenerateHandler(pipeline), "/text");
70          httpServer.getServerConfiguration().addHttpHandler(new EverythingHandler(pipeline), "/all");
71          httpServer.getServerConfiguration().addHttpHandler(new Text2NafHandler(pipeline), "/text2naf");
72          httpServer.getServerConfiguration().addHttpHandler(new TriplesHandler(pipeline), "/text2rdf");
73  
74          try {
75              httpServer.start();
76              Thread.currentThread().join();
77          } catch (Exception e) {
78              logger.error("error running " + host + ":" + port);
79          }
80      }
81  
82      public static void main(String[] args) {
83  
84          try {
85              final CommandLine cmd = CommandLine
86                      .parser()
87                      .withName("./tintop-server")
88                      .withHeader("Run the Tintop Server")
89                      .withOption("c", "config", "Configuration file", "FILE", CommandLine.Type.FILE_EXISTING, true,
90                              false, false)
91                      .withOption("p", "port", String.format("Host port (default %d)", DEFAULT_PORT), "NUM",
92                              CommandLine.Type.INTEGER, true, false, false)
93                      .withOption("h", "host", String.format("Host address (default %s)", DEFAULT_HOST), "NUM",
94                              CommandLine.Type.STRING, true, false, false)
95                      .withOption(null, "properties", "Additional properties", "PROPS", CommandLine.Type.STRING, true,
96                              true, false)
97                      .withLogger(LoggerFactory.getLogger("eu.fbk")).parse(args);
98  
99              String host = cmd.getOptionValue("host", String.class, DEFAULT_HOST);
100             Integer port = cmd.getOptionValue("port", Integer.class, DEFAULT_PORT);
101             File configFile = cmd.getOptionValue("config", File.class);
102 
103             List<String> addProperties = cmd.getOptionValues("properties", String.class);
104 
105             Properties additionalProps = new Properties();
106             for (String property : addProperties) {
107                 try {
108                     additionalProps.load(new StringReader(property));
109                 } catch (Exception e) {
110                     logger.warn(e.getMessage());
111                 }
112             }
113 
114             PipelineServer pipelineServer = new PipelineServer(host, port, configFile, additionalProps);
115 
116         } catch (Exception e) {
117             CommandLine.fail(e);
118         }
119 
120     }
