32      private static final Logger LOGGER = LoggerFactory.getLogger(LoadTester.class);

34      public static void main(String[] args) throws Throwable {
35          //change the file path where the record ids will be saved.
36          System.out.print("cmd: [ ");
37          for (String tmp : args)
38              System.out.print(tmp + ", ");
39          System.out.println("]");
40          try {
41              // Parse command line (add additional options here)
42              CommandLine cmd = CommandLine
43                      .parser()
44                      .withOption("b", "binary", "parse binary data (two input files used")
45                      .withOption("p", "parse", "parse input data without loading it (for testig)")
46                      .withOption("d", "data", "load records from FILE", "FILE", null, true, false,
47                              true)
48                      .withOption("m", "max-records", "process at most NUM records", "NUM", null,
49                              true, false, false)
50                      .withOption("c", "config-path",
51                              "path of the configuration file for the Data Store", "STRING", null,
52                              true, false, false)
53                      .withOption("sids", "save ids",
54                              "save the ids in ./ids.txt").parse(args);
55  
56              // Extract data file, parse-only model, and max count options
57              File dataFile = cmd.getOptionValue("d", File.class);
58              boolean parseOnly = cmd.hasOption("p");
59              boolean binary = cmd.hasOption("b");
60              int maxRecords = cmd.getOptionValue("m", Integer.class, Integer.MAX_VALUE);
61  
62              // The data store and data transaction are not created in parse only mode
63              DataStore store = null;
64              DataTransaction tx = null;
65              PrintWriter writer = null;
66              if(cmd.hasOption("sids")){
67                  File outFile = new File("./ids.txt");
68                  writer = new PrintWriter(outFile);
69              }
70              try {
71                  // Create and Initialize the data store
72                  if (!parseOnly) {
73                      store = createStore(cmd);
74                      store.init();
75                      tx = store.begin(false);
76                      LOGGER.info("Store successfully initialized");
77                  }
78  
79                  // Load the records in the data file, within a single transaction
80                  int counter = 0;
81                  long ts = System.currentTimeMillis();
82                  try (Stream<Record> stream = binary ? readBinary(dataFile) : readRDF(dataFile)) {
83                      for (Record record : stream) {
84                          if (counter >= maxRecords) {
85                              break;
86                          }
87                          if(writer != null)
88                              writer.print(record.getID()+ "\n");
89                          
90                          LOGGER.trace("Processing record #{} {} ...", ++counter, record.getID());
91                          if (tx != null) {
92                              tx.store(record.getSystemType(), record);
93                          }
94                          if (LOGGER.isInfoEnabled() && counter % 5000 == 0) {
95                              LOGGER.info("{} records processed ({} records/s avg)", counter,
96                                      (counter * 1000L / (System.currentTimeMillis() - ts)));
97                          }
98                      }
99                      
100                     if(writer != null)
101                         writer.flush();
102                     
103                 }
104                 if (tx != null) {
105                     // ((DataTransactionElastic) tx).deleteAll();
106                     tx.end(true);
107                 }
108                 LOGGER.info("Processed {} records in {} ms", counter, System.currentTimeMillis() - ts);
109             } finally {
110                 // Make sure to close the store in any case
111                 if (store != null) {
112                     store.close();
113                     LOGGER.info("Store successfully closed");
114                 }
115             }
116            /* 
117             long ts = System.currentTimeMillis();
118             ((DataTransactionElastic)tx).optimizeIndex();
119             LOGGER.info("optimized in: {} ms", System.currentTimeMillis() - ts);
120             */
121         } catch (Throwable ex) {
122             // Report syntax errors as well as program failures
123             CommandLine.fail(ex);
124         }
125     }

127     private static DataStore createStore(CommandLine cmd) {
128         return new DataStoreElastic(cmd.getOptionValue("c", String.class));
129     }
130 
131     private static Stream<Record> readRDF(File file) throws IOException {
132         RDFFormat format = Rio.getParserFormatForFileName(file.getName());
133         InputStream in = IO.read(file.getAbsolutePath());
134         return Record.decode(RDFUtil.readRDF(in, format, null, null, true),
135                 ImmutableList.of(KS.RESOURCE, KS.MENTION), true).onClose(in);
136     }
137     
138     private static Stream<Record> readBinary(File file) throws IOException {
139         String base = file.getAbsolutePath();
140         Dictionary<URI> dictionary = new Dictionary<>(URI.class, base + ".dict");
141         Serializer serializer = new Serializer(false, dictionary, Statements.VALUE_FACTORY);
142         InputStream in = IO.read(base + ".gz");
143         return new Stream<Record>() {
144             
145             @Override
146             protected void doToHandler(Handler<? super Record> handler) throws Throwable {
147                 while (true) {
148                     try {
149                         Record record = (Record) serializer.fromStream(in);
150                         handler.handle(record);
151                     } catch (EOFException ex) {
152                         break;
153                     }
154                 }
155                 handler.handle(null);
156             }
157             
158         };
159     }
