1   import ch.qos.logback.classic.Level;
2   import com.google.common.base.Charsets;
3   import eu.fbk.dkm.pikes.rdf.vocab.KS;
4   import eu.fbk.dkm.utils.CommandLine;
5   import eu.fbk.rdfpro.RDFHandlers;
6   import eu.fbk.rdfpro.RDFProcessors;
7   import eu.fbk.rdfpro.RDFSource;
8   import eu.fbk.rdfpro.RDFSources;
9   import org.openrdf.model.Statement;
10  import org.openrdf.model.URI;
11  import org.openrdf.model.impl.ValueFactoryImpl;
12  import org.openrdf.model.vocabulary.XMLSchema;
13  import org.openrdf.rio.RDFHandler;
14  import org.openrdf.rio.RDFHandlerException;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  import java.io.File;
19  import java.nio.file.Files;
20  import java.util.ArrayList;
21  import java.util.HashMap;
22  import java.util.List;
23  import java.util.Map;
24  
25  /**
26   * Created by alessio on 27/11/15.
27   */
28  
29  public class ConvertTsvToRdf {
30  
31      private static final Logger LOGGER = LoggerFactory.getLogger(ConvertTsvToRdf.class);
32      private static String PIKES_PREFIX = "http://pikes.fbk.eu/mappings/";
33      private static String KS_PREFIX = "http://dkm.fbk.eu/ontologies/knowledgestore#";
34      private static String PREMON_PREFIX = "http://premon.fbk.eu/resource/";
35  
36      public static void main(String[] args) {
37          try {
38              final CommandLine cmd = CommandLine
39                      .parser()
40                      .withName("./converter")
41                      .withHeader("Convert Pikes TSVs in RDF")
42                      .withOption("p", "predicates", "TSV file with predicate mappings", "FILE",
43                              CommandLine.Type.FILE_EXISTING, true, false, false)
44                      .withOption("r", "roles", "TSV file with role mappings", "FILE",
45                              CommandLine.Type.FILE_EXISTING, true, false, false)
46                      .withOption("O", "output", "Output file prefix", "PREFIX",
47                              CommandLine.Type.STRING, true, false, true)
48                      .withLogger(LoggerFactory.getLogger("eu.fbk")).parse(args);
49  
50              ((ch.qos.logback.classic.Logger) LOGGER).setLevel(Level.INFO);
51  
52              ValueFactoryImpl valueFactory = ValueFactoryImpl.getInstance();
53              Statement statement;
54  
55              Map<String, String> prefixes = new HashMap<>();
56              prefixes.put("pikes", PIKES_PREFIX);
57              prefixes.put("ks", KS_PREFIX);
58              prefixes.put("pm", PREMON_PREFIX);
59              prefixes.put("xs", XMLSchema.NAMESPACE);
60  
61              File predicatesFile = cmd.getOptionValue("predicates", File.class);
62              File rolesFile = cmd.getOptionValue("roles", File.class);
63  
64              String outputPattern = cmd.getOptionValue("output", String.class);
65  
66              URI ksPredicate = valueFactory.createURI(KS_PREFIX + "predicate");
67              URI ksLemma = valueFactory.createURI(KS_PREFIX + "lemma");
68              URI ksPos = valueFactory.createURI(KS_PREFIX + "pos");
69              URI ksClass = valueFactory.createURI(KS_PREFIX + "class");
70              URI ksRole = valueFactory.createURI(KS_PREFIX + "role");
71              URI ksProperty = valueFactory.createURI(KS_PREFIX + "property");
72  
73              if (predicatesFile != null) {
74                  List<String> lines = Files.readAllLines(predicatesFile.toPath(), Charsets.UTF_8);
75                  List<Statement> statements = new ArrayList<>();
76                  File outputFile = new File(outputPattern + "-predicates.ttl.gz");
77  
78                  for (String line : lines) {
79                      line = line.trim();
80                      if (line.length() == 0) {
81                          continue;
82                      }
83  
84                      String[] parts = line.split("\\s+");
85                      if (parts.length < 4) {
86                          LOGGER.error("{} is not a valid line", line);
87                          continue;
88                      }
89  
90                      StringBuilder mappingBuffer = new StringBuilder();
91                      mappingBuffer.append(PIKES_PREFIX);
92                      mappingBuffer.append(parts[0].replace(':', '-')).append("-");
93                      mappingBuffer.append(parts[1]).append("-");
94                      mappingBuffer.append(parts[2]);
95  
96                      // Predicate
97                      String[] pParts = parts[0].split(":");
98                      if (pParts.length < 2) {
99                          LOGGER.error("{} is not a valid predicate", parts[0]);
100                         continue;
101                     }
102 
103                     StringBuilder predicateBuffer = new StringBuilder();
104                     predicateBuffer.append(PREMON_PREFIX);
105                     if (pParts[0].equals("fn")) {
106                         predicateBuffer.append("fn15-");
107                     } else if (pParts[0].equals("pb")) {
108                         predicateBuffer.append("pbon5-");
109                     } else if (pParts[0].equals("nb")) {
110                         predicateBuffer.append("nb10-");
111                     } else {
112                         LOGGER.error("{} is not a valid prefix", pParts[0]);
113                         continue;
114                     }
115                     predicateBuffer.append(parts[0].substring(3));
116 
117                     URI mappingURI = valueFactory.createURI(mappingBuffer.toString());
118                     URI predicateURI = valueFactory.createURI(predicateBuffer.toString());
119 
120                     statement = valueFactory.createStatement(mappingURI, ksPredicate, predicateURI);
121                     statements.add(statement);
122                     statement = valueFactory.createStatement(mappingURI, ksLemma, valueFactory.createLiteral(parts[1]));
123                     statements.add(statement);
124                     statement = valueFactory.createStatement(mappingURI, ksPos, valueFactory.createLiteral(parts[2]));
125                     statements.add(statement);
126                     statement = valueFactory.createStatement(mappingURI, ksClass, valueFactory.createURI(parts[3]));
127                     statements.add(statement);
128                 }
129 
130                 RDFSource source = RDFSources.wrap(statements, prefixes);
131 
132                 try {
133                     RDFHandler rdfHandler = RDFHandlers.write(null, 1000, outputFile.getAbsolutePath());
134                     source.emit(rdfHandler, 1);
135                 } catch (Exception e) {
136                     LOGGER.error("Input/output error, the file {} has not been saved ({})",
137                             outputFile.getAbsolutePath(), e.getMessage());
138                     throw new RDFHandlerException(e);
139                 }
140 
141             }
142 
143             if (rolesFile!= null) {
144                 List<String> lines = Files.readAllLines(rolesFile.toPath(), Charsets.UTF_8);
145                 List<Statement> statements = new ArrayList<>();
146                 File outputFile = new File(outputPattern + "-roles.ttl.gz");
147 
148                 for (String line : lines) {
149                     line = line.trim();
150                     if (line.length() == 0) {
151                         continue;
152                     }
153 
154                     String[] parts = line.split("\\s+");
155                     if (parts.length < 2) {
156                         LOGGER.error("{} is not a valid line", line);
157                         continue;
158                     }
159 
160                     StringBuilder mappingBuffer = new StringBuilder();
161                     mappingBuffer.append(PIKES_PREFIX);
162                     mappingBuffer.append(parts[0].replace(':', '-').replace('@', '-'));
163 
164                     // Predicate
165                     String[] pParts = parts[0].split(":");
166                     if (pParts.length < 2) {
167                         LOGGER.error("{} is not a valid role", parts[0]);
168                         continue;
169                     }
170 
171                     StringBuilder roleBuffer = new StringBuilder();
172                     roleBuffer.append(PREMON_PREFIX);
173                     if (pParts[0].equals("fn")) {
174                         roleBuffer.append("fn15-");
175                     } else if (pParts[0].equals("pb")) {
176                         roleBuffer.append("pbon5-");
177                     } else if (pParts[0].equals("nb")) {
178                         roleBuffer.append("nb10-");
179                     } else {
180                         LOGGER.error("{} is not a valid prefix", pParts[0]);
181                         continue;
182                     }
183                     roleBuffer.append(parts[0].substring(3).replace('@', '-'));
184 
185                     URI mappingURI = valueFactory.createURI(mappingBuffer.toString());
186                     URI roleURI = valueFactory.createURI(roleBuffer.toString());
187 
188                     statement = valueFactory.createStatement(mappingURI, ksRole, roleURI);
189                     statements.add(statement);
190                     statement = valueFactory.createStatement(mappingURI, ksProperty, valueFactory.createURI(parts[1]));
191                     statements.add(statement);
192                 }
193 
194                 RDFSource source = RDFSources.wrap(statements, prefixes);
195 
196                 try {
197                     RDFHandler rdfHandler = RDFHandlers.write(null, 1000, outputFile.getAbsolutePath());
198                     source.emit(rdfHandler, 1);
199                 } catch (Exception e) {
200                     LOGGER.error("Input/output error, the file {} has not been saved ({})",
201                             outputFile.getAbsolutePath(), e.getMessage());
202                     throw new RDFHandlerException(e);
203                 }
204             }
205         } catch (Exception e) {
206             CommandLine.fail(e);
207         }
208     }
209 }
