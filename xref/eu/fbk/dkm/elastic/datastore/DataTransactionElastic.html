87       * maps the URI type to a String, that can be used as a type for the ES index.
88       * @param type
89       * @return
90       */
91      private String getTypeAsString(URI type){
92          if(type.equals(KS.MENTION)) return "mention";
93          if(type.equals(KS.RESOURCE)) return "resource";
94          throw new IllegalArgumentException("unknow type: " + type.toString());

98      /**
99       * initialization of the bulk.
100      * Uses the default values only if bulkSize, flushInterval, concurrentRequest are never been set.
101      * Otherwise uses the values set.
102      */
103     private void initBulk(){
104         LOGGER.trace("starting a new bulk");
105         BulkProcessor.Builder builder = BulkProcessor.builder(client, new BulkProcessor.Listener(){
106             @Override
107             public void beforeBulk(long l, BulkRequest br) {
108                 client.admin().cluster().prepareHealth().setWaitForYellowStatus().execute().actionGet();
109                 LOGGER.trace("starting bulk operation with {} request", br.numberOfActions());
110             }
111             
112             @Override
113             public void afterBulk(long l, BulkRequest br, BulkResponse br1) {
114                 LOGGER.trace("finished bulk operation with {} request", br.numberOfActions());
115                 if(br1.hasFailures()){
116                     String log = "errors in bulk execution:";
117                     for(BulkItemResponse item : br1){
118                         if(item.isFailed())
119                             log += "\n\t id: " + item.getId() + " ; message: " + item.getFailureMessage();
120                     }
121                     LOGGER.error(log);
122                     //bulkException.set(new RuntimeException("failure in response -:" + br1.buildFailureMessage()));
123                 }
124             }
125             
126             @Override
127             public void afterBulk(long l, BulkRequest br, Throwable thrwbl){
128                 LOGGER.trace("finished bulk operation with {} request and errors", br.numberOfActions());
129                 bulkException.set(new RuntimeException("Caught exception in bulk: " + br + ", failure: " + thrwbl, thrwbl));
130             }
131         });
132         
133         if(configs.getBulkSize() != null) builder.setBulkSize(configs.getBulkSize());
134         //  if(configs.getFlushInterval() != null) builder.setFlushInterval(configs.getFlushInterval());
135         if(configs.getConcurrentRequests() != -1) builder.setConcurrentRequests(configs.getConcurrentRequests());
136         
137         bulk = builder.build();
138     }
139     
140     private void checkNotFailed(){
141         RuntimeException exception = bulkException.get();
142         if(exception != null){
143             if(!(exception instanceof IllegalStateException))
144                 bulkException.set(new IllegalStateException("previous bulk operation failed"));
145             throw exception;
146         }
147     }

149     /**
150      *
151      * @param timeout max time that the bulk execution can take.
152      * @param unit TimeUnit of timeout
153      * @throws InterruptedException
154      */
155     private void flushBulk(TimeValue time) throws IllegalStateException{
156         if(bulk != null){
157             LOGGER.debug("flushing bulk");
158             try{
159                 bulk.flush();
160                 if(!bulk.awaitClose(time.getMillis(), TimeUnit.MILLISECONDS))
161                     throw new RuntimeException("bulk request did not completed in " + time.getMillis() + TimeUnit.MILLISECONDS);
162                 
163                 bulk = null; //for signal that there is no bulk running.
164 //refresh of the index for make the documets available for search.
165                 RefreshResponse fr = client.admin().indices().refresh(new RefreshRequest(configs.getIndexName())).actionGet();
166                 LOGGER.debug(String.format("Flush: %s failed,  %s successful, %s total",fr.getFailedShards(),fr.getSuccessfulShards(),fr.getTotalShards()));
167             }catch(InterruptedException ex){
168                 throw new IllegalStateException("bulk execution interrupted", ex);
169             }
170         }
171         checkNotFailed();
172     }

181     /**
182      * search for the specified ids in the database
183      * @param type mention or resource
184      * @param ids identifiers of the documents to search.
185      * @param properties properties to keep. (null for all)
186      * @return the documents found.
187      * @throws IOException
188      * @throws IllegalArgumentException
189      * @throws IllegalStateException
190      */
191     @Override
192     public Stream<Record> lookup(URI type, Set<? extends URI> ids,  @Nullable Set<? extends URI> properties) throws IOException, IllegalArgumentException, IllegalStateException {
193         LOGGER.debug("lookup, type: " + type + " ; ids: " + ids + " ; properties: " + properties);
194         checkNotFailed();
195         if(ids.isEmpty()){
196             return Stream.create();
197         }
198 //flush the operation buffered in the bulk.
199         this.flushBulk(configs.getBulkTime());
200         
201         MultiGetRequestBuilder request = client.prepareMultiGet();
202         //set the type name where to search the data.
203         String typeName = getTypeAsString(type);
204         
205         for(URI id : ids){ //the projection is done after the get.
206             Item item = new MultiGetRequest.Item(configs.getIndexName(), typeName, uriHandler.encode(id));
207             request.add(item);
208         }
209         LOGGER.debug("multiGetRequest: "+ request);
210         this.waitYellowStatus();
211         
212 //convert from MultiGetItemResponse[] to Stream
213         MultiGetResponse response = request.execute().actionGet();
214         ArrayList<Record> records = new ArrayList<>();
215         //add records to results.
216         for(MultiGetItemResponse item : response) {
217             if(item.isFailed()){
218                 Failure failure = item.getFailure();
219                 LOGGER.error("failed MultiGet: id: " + failure.getId() + " message: " + failure.getMessage());
220             }else{
221                 GetResponse tmp = item.getResponse();
222                 Record recToAdd =  Utility.deserialize(tmp, mapper, uriHandler);
223                 if(recToAdd != null){
224                     if(properties != null)
225                         recToAdd = recToAdd.retain(Iterables.toArray(properties, URI.class));
226                     records.add(recToAdd);
227                 }
228             }
229         }
230         return Stream.create(records);
231     }

234     /**
235      * @param condition condition to use for filter the Records.
236      * @param partialAccept is a partial filter accepted.
237      * @return the filter. null if has not managed to build a filter.
238      */
239     private FilterBuilder buildFilter(XPath condition, boolean partialAccept) throws IOException{
240         LOGGER.debug("buildFilter condition: " + condition + " ; partialAccept: " + partialAccept);
241         Map<URI, Set<Object>> map = new HashMap<>();
242         XPath remain = condition.decompose(map); //splits the xPath
243         
244         if((!partialAccept && remain != null) || map.isEmpty()){
245             LOGGER.debug("entrySet number: " + map.size());
246             return null;
247         }
248         List<FilterBuilder> filters = new ArrayList<>();
249         
250         for(Map.Entry el : map.entrySet()){
251             LOGGER.debug("analyze key: " + el.getKey());
252             String elKey = null;
253             if(el.getKey() instanceof URI)
254                 elKey = uriHandler.encode((URI)el.getKey());
255             else{
256                 //something wrog, it's not a Value
257                 throw new IllegalArgumentException("found a key in the xPath map that is not a URI");
258             }
259             
260             if(mapper.getValueClass(elKey).equals(byte[].class)){ //byte[] are not pre-filtered. Only post-filter
261                 LOGGER.debug("byte[], ignore");
262                 if(!partialAccept) return null;
263                 //else ignore.
264             }else{
265                 LOGGER.debug("analizing key: " + elKey);
266                 ArrayList<FilterBuilder> tmpFilters = new ArrayList<>();
267                 LOGGER.debug("number of orFilters: " + ((Set<Object>)el.getValue()).size());
268                 for(Object item : (Set<Object>)el.getValue()){
269                     FilterBuilder tmpFilter = null;
270                     LOGGER.debug("analizing value: " + item.toString() + " ; class: " + item.getClass());
271                     if(item instanceof Range){
272                         tmpFilter = Utility.buildRangeFilter(elKey, (Range)item, mapper, uriHandler);
273                     }else{
274                         LOGGER.debug("term filter: " + item.getClass());
275                         tmpFilter = Utility.buildTermFilter(elKey, item, mapper, uriHandler);
276                     }
277                     if(tmpFilter != null){ //if the creation of the filter failed.
278                         tmpFilters.add(tmpFilter);
279                     }else{
280                         LOGGER.debug("creation of the filter on {} failed", condition);
281                         if(!partialAccept) return null;
282                     }
283                 }
284                 filters.add(FilterBuilders.orFilter(Iterables.toArray(tmpFilters, FilterBuilder.class))); //if a document passes 1 of the tmpFilter I keep that
285             }
286         }
287         FilterBuilder mainFilter = null;
288         if(!filters.isEmpty())
289             mainFilter = FilterBuilders.andFilter(Iterables.toArray(filters, FilterBuilder.class)); //a documet has to pass all the filters.
290         
291         return mainFilter;
292     }

294     /**
295      * returns all the documents in the database that match the conditions
296      * @param type mention or resource
297      * @param condition condition to match
298      * @param properties properties of the documents to keep
299      * @return the documents found.
300      * @throws IOException
301      * @throws IllegalArgumentException
302      * @throws IllegalStateException
303      */
304     @Override
305     public Stream<Record> retrieve(URI type, @Nullable XPath condition, @Nullable Set<? extends URI> properties) throws IOException, IllegalArgumentException, IllegalStateException {
306         checkNotFailed();
307         LOGGER.debug("retrieve type: " + type + " ; condition: " + condition + " ; properties: " + properties);
308         
309         this.flushBulk(configs.getBulkTime());
310         
311         String typeName = getTypeAsString(type);
312         
313         SearchRequestBuilder responseBuilder;
314         QueryBuilder mainQuery;
315         //if there are no conditions I take all the documents.
316         if(condition == null){ //matchAll query.
317             LOGGER.debug("match all query");
318             mainQuery = QueryBuilders.constantScoreQuery(FilterBuilders.matchAllFilter());
319             
320         }else{ //if there are some conditions, take only the documents that satisfy all of them. pre-checking
321             LOGGER.debug("try to build a filter of condition: " + condition);
322             FilterBuilder filter = buildFilter(condition, true);
323             if(filter == null){ //if the build of the filter fails, match all and post-filter.
324                 LOGGER.debug("prefilter failed");
325                 filter = FilterBuilders.matchAllFilter();
326             }
327             mainQuery = QueryBuilders.constantScoreQuery(filter);
328         }
329         
330         LOGGER.debug("Query: " + mainQuery);
331         responseBuilder = client.prepareSearch(configs.getIndexName()).setTypes(typeName)
332                 .setQuery(mainQuery).setScroll(configs.getTimeout());
333         
334         this.waitYellowStatus();
335         
336         SearchResponse response = responseBuilder.execute().actionGet();
337         Stream<Record> res = new SearchResponseStream(response, client, configs.getTimeout(), mapper, uriHandler);
338         
339         if(condition != null){
340             LOGGER.debug("post-filtering: " + condition);
341             res = res.filter(condition.asPredicate(), 0); //post-checking
342         }
343         
344         if(properties != null){ //projection
345             final URI[] propURIs = Iterables.toArray(properties, URI.class);
346             res = res.transform((Record r) -> {return r.retain(propURIs);}, 0);
347         }        
348         return res;
349     }

351     /**
352      * the number of documents matching the conditions
353      * @param type mention or resource
354      * @param condition condition to match
355      * @return the number of documents matching the conditions
356      * @throws IOException
357      * @throws IllegalArgumentException
358      * @throws IllegalStateException
359      */
360     @Override
361     public long count(URI type, @Nullable XPath condition) throws IOException, IllegalArgumentException, IllegalStateException {
362         checkNotFailed();
363         LOGGER.debug("count type: " + type + " ; condition: " + condition);
364         
365 //flush the bulk.
366         this.flushBulk(configs.getBulkTime());
367         
368         
369         String typeName = getTypeAsString(type);
370         
371         CountRequestBuilder responseBuilder = null;
372         //if there are no conditions I take all the documents (from, size).
373         if(condition == null){ //matchAll query.
374             responseBuilder = client.prepareCount(configs.getIndexName()).setTypes(typeName)
375                     .setQuery(QueryBuilders.matchAllQuery());
376             
377         }else{ //if there are some conditions, take only the documents that satisfy all of them. pre-checking
378             FilterBuilder mainFilter = buildFilter(condition, false);
379             if(mainFilter != null){
380                 QueryBuilder mainQuery = QueryBuilders.constantScoreQuery(mainFilter);
381                 responseBuilder = client.prepareCount(configs.getIndexName()).setTypes(typeName)
382                         .setQuery(mainQuery);
383             }else{ //if we can't pre-filter all, do a retrive with 0 properties.
384                 this.waitYellowStatus();
385                 LOGGER.debug("can not prefilter all, have to do a matchall and postfilter");
386                 Stream<Record> stream = this.retrieve(type, condition, new HashSet<>());
387                 return stream.count();
388             }
389         }
390         this.waitYellowStatus();
391         CountResponse response = responseBuilder.execute().actionGet();
392         return response.getCount();
393     }

400     /**
401      * stores the specified record in the database.
402      * @param type mention or resource
403      * @param record record to store
404      * @throws IOException
405      * @throws IllegalStateException
406      */
407     @Override
408     public void store(URI type, Record record) throws IOException, IllegalStateException {
409         checkNotFailed();
410         if(bulk == null) //start a new bulk if it's not already started.
411             this.initBulk();
412         
413         String typeName = getTypeAsString(type);
414         
415         XContentBuilder source = Utility.serialize(record, uriHandler);
416         LOGGER.trace("storing\n\trecord: " + record.toString(null, true) + "\n\t-> serialized: " + source.string());
417         IndexRequest indexRequest = new IndexRequest(configs.getIndexName(), typeName, uriHandler.encode(record.getID())).source(source);
418         UpdateRequest updateRequest = new UpdateRequest(configs.getIndexName(), typeName,  uriHandler.encode(record.getID())).doc(source).upsert(indexRequest);
419         bulk.add(updateRequest);
420     }

422     /**
423      * removes a document from the database given it's id
424      * @param type mention or resource
425      * @param id identifier of the document to delete
426      * @throws IOException
427      * @throws IllegalStateException
428      */
429     @Override
430     public void delete(URI type, URI id) throws IOException, IllegalStateException {
431         checkNotFailed();
432         LOGGER.trace("delete document type: " + type + " ; id: " + id);
433         
434         String typeName = getTypeAsString(type);
435         String indexName = configs.getIndexName();
436         String idStr = uriHandler.encode(id);
437         DeleteRequest deleteRequest = new DeleteRequest(indexName, typeName, idStr);
438         
439         if(bulk == null) //start a new bulk if it's not already started.
440             this.initBulk();
441         bulk.add(deleteRequest);
442     }

444     /**
445      * end the "transaction".
446      * @param commit
447      * @throws DataCorruptedException
448      * @throws IOException
449      * @throws IllegalStateException
450      */
451     @Override
452     public void end(boolean commit) throws DataCorruptedException, IOException, IllegalStateException {
453         checkNotFailed();
454         LOGGER.debug("end");
455         if(commit){
456             LOGGER.debug("flushing pending operation");
457             this.flushBulk(configs.getBulkTime());
458             LOGGER.debug("done");
459         }
460     }

470     public void deleteAllIndexContent(){
471         this.waitYellowStatus();
472         if(!client.admin().indices().prepareDelete("*").execute().actionGet().isAcknowledged())
473             throw new RuntimeException("can not delete the index ("+configs.getIndexName()+") content ");
474     }
475     public void deleteAll(){
476         this.waitYellowStatus();
477         DeleteIndexResponse response = null;
478         try{
479             response = client.admin().indices().delete(new DeleteIndexRequest(configs.getIndexName())).actionGet();
480         }catch(IndexMissingException ex){
481             return;
482         }
483         if(!response.isAcknowledged())
484             throw new RuntimeException("can not delete the index: " + configs.getIndexName());
485     }
