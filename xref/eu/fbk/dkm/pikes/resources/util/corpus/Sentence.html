39      /**
40       * Given a word, it gives the list of ancestors in the dependency tree.
41       *
42       * @param i the index (from 0 to n-1) of the Word
43       * @return the ist of ancestors
44       */
45      public List<Integer> getAncestors(int i) {
46          ArrayList<Integer> ret = new ArrayList<>();
47  
48          int noLoop = 0;
49          while (i >= 0 && words.get(i) != null) {
50              ret.add(i);
51              i = words.get(i).getDepParent() - 1;
52              noLoop++;
53              if (noLoop > LOOP_STOP) {
54                  break;
55              }
56          }
57  
58          return ret;
59      }
60  
61      /**
62       * Given a word, it gives the list of descendants.
63       *
64       * @param i the index (from 1 to n) of the Word
65       * @return the ist of ancestors
66       */
67      public Set<Integer> getDescendants(int i) {
68          Set<Integer> ret = new HashSet<>();
69  
70          HashMultimap<Integer, Integer> children = HashMultimap.create();
71          for (Word word : this) {
72              if (word.getDepParent() > 0) {
73                  children.put(word.getDepParent(), word.getId());
74              }
75          }
76  
77          int noLoop = 0;
78          addChildren(children, ret, i, noLoop);
79  
80          return ret;
81      }
82  
83      public void addChildren(HashMultimap<Integer, Integer> map, Set<Integer> ret, Integer parent, int noLoop) {
84          if (noLoop > LOOP_STOP) {
85              return;
86          }
87  
88          ret.add(parent);
89          Set<Integer> children = map.get(parent);
90          ret.addAll(children);
91          for (Integer child : children) {
92              addChildren(map, ret, child, noLoop + 1);
93          }
94  
95      }

97      public Integer searchHead(List<Integer> span) {
98          List<Integer>[] ancestorLists = new List[span.size()];
99          for (int i = 0; i < span.size(); i++) {
100             Integer integer = span.get(i);
101             ancestorLists[i] = getAncestors(integer);
102         }
103 
104         return giveHead(ancestorLists);
105     }
106 
107     private Integer giveHead(List<Integer>... ancestorLists) {
108         if (ancestorLists.length == 1) {
109             return ancestorLists[0].get(0);
110         }
111 
112         int k = 1;
113         int last = 0;
114         while (true) {
115             Set<Integer> integers = new HashSet<>();
116             for (int i = 0; i < ancestorLists.length; i++) {
117                 try {
118                     integers.add(ancestorLists[i].get(ancestorLists[i].size() - k));
119                 } catch (ArrayIndexOutOfBoundsException e) {
120                     return last;
121                 }
122             }
123             if (integers.size() == 1) {
124                 for (Integer integer : integers) {
125                     last = integer;
126                 }
127             } else {
128                 return last;
129             }
130             k++;
131         }
132     }

145     public String toConllString() {
146 
147         Map<Integer, Srl> srlMap = new TreeMap<>();
148         for (Srl srl : srls) {
149             Word word = srl.getTarget().get(0);
150             srlMap.put(word.getId(), srl);
151         }
152 
153         int additionalColumnsNo = 2 + srlMap.size();
154         String[][] additionalColumns = new String[getWords().size()][additionalColumnsNo];
155         for (int j = 0; j < getWords().size(); j++) {
156             for (int i = 0; i < additionalColumnsNo; i++) {
157                 additionalColumns[j][i] = "_";
158             }
159         }
160 
161         int roleIndex = 0;
162         for (Integer key : srlMap.keySet()) {
163             Srl srl = srlMap.get(key);
164 
165             additionalColumns[key - 1][0] = "Y";
166             additionalColumns[key - 1][1] = srl.getLabel();
167 
168             for (Role role : srl.getRoles()) {
169                 int roleKey = role.getSpan().get(0).getId();
170                 additionalColumns[roleKey - 1][2 + roleIndex] = role.getLabel();
171             }
172 
173             roleIndex++;
174         }
175 
176         StringBuilder builder = new StringBuilder();
177 
178         int i = 0;
179         for (Word word : getWords()) {
180             builder.append(word.getId()).append("\t");
181             builder.append(word.getForm()).append("\t");
182             builder.append(word.getForm()).append("\t");
183             builder.append(word.getLemma()).append("\t");
184             builder.append(word.getPos()).append("\t");
185             builder.append(word.getPos()).append("\t");
186             builder.append("_").append("\t");
187             builder.append("_").append("\t");
188             builder.append(word.getDepParent()).append("\t");
189             builder.append(word.getDepParent()).append("\t");
190             builder.append(word.getDepLabel()).append("\t");
191             builder.append(word.getDepLabel());
192             for (String col : additionalColumns[i]) {
193                 builder.append("\t").append(col);
194             }
195             builder.append("\n");
196 
197             i++;
198         }
199         builder.append("\n");
200 
201         return builder.toString();
202     }
