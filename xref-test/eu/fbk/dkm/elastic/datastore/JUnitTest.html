401         System.out.println("condition: " + condition.toString());
402         Stream<Record> res;
403         try {
404             res = instance.retrieve(type, condition, properties);
405         } catch (IOException | IllegalArgumentException | IllegalStateException ex) {

407             return;
408         }
409         checkEquals(exp, fromStreamToList(res));
410     }
