167     private static Value castValue(final Object value) throws IllegalArgumentException {
168 
169         final ValueFactory vf = Data.getValueFactory();
170 
171         if (value == null) {
172             return null;
173         } else if (value instanceof ExtendedString) {
174             final ExtendedString es = (ExtendedString) value;
175             String string = es.toString();
176             try {
177                 if (es.getIriType() == ExtendedString.IRI && (es.getStrType() & 0x01) == 0x01) {
178                     if (string.startsWith("_:")) {
179                         string = string.substring(2);
180                         return vf.createBNode(string);
181                     } else if (string.indexOf(':') < 0) {
182                         return vf.createURI(":" + string);
183                     } else {
184                         return vf.createURI(string);
185                     }
186                 } else if (es.getIriType() == ExtendedString.BNODE) {
187                     return vf.createBNode(string);
188                 } else {
189                     return vf.createLiteral(string);
190                 }
191             } catch (final Throwable ex) {
192                 throw new IllegalArgumentException("Invalid value from Virtuoso: \"" + string
193                         + "\", STRTYPE = " + es.getIriType(), ex);
194             }
195         } else if (value instanceof RdfBox) {
196             final RdfBox rb = (RdfBox) value;
197             if (rb.getLang() != null) {
198                 return vf.createLiteral(rb.toString(), rb.getLang());
199             } else if (rb.getType() != null) {
200                 return vf.createLiteral(rb.toString(), vf.createURI(rb.getType()));
201             } else {
202                 return vf.createLiteral(rb.toString());
203             }
204         } else if (value instanceof Blob) {
205             return vf.createLiteral(value.toString(), XMLSchema.HEXBINARY);
206         } else if (value instanceof Date) {
207             return Data.convert(new java.util.Date(((Date) value).getTime()), Value.class);
208         } else if (value instanceof Timestamp) {
209             return Data.convert(new Date(((Timestamp) value).getTime()), Value.class);
210         } else if (value instanceof Time) {
211             return vf.createLiteral(value.toString(), XMLSchema.TIME);
212         } else {
213             try {
214                 return Data.convert(value, Value.class);
215             } catch (final Throwable ex) {
216                 throw new IllegalArgumentException("Could not parse value: " + value, ex);
217             }
218         }
219     }

221     private static String sqlForQuery(final String query, @Nullable final Dataset dataset,
222             @Nullable final BindingSet bindings) {
223 
224         // Start composing a 'sparql' SQL command
225         final StringBuilder builder = new StringBuilder("sparql\n ");
226 
227         // Generate define directives for graphs in the FROM / FROM NAMED clauses
228         if (dataset != null) {
229             final Set<URI> empty = Collections.emptySet();
230             for (final URI uri : MoreObjects.firstNonNull(dataset.getDefaultGraphs(), empty)) {
231                 builder.append(" define input:default-graph-uri <" + uri + "> \n");
232             }
233             for (final URI uri : MoreObjects.firstNonNull(dataset.getNamedGraphs(), empty)) {
234                 builder.append(" define input:named-graph-uri <" + uri + "> \n");
235             }
236         }
237 
238         // Apply variable bindings, i.e., replace certain variables with supplied values
239         if (bindings != null && bindings.size() > 0) {
240             int i = 0;
241             final int length = query.length();
242             while (i < query.length()) {
243                 final char ch = query.charAt(i++);
244                 if (ch == '\\' && i < length) {
245                     builder.append(ch).append(query.charAt(i++));
246                 } else if (ch == '"' || ch == '\'') {
247                     builder.append(ch);
248                     while (i < length) {
249                         final char c = query.charAt(i++);
250                         builder.append(c);
251                         if (c == ch) {
252                             break;
253                         }
254                     }
255                 } else if ((ch == '?' || ch == '$') && i < length
256                         && isVarFirstChar(query.charAt(i))) {
257                     int j = i + 1;
258                     while (j < length && isVarMiddleChar(query.charAt(j))) {
259                         ++j;
260                     }
261                     final String name = query.substring(i, j);
262                     final Value value = bindings.getValue(name);
263                     if (value != null) {
264                         builder.append(Data.toString(value, null));
265                     } else {
266                         builder.append(ch).append(name);
267                     }
268                     i = j;
269                 } else {
270                     builder.append(ch);
271                 }
272             }
273         } else {
274             builder.append(query);
275         }
276 
277         // Return the resulting SQL statement
278         return builder.toString();
279     }

281     private static boolean isVarFirstChar(final char c) {
282         // Returns true if the supplied char can be used as first char in a variable name
283         return '0' <= c && c <= '9' || 'A' <= c && c <= 'Z' || 'a' <= c && c <= 'z' || c == '_'
284                 || 0x00C0 <= c && c <= 0x00D6 || 0x00D8 <= c && c <= 0x00F6 || 0x00F8 <= c
285                 && c <= 0x02FF || 0x0370 <= c && c <= 0x037D || 0x037F <= c && c <= 0x1FFF
286                 || 0x200C <= c && c <= 0x200D || 0x2070 <= c && c <= 0x218F || 0x2C00 <= c
287                 && c <= 0x2FEF || 0x3001 <= c && c <= 0xD7FF || 0xF900 <= c && c <= 0xFDCF
288                 || 0xFDF0 <= c && c <= 0xFFFD;
289     }
290 
291     private static boolean isVarMiddleChar(final char c) {
292         // Returns true if the supplied char can be used as i-th char (i > 1) in a variable name
293         return isVarFirstChar(c) || c == 0x00B7 || 0x0300 <= c && c <= 0x036F || 0x203F <= c
294                 && c <= 0x2040;
295     }

297     private static boolean isPartialResultException(final Throwable ex) {
298         // TODO: try to better implement this method without checking for string containment
299         // (perhaps now we have access to some SQL code)
300         return ex.getMessage() != null && ex.getMessage().contains("Returning incomplete results");

303     private static void killConnection(final Object connection) throws Throwable {
304         if (connection instanceof ConnectionWrapper) {
305             final Field field = ConnectionWrapper.class.getDeclaredField("pconn");
306             field.setAccessible(true);
307             killConnection(field.get(connection));
308         } else if (connection instanceof VirtuosoPooledConnection) {
309             killConnection(((VirtuosoPooledConnection) connection).getVirtuosoConnection());
310         } else if (connection instanceof VirtuosoConnection) {
311             final Field field = VirtuosoConnection.class.getDeclaredField("socket");
312             field.setAccessible(true);
313             final Closeable socket = (Closeable) field.get(connection);
314             socket.close(); // as Virtuoso driver ignores polite interrupt
315         } else {
316             throw new Exception("Don't know how to kill connection "
317                     + connection.getClass().getName());
318         }
319     }

321     private final class VirtuosoTransaction implements TripleTransaction {

324 
325         private final boolean readOnly; // whether only get() and query() are allowed

327         private final String id; // transaction name (for logging purposes)

329         VirtuosoTransaction(final boolean readOnly) throws IOException {
330 
331             // Setup ID and read-only setting
332             this.readOnly = readOnly;
333             this.id = "Virtuoso TX"
334                     + VirtuosoJdbcTripleStore.this.transactionCounter.incrementAndGet();
335 
336             // Acquire a JDBC connection from the pool
337             Connection connection = null;
338             try {
339                 connection = VirtuosoJdbcTripleStore.this.source.getConnection();
340                 connection.setTransactionIsolation(Connection.TRANSACTION_READ_UNCOMMITTED);
341                 connection.setReadOnly(readOnly);
342                 connection.setAutoCommit(true);
343             } catch (final SQLException ex) {
344                 Util.closeQuietly(connection);
345                 throw new IOException("Could not connect to Virtuoso", ex);
346             }
347             this.connection = connection;
348         }

350         private void checkWritable() {
351             if (this.readOnly) {
352                 throw new IllegalStateException(
353                         "Write operation not allowed on read-only transaction");
354             }
355         }

357         @Override
358         public CloseableIteration<? extends Statement, ? extends Exception> get(
359                 final Resource subject, final URI predicate, final Value object,
360                 final Resource context) throws IOException, IllegalStateException {
361 
362             // Generate the query retrieving subset of variables ?s, ?p, ?o, ?c for req. stmts
363             // NOTE: we always use GRAPH as triple in Virtuoso are *always* stored inside a graph
364             final StringBuilder builder = new StringBuilder();
365             builder.append("SELECT * WHERE { GRAPH ");
366             builder.append(context == null ? "?c" : Data.toString(context, null));
367             builder.append(" { ");
368             builder.append(subject == null ? "?s" : Data.toString(subject, null));
369             builder.append(' ');
370             builder.append(predicate == null ? "?p" : Data.toString(predicate, null));
371             builder.append(' ');
372             builder.append(object == null ? "?o" : Data.toString(object, null));
373             builder.append(" } }");
374             final String query = builder.toString();
375 
376             // Execute the query, converting each result from BindingSet to Statement
377             return new ConvertingIteration<BindingSet, Statement, Exception>(
378                     new VirtuosoQueryIteration(query, null, null, null)) {
379 
380                 @Override
381                 protected Statement convert(final BindingSet tuple) throws Exception {
382                     final Resource s = subject != null ? subject : (Resource) tuple.getValue("s");
383                     final URI p = predicate != null ? predicate : (URI) tuple.getValue("p");
384                     final Value o = object != null ? object : tuple.getValue("o");
385                     final Resource c = context != null ? context : (Resource) tuple.getValue("c");
386                     return Data.getValueFactory().createStatement(s, p, o, c);
387                 }
388 
389             };
390         }

431         private void update(final boolean insert, final Iterable<? extends Statement> statements)
432                 throws IOException, IllegalStateException {
433 
434             // Check arguments and state
435             Preconditions.checkNotNull(statements);
436             checkWritable();
437 
438             try {
439                 // Compose the SQL statement
440                 final StringBuilder builder = new StringBuilder();
441                 builder.append("SPARQL ");
442                 builder.append(insert ? "INSERT" : "DELETE");
443                 builder.append(" DATA {");
444                 for (final Statement stmt : statements) {
445                     final Resource ctx = MoreObjects.firstNonNull(stmt.getContext(), SESAME.NIL);
446                     builder.append("\n  GRAPH ");
447                     builder.append(Data.toString(ctx, null));
448                     builder.append(" { ");
449                     builder.append(Data.toString(stmt.getSubject(), null));
450                     builder.append(' ');
451                     builder.append(Data.toString(stmt.getPredicate(), null));
452                     builder.append(' ');
453                     builder.append(Data.toString(stmt.getObject(), null));
454                     builder.append(" }");
455                 }
456                 builder.append("\n}");
457 
458                 // Issue the statement
459                 final java.sql.Statement statement = this.connection.createStatement();
460                 statement.executeUpdate(builder.toString());
461 
462             } catch (final SQLException ex) {
463                 throw new IOException(ex);
464             }
465         }

467         @Override
468         public void end(final boolean commit) throws DataCorruptedException, IOException,
469                 IllegalStateException {
470 
471             try {
472                 // Schedule a task for killing the connection by forcedly closing its socket
473                 final Future<?> future = Data.getExecutor().schedule(new Runnable() {
474 
475                     @Override
476                     public void run() {
477                         try {
478                             killConnection(VirtuosoTransaction.this.connection);
479                             LOGGER.warn("{} - killed Virtuoso JDBC connection", this);
480                         } catch (final Throwable ex) {
481                             LOGGER.debug(this + " - failed to kill Virtuoso JDBC connection "
482                                     + "(connection class is "
483                                     + VirtuosoTransaction.this.connection.getClass() + ")", ex);
484                         }
485                     }
486 
487                 }, 1000, TimeUnit.MILLISECONDS);
488 
489                 // Perform commit or rollback, in case of a non-read-only transaction
490                 if (!this.readOnly) {
491                     if (commit) {
492                         this.connection.commit();
493                     } else {
494                         this.connection.rollback();
495                     }
496                 }
497 
498                 // Try to close the connection 'kindly'. On success, unschedule the killing job
499                 this.connection.close();
500                 future.cancel(false);
501 
502             } catch (final SQLException ex) {
503                 LOGGER.error(this + " - failed to close connection", ex);
504             }
505         }

512         private final class VirtuosoQueryIteration implements
513                 CloseableIteration<BindingSet, QueryEvaluationException> {
514 
515             private final List<String> variables; // output variables
516 
517             private java.sql.Statement statement; // to be closed at the end of the iteration
518 
519             private ResultSet cursor; // to be closed at the end of the iteration
520 
521             private BindingSet tuple; // next tuple to be returned
522 
523             public VirtuosoQueryIteration(final String query, @Nullable final Dataset dataset,
524                     @Nullable final BindingSet bindings, @Nullable final Long timeout)
525                     throws IOException {
526 
527                 try {
528                     // Convert the SPARQL query to the corresponding Virtuoso SQL command
529                     final String sql = sqlForQuery(query, dataset, bindings);
530 
531                     // Set the server-side timeout that control returning of partial results
532                     final int msTimeout = timeout == null ? 0 : timeout.intValue();
533                     try {
534                         VirtuosoTransaction.this.connection.prepareCall(
535                                 "set result_timeout = " + msTimeout).execute();
536                     } catch (final Throwable ex) {
537                         LOGGER.warn(VirtuosoTransaction.this
538                                 + " - failed to set result_timeout = " + msTimeout
539                                 + " on Virtuoso JDBC connection (proceeding anyway)", ex);
540                     }
541 
542                     // Create and configure the SQL Statement object for executing the query
543                     this.statement = VirtuosoTransaction.this.connection.createStatement();
544                     this.statement.setFetchDirection(ResultSet.FETCH_FORWARD);
545                     this.statement.setFetchSize(VirtuosoJdbcTripleStore.this.fetchSize);
546                     if (timeout != null) {
547                         // Set the client-side timeout in seconds for getting the first result
548                         this.statement.setQueryTimeout((int) ((timeout + GRACE_PERIOD) / 1000));
549                     }
550 
551                     // Start with empty variable list and null output tuple (they will be changed
552                     // upon successful query execution).
553                     this.variables = Lists.newArrayList();
554                     this.tuple = null;
555 
556                     // Execute the query (this may fail in case of partial results)
557                     this.cursor = this.statement.executeQuery(sql);
558 
559                     // Retrieve output variables.
560                     final ResultSetMetaData metadata = this.cursor.getMetaData();
561                     for (int i = 1; i <= metadata.getColumnCount(); ++i) {
562                         this.variables.add(metadata.getColumnName(i));
563                     }
564 
565                 } catch (final Throwable ex) {
566                     if (isPartialResultException(ex)) {
567                         // Ignore the excetion and close immediately the allocated resources.
568                         // An empty iteration will be returned
569                         LOGGER.debug(
570                                 "{} -no results / partial results returned due to expired timeout",
571                                 VirtuosoTransaction.this);
572                         Util.closeQuietly(this);
573                     }
574                     throw new IOException("Could not obtain query result set", ex);
575                 }
576             }
577 
578             @Override
579             public boolean hasNext() throws QueryEvaluationException {
580                 if (this.tuple == null) {
581                     this.tuple = advance();
582                 }
583                 return this.tuple != null;
584             }
585 
586             @Override
587             public BindingSet next() throws QueryEvaluationException {
588                 if (this.tuple == null) {
589                     this.tuple = advance();
590                 }
591                 if (this.tuple == null) {
592                     throw new NoSuchElementException();
593                 }
594                 final BindingSet result = this.tuple;
595                 this.tuple = null;
596                 return result;
597             }
598 
599             @Override
600             public void remove() throws QueryEvaluationException {
601                 throw new UnsupportedOperationException();
602             }
603 
604             @Override
605             public void close() throws QueryEvaluationException {
606                 if (this.statement != null) {
607 
608                     final Future<?> future = Data.getExecutor().schedule(new Runnable() {
609 
610                         @Override
611                         public void run() {
612                             try {
613                                 end(false);
614                                 LOGGER.warn(VirtuosoTransaction.this
615                                         + " - forced closure of Virtuoso transaction "
616                                         + "after unsuccessfull attempt at closing Virtuoso iteration");
617                             } catch (final Throwable ex) {
618                                 LOGGER.debug(VirtuosoTransaction.this
619                                         + " - failed to close Virtuoso transaction after "
620                                         + "unsuccessfull attempt at closing Virtuoso iteration",
621                                         ex);
622                             }
623                         }
624 
625                     }, 1000, TimeUnit.MILLISECONDS);
626 
627                     try {
628                         // Try to close 'politely' the Virtuoso cursor and the associated
629                         // statements. After 1 second blocked in this operation, we will force
630                         // closure of the whole triple transaction, which in turn may cause the
631                         // killing of the Virtuoso JDBC connection (after one second waiting for
632                         // politely closing it)
633                         this.cursor.close();
634                         this.statement.close();
635                         future.cancel(false);
636 
637                     } catch (final SQLException e) {
638                         throw new QueryEvaluationException(e);
639 
640                     } finally {
641                         this.statement = null;
642                         this.cursor = null;
643                     }
644                 }
645             }
646 
647             @Override
648             protected void finalize() throws Throwable {
649                 Util.closeQuietly(this);
650             }

652             private BindingSet advance() throws QueryEvaluationException {
653                 try {
654                     final BindingSet result = null;
655                     if (this.cursor != null) {
656                         if (this.cursor.next()) {
657                             final int size = this.variables.size();
658                             final Value[] values = new Value[size];
659                             for (int i = 0; i < size; ++i) {
660                                 values[i] = castValue(this.cursor.getObject(this.variables.get(i)));
661                             }
662                             return new ListBindingSet(this.variables, values);
663                         } else {
664                             close();
665                         }
666                     }
667                     return result;
668                 } catch (final Exception ex) {
669                     if (isPartialResultException(ex)) {
670                         // On partial results, terminate the iteration ignoring the exception
671                         LOGGER.debug("{} - partial results returned due to expired timeout",
672                                 VirtuosoTransaction.this);
673                         return null;
674                     }
675                     throw new QueryEvaluationException("Could not retrieve next query result", ex);
676                 }
677             }
