98      static boolean compare(String fileName1, String fileName2) throws IOException {
99          FileInputStream fstream1 = new FileInputStream(fileName1);
100         FileInputStream fstream2 = new FileInputStream(fileName2);
101 
102         DataInputStream in1 = new DataInputStream(fstream1);
103         DataInputStream in2 = new DataInputStream(fstream2);
104 
105         BufferedReader br1 = new BufferedReader(new InputStreamReader(in1));
106         BufferedReader br2 = new BufferedReader(new InputStreamReader(in2));
107 
108         String strLine1, strLine2;
109         boolean problem = false;
110 
111         while ((strLine1 = br1.readLine()) != null && (strLine2 = br2.readLine()) != null) {
112             if (!strLine1.equals(strLine2)) {
113                 problem = true;
114             }
115         }
116         fstream1.close();
117         fstream2.close();
118         return problem;
119     }
