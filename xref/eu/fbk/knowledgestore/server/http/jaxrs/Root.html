93      @GET
94      @Path("/static/{name:.*}")
95      public Response download(@PathParam("name") final String name) throws Throwable {
96          final InputStream stream = Root.class.getResourceAsStream(name);
97          if (stream == null) {
98              throw new WebApplicationException("No resource named " + name, Status.NOT_FOUND);
99          }
100         final String type = Data.extensionToMimeType(name);
101         init(false, type, null, null);
102         final CacheControl control = new CacheControl();
103         control.setMaxAge(3600 * 24);
104         control.setMustRevalidate(true);
105         control.setPrivate(false);
106         return newResponseBuilder(Status.OK, stream, null).cacheControl(control).build();

109     @GET
110     @Path("/ui")
111     @Produces("text/html;charset=UTF-8")
112     public Viewable ui() throws Throwable {
113 
114         final Map<String, Object> model = Maps.newHashMap();
115         model.put("maxTriples", getUIConfig().getResultLimit());
116         String view = "/status";
117 
118         final String action = getParameter("action", String.class, null, model);
119         final Long timeoutSec = getParameter("timeout", Long.class, null, model);
120         final Long timeout = timeoutSec == null ? null : timeoutSec * 1000;
121         final int limit = getParameter("limit", Integer.class, getUIConfig().getResultLimit(),
122                 model);
123 
124         try {
125             if ("lookup".equals(action)) {
126                 final URI id = getParameter("id", URI.class, null, model);
127                 final URI selection = getParameter("selection", URI.class, null, model);
128                 view = "/lookup";
129                 model.put("tabLookup", Boolean.TRUE);
130                 uiLookup(model, id, selection, limit);
131 
132             } else if ("sparql".equals(action)) {
133                 final String query = getParameter("query", String.class, null, model);
134                 view = "/sparql";
135                 model.put("tabSparql", Boolean.TRUE);
136                 uiSparql(model, query, timeout);
137 
138             } else if ("entity-mentions".equals(action)) {
139                 final URI entityID = getParameter("entity", URI.class, null, model);
140                 final URI property = getParameter("property", URI.class, null, model);
141                 final Value value = getParameter("value", Value.class, null, model);
142                 view = "/entity-mentions";
143                 model.put("tabReports", Boolean.TRUE);
144                 model.put("subtabEntityMentions", Boolean.TRUE);
145                 uiReportEntityMentions(model, entityID, property, value, limit);
146 
147             } else if ("entity-mentions-aggregate".equals(action)) {
148                 final URI entityID = getParameter("entity", URI.class, null, model);
149                 view = "/entity-mentions-aggregate";
150                 model.put("tabReports", Boolean.TRUE);
151                 model.put("subtabEntityMentionsAggregate", Boolean.TRUE);
152                 uiReportEntityMentionsAggregate(model, entityID);
153 
154             } else if ("mention-value-occurrences".equals(action)) {
155                 final URI entityID = getParameter("entity", URI.class, null, model);
156                 final URI property = getParameter("property", URI.class, null, model);
157                 view = "/mention-value-occurrences";
158                 model.put("tabReports", Boolean.TRUE);
159                 model.put("subtabMentionValueOccurrences", Boolean.TRUE);
160                 uiReportMentionValueOccurrences(model, entityID, property);
161 
162             } else if ("mention-property-occurrences".equals(action)) {
163                 final URI entityID = getParameter("entity", URI.class, null, model);
164                 view = "/mention-property-occurrences";
165                 model.put("tabReports", Boolean.TRUE);
166                 model.put("subtabMentionPropertyOccurrences", Boolean.TRUE);
167                 uiReportMentionPropertyOccurrences(model, entityID);
168 
169             } else {
170                 uiStatus(model);
171             }
172 
173         } catch (final Throwable ex) {
174             if (ex instanceof OperationException) {
175                 final OperationException oex = (OperationException) ex;
176                 model.put("error", oex.getOutcome().toString());
177                 if (oex.getOutcome().getStatus() == Outcome.Status.ERROR_UNEXPECTED) {
178                     LOGGER.error("Unexpected error", ex);
179                 }
180             } else {
181                 model.put("error", ex.getMessage());
182                 LOGGER.error("Unexpected error", ex);
183             }
184         }
185 
186         return new Viewable(view, model);
187     }

189     @SuppressWarnings("unchecked")
190     private <T> T getParameter(final String name, final Class<T> clazz,
191             @Nullable final T defaultValue, @Nullable final Map<String, Object> model) {
192         T result = defaultValue;
193         final String stringValue = getUriInfo().getQueryParameters().getFirst(name);
194         if (stringValue != null && !"".equals(stringValue)) {
195             if (Value.class.isAssignableFrom(clazz)) {
196                 final char c = stringValue.charAt(0);
197                 if (c == '\'' || c == '"' || c == '<' || //
198                         stringValue.indexOf(':') >= 0 && stringValue.indexOf('/') < 0) {
199                     try {
200                         final Value value = Data.parseValue(stringValue, Data.getNamespaceMap());
201                         if (clazz.isInstance(value)) {
202                             result = clazz.cast(value);
203                         }
204                     } catch (final Throwable ex) {
205                         // ignore
206                     }
207                 }
208                 if (result == defaultValue) {
209                     if (URI.class.equals(clazz)) {
210                         result = (T) Data.getValueFactory().createURI(Data.cleanIRI(stringValue));
211                     } else if (clazz.isAssignableFrom(Literal.class)) {
212                         result = (T) Data.getValueFactory().createLiteral(stringValue);
213                     }
214                 }
215             } else {
216                 result = Data.convert(stringValue, clazz, defaultValue);
217             }
218         }
219         if (result != null) {
220             model.put(name, result);
221         }
222         return result;
223     }

225     private void uiStatus(final Map<String, Object> model) {
226 
227         // Emit uptime and percentage spent in GC
228         final StringBuilder builder = new StringBuilder();
229         final long uptime = ManagementFactory.getRuntimeMXBean().getUptime();
230         final long days = uptime / (24 * 60 * 60 * 1000);
231         final long hours = uptime / (60 * 60 * 1000) - days * 24;
232         final long minutes = uptime / (60 * 1000) - (days * 24 + hours) * 60;
233         long gctime = 0;
234         for (final GarbageCollectorMXBean bean : ManagementFactory.getGarbageCollectorMXBeans()) {
235             gctime += bean.getCollectionTime(); // assume 1 bean or they don't work in parallel
236         }
237         builder.append(days == 0 ? "" : days + "d").append(hours == 0 ? "" : hours + "h")
238                 .append(minutes).append("m uptime, ").append(gctime * 100 / uptime).append("% gc");
239 
240         // Emit memory usage
241         final MemoryUsage heap = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
242         final MemoryUsage nonHeap = ManagementFactory.getMemoryMXBean().getNonHeapMemoryUsage();
243         final long used = heap.getUsed() + nonHeap.getUsed();
244         final long committed = heap.getCommitted() + nonHeap.getCommitted();
245         final long mb = 1024 * 1024;
246         long max = 0;
247         for (final MemoryPoolMXBean bean : ManagementFactory.getMemoryPoolMXBeans()) {
248             max += bean.getPeakUsage().getUsed(); // assume maximum at same time in all pools
249         }
250         builder.append("; ").append(used / mb).append("/").append(max / mb).append("/")
251                 .append(committed / mb).append(" MB memory used/peak/committed");
252 
253         // Emit thread numbers
254         final int numThreads = ManagementFactory.getThreadMXBean().getThreadCount();
255         final int maxThreads = ManagementFactory.getThreadMXBean().getPeakThreadCount();
256         final long startedThreads = ManagementFactory.getThreadMXBean()
257                 .getTotalStartedThreadCount();
258         builder.append("; ").append(numThreads).append("/").append(maxThreads).append("/")
259                 .append(startedThreads).append(" threads active/peak/started");
260 
261         model.put("version", VERSION);
262         model.put("status", builder.toString());
263     }

265     private void uiSparql(final Map<String, Object> model, @Nullable final String query,
266             @Nullable final Long timeout) throws Throwable {
267 
268         // Emit the example queries
269         if (!getUIConfig().getSparqlExamples().isEmpty()) {
270             final List<String> links = Lists.newArrayList();
271             final StringBuilder script = new StringBuilder();
272             int index = 0;
273             for (final Example example : getUIConfig().getSparqlExamples()) {
274                 links.add("<a href=\"#\" onclick=\"$('#query').val(sparqlExample(" + index
275                         + "))\">" + RenderUtils.escapeHtml(example.getLabel()) + "</a>");
276                 script.append("if (queryNum == ").append(index).append(") {\n");
277                 script.append("  return \"")
278                         .append(example.getValue().replace("\n", "\\n").replace("\"", "\\\""))
279                         .append("\";\n");
280                 script.append("}\n");
281                 ++index;
282             }
283             script.append("return \"\";\n");
284             model.put("examplesScript", script.toString());
285             model.put("examplesLinks", links);
286         }
287 
288         // Emit the query and evaluate its results, if possible
289         if (query != null) {
290 
291             // Emit the query results (only partially materialized).
292             final long ts = System.currentTimeMillis();
293             final Stream<BindingSet> stream = sendQuery(query, timeout);
294             @SuppressWarnings("unchecked")
295             final List<String> vars = stream.getProperty("variables", List.class);
296             final Iterator<BindingSet> iterator = stream.iterator();
297             final List<BindingSet> fetched = ImmutableList.copyOf(Iterators.limit(iterator,
298                     MAX_FETCHED_RESULTS));
299             model.put(
300                     "results",
301                     RenderUtils.renderSolutionTable(vars,
302                             Iterables.concat(fetched, Stream.create(iterator))));
303             final long elapsed = System.currentTimeMillis() - ts;
304 
305             // Emit the results message
306             final StringBuilder builder = new StringBuilder();
307             if (fetched.size() < MAX_FETCHED_RESULTS) {
308                 builder.append(fetched.size()).append(" results in ");
309                 builder.append(elapsed).append(" ms");
310             } else {
311                 builder.append("more than ").append(MAX_FETCHED_RESULTS);
312             }
313             if (timeout != null && elapsed > timeout) {
314                 builder.append(" (timed out, more results may be available)");
315             }
316             model.put("resultsMessage", builder.toString());
317         }
318     }

337     private boolean uiLookupResource(final Map<String, Object> model, final URI resourceID,
338             final URI selection, final int limit) throws Throwable {
339 
340         // Retrieve the resource record for the URI specified. Return false if not found
341         final Record resource = getRecord(KS.RESOURCE, resourceID);
342         if (resource == null) {
343             return false;
344         }
345 
346         // Get mentions, generating links and identifying selected mentions and entities
347         final List<Record> mentions = getResourceMentions(resourceID);
348         URI selectedEntityID = null;
349         Record selectedMention = null;
350         final List<String> mentionLinks = Lists.newArrayList();
351         final Set<String> entityLinks = Sets.newTreeSet();
352         final String linkTemplate = "<a onclick=\"select('%s')\" href=\"#\">%s</a>";
353         for (final Record mention : mentions) {
354             final URI mentionID = mention.getID();
355             mentionLinks.add(String.format(linkTemplate, mentionID,
356                     RenderUtils.shortenURI(mentionID)));
357             if (mention.getID().equals(selection)) {
358                 selectedMention = mention;
359             }
360             for (final URI entityID : mention.get(KS.REFERS_TO, URI.class)) {
361                 entityLinks.add(String.format(linkTemplate, entityID,
362                         RenderUtils.shortenURI(entityID)));
363                 if (entityID.equals(selection)) {
364                     selectedEntityID = selection;
365                 }
366             }
367         }
368 
369         // Select the resource template
370         model.put("resource", Boolean.TRUE);
371 
372         // Emit mentions and entities dropdown lists
373         if (!mentionLinks.isEmpty()) {
374             model.put("resourceMentionsCount", mentionLinks.size());
375             model.put("resourceMentions", mentionLinks);
376         }
377         if (!entityLinks.isEmpty()) {
378             model.put("resourceEntitiesCount", entityLinks.size());
379             model.put("resourceEntities", entityLinks);
380         }
381 
382         // Emit resource text box
383         final Representation representation = getRepresentation(resourceID);
384         if (representation != null) {
385             final String text = representation.writeToString();
386             final StringBuilder builder = new StringBuilder();
387             if (!mentions.isEmpty()) {
388                 RenderUtils.renderText(text, mentions, selection, true, false, getUIConfig(),
389                         builder);
390             } else {
391                 final Record metadata = representation.getMetadata();
392                 model.put("resourcePrettyPrint", Boolean.TRUE);
393                 RenderUtils.renderText(text, metadata.getUnique(NIE.MIME_TYPE, String.class),
394                         builder);
395             }
396             model.put("resourceText", builder.toString());
397         }
398 
399         // Emit the details box (mention / entity / resource metadata)
400         if (selectedEntityID != null) {
401             // One entity selected - emit its describe triples
402             final List<BindingSet> bindings = getEntityDescribeTriples(selection, limit);
403             final int total = bindings.size() < limit ? bindings.size()
404                     : countEntityDescribeTriples(selection);
405             model.put("resourceDetailsBody",
406                     String.join("", RenderUtils.renderSolutionTable(DESCRIBE_VARS, bindings)));
407             model.put("resourceDetailsTitle", String.format("<strong> Entity %s "
408                     + "(%d triples out of %d)</strong>", RenderUtils.render(selection),
409                     bindings.size(), total));
410 
411         } else if (selectedMention != null) {
412             // One mention selected - emit its details
413             final StringBuilder builder = new StringBuilder("<strong>Mention ");
414             RenderUtils.render(selection, builder);
415             builder.append("</strong>");
416             final List<URI> entityURIs = selectedMention.get(KS.REFERS_TO, URI.class);
417             if (!entityURIs.isEmpty()) {
418                 builder.append("&nbsp;&nbsp;&#10143;&nbsp;&nbsp;<strong>")
419                         .append(entityURIs.size() == 1 ? "Entity" : "Entities")
420                         .append("</strong>");
421                 for (final URI entityURI : entityURIs) {
422                     builder.append("&nbsp;&nbsp;<strong>");
423                     RenderUtils.render(entityURI, builder);
424                     builder.append("</strong> <a href=\"#\" onclick=\"select('")
425                             .append(RenderUtils.escapeHtml(entityURI)).append("')\">(select)</a>");
426                 }
427             }
428             model.put("resourceDetailsTitle", builder.toString());
429             model.put("resourceDetailsBody", RenderUtils.render(selectedMention));
430 
431         } else {
432             // Nothing selected - emit resource metadata
433             model.put("resourceDetailsTitle", "<strong>Resource metadata</strong>");
434             model.put("resourceDetailsBody", RenderUtils.render(resource));
435         }
436 
437         // Signal success
438         return true;
439     }

441     private boolean uiLookupMention(final Map<String, Object> model, final URI mentionID,
442             final int limit) throws Throwable {
443 
444         // Retrieve the mention for the URI specified. Return false if not found
445         final Record mention = getRecord(KS.MENTION, mentionID);
446         if (mention == null) {
447             return false;
448         }
449 
450         // Select the mention template
451         model.put("mention", Boolean.TRUE);
452 
453         // Emit the mention description box
454         model.put("mentionData", RenderUtils.render(mention));
455 
456         // Emit the resource box, including the mention snipped
457         final URI resourceID = mention.getUnique(KS.MENTION_OF, URI.class, null);
458         if (resourceID != null) {
459             model.put("mentionResourceLink",
460                     RenderUtils.render(resourceID, mentionID, new StringBuilder()).toString());
461             final Representation representation = getRepresentation(resourceID);
462             if (representation == null) {
463                 model.put("mentionResourceExcerpt", "RESOURCE CONTENT NOT AVAILABLE");
464             } else {
465                 final String text = representation.writeToString();
466                 model.put("mentionResourceExcerpt", RenderUtils.renderText(text,
467                         ImmutableList.of(mention), null, false, true, getUIConfig(),
468                         new StringBuilder()));
469             }
470         }
471 
472         // Emit the denoted entities box
473         final List<URI> entityIDs = mention.get(KS.REFERS_TO, URI.class);
474         if (!entityIDs.isEmpty()) {
475 
476             // Emit the triples of the first denoted entity, including their total number
477             final URI entityID = entityIDs.iterator().next();
478             final List<BindingSet> describeTriples = getEntityDescribeTriples(entityID, limit);
479             final int total = describeTriples.size() < limit ? describeTriples.size()
480                     : countEntityDescribeTriples(entityID);
481             model.put("mentionEntityTriplesShown", describeTriples.size());
482             model.put("mentionEntityTriplesTotal", total);
483             model.put("mentionEntityTriples", String.join("", RenderUtils.renderSolutionTable( //
484                     ImmutableList.of("subject", "predicate", "object", "graph"), describeTriples)));
485 
486             // Emit the link(s) to the pages for all the denoted entities
487             if (entityIDs.size() == 1) {
488                 model.put("mentionEntityLink", RenderUtils.render(entityID));
489             } else {
490                 final StringBuilder builder = new StringBuilder();
491                 for (final URI id : entityIDs) {
492                     builder.append(builder.length() > 0 ? "&nbsp;&nbsp;" : "");
493                     RenderUtils.render(id, builder);
494                 }
495                 model.put("mentionEntityLinks", builder.toString());
496             }
497         }
498 
499         // Signal success
500         return true;
501     }

503     private boolean uiLookupEntity(final Map<String, Object> model, final URI entityID,
504             final int limit) throws Throwable {
505 
506         // Lookup (a subset of) describe triples and graph triples for the specified entity
507         final List<BindingSet> describeTriples = getEntityDescribeTriples(entityID, limit);
508         final List<BindingSet> graphTriples = getEntityGraphTriples(entityID, limit);
509         if (describeTriples.isEmpty() && graphTriples.isEmpty()) {
510             return false;
511         }
512 
513         // Select the entity template
514         model.put("entity", Boolean.TRUE);
515 
516         // Emit the describe box
517         if (!describeTriples.isEmpty()) {
518             final int total = describeTriples.size() < limit ? describeTriples.size()
519                     : countEntityDescribeTriples(entityID);
520             model.put("entityTriplesShown", describeTriples.size());
521             model.put("entityTriplesTotal", total);
522             model.put("entityTriples", String.join("", RenderUtils.renderSolutionTable( //
523                     ImmutableList.of("subject", "predicate", "object", "graph"), describeTriples)));
524         }
525 
526         // Emit the graph box
527         if (!graphTriples.isEmpty()) {
528             final int total = graphTriples.size() < limit ? graphTriples.size()
529                     : countEntityGraphTriples(entityID);
530             model.put("entityGraphShown", graphTriples.size());
531             model.put("entityGraphTotal", total);
532             model.put("entityGraph", String.join("", RenderUtils.renderSolutionTable( //
533                     ImmutableList.of("subject", "predicate", "object"), graphTriples)));
534         }
535 
536         // Emit the resources box
537         final List<Record> resources = getEntityResources(entityID, getUIConfig().getResultLimit());
538         if (!resources.isEmpty()) {
539 
540             // Emit resource and mention counts
541             final int[] counts = countEntityResourcesAndMentions(entityID);
542             model.put("entityResourcesShown", resources.size());
543             model.put("entityResourcesCount", counts[0]);
544             model.put("entityMentionsCount", counts[1]);
545 
546             // Emit the resources table
547             final StringBuilder builder = new StringBuilder();
548             final List<URI> overviewProperties = getUIConfig().getResourceOverviewProperties();
549             final int width = 75 / (overviewProperties.size() + 2);
550             final String th = "<th style=\"width: " + width + "%\">";
551             builder.append("<table class=\"sparql table table-condensed tablesorter\"><thead>\n");
552             builder.append("<tr>").append(th).append("resource ID</th>");
553             for (final URI property : overviewProperties) {
554                 builder.append(th)
555                         .append(RenderUtils.escapeHtml(Data.toString(property,
556                                 Data.getNamespaceMap()))).append("</th>");
557             }
558             builder.append(th);
559             if (resources.size() < getUIConfig().getResultLimit()) {
560                 builder.append("# mentions");
561             } else {
562                 builder.append("<span title=\"Number of mentions per resource may be lower than "
563                         + "the exact value as only a subset of all the entity mentions has been "
564                         + "considered for building this page\"># mentions (truncated)</title>");
565             }
566             builder.append("</th></tr>\n</thead><tbody>\n");
567             for (final Record resource : resources) {
568                 builder.append("<tr><td>");
569                 RenderUtils.render(resource.getID(), entityID, builder);
570                 for (final URI property : overviewProperties) {
571                     builder.append("</td><td>");
572                     RenderUtils.render(resource.get(property), builder);
573                 }
574                 builder.append("</td><td>");
575                 RenderUtils.render(resource.getUnique(NUM_MENTIONS, Integer.class, null), builder);
576                 builder.append("</td></tr>\n");
577             }
578             builder.append("</tbody></table>");
579             model.put("entityResources", builder.toString());
580         }
581 
582         // Signal success
583         return true;
584     }

586     private void uiReportEntityMentions(final Map<String, Object> model,
587             @Nullable final URI entityID, @Nullable final URI property,
588             @Nullable final Value value, final int limit) throws Throwable {
589 
590         // Do nothing in case the entity ID is missing
591         if (entityID == null) {
592             return;
593         }
594 
595         // Retrieve all the mentions satisfying the property[=value] optional filter
596         int numMentions = 0;
597         final List<Record> mentions = Lists.newArrayList();
598         for (final Record mention : getEntityMentions(entityID, Integer.MAX_VALUE, null)) {
599             if (property == null || !mention.isNull(property)
600                     && (value == null || mention.get(property).contains(value))) {
601                 ++numMentions;
602                 if (mentions.size() < limit) {
603                     mentions.add(mention);
604                 }
605             }
606         }
607 
608         // Render the mention table, including column toggling functionality
609         model.put("message", mentions.size() + " mentions shown out of " + numMentions);
610         model.put("mentionTable",
611                 RenderUtils.renderRecordsTable(new StringBuilder(), mentions, null, null));
612     }

652                 + "&value=${element}";
653         model.put("valueOccurrencesTable", RenderUtils.renderMultisetTable(new StringBuilder(),
654                 propertyValues, "Property value", "# Mentions", linkTemplate));

682     private Record getRecord(final URI layer, @Nullable final URI id) throws Throwable {
683         final Record record = id == null ? null : getSession().retrieve(layer).ids(id).exec()
684                 .getUnique();
685         if (record != null && layer.equals(KS.MENTION)) {
686             final String template = "SELECT ?e WHERE { ?e $$ $$ "
687                     + (getUIConfig().isDenotedByAllowsGraphs() ? ""
688                             : "FILTER NOT EXISTS { GRAPH ?e { ?s ?p ?o } } ") + "}";
689             for (final URI entityID : getSession()
690                     .sparql(template, getUIConfig().getDenotedByProperty(), id).execTuples()
691                     .transform(URI.class, true, "e")) {
692                 record.add(KS.REFERS_TO, entityID);
693             }
694         }
695         return record;
696     }

804     private Stream<Record> getEntityMentions(final URI entityID, final int maxResults,
805             @Nullable final int[] numMentions) throws Throwable {
806 
807         // First retrieve all the URIs of the mentions denoting the entity, via SPARQL query
808         final List<URI> mentionURIs = getSession()
809                 .sparql("SELECT ?m WHERE { $$ $$ ?m}", entityID,
810                         getUIConfig().getDenotedByProperty()).execTuples()
811                 .transform(URI.class, true, "m").toList();
812 
813         // Return the total number of mentions, if an holder variable has been supplied
814         if (numMentions != null) {
815             numMentions[0] = mentionURIs.size();
816         }
817 
818         // Then return a stream that returns the mention records as they are fetched from the KS
819         return getSession().retrieve(KS.MENTION).limit((long) maxResults).ids(mentionURIs).exec();
820     }

822     private List<Record> getEntityResources(final URI entityID, final int maxResults)
823             throws Throwable {
824 
825         // Retrieve up to maxResults IDs of resources mentioning the entity
826         final Multiset<URI> resourceIDs = HashMultiset.create();
827         try (Stream<URI> stream = getSession()
828                 .sparql("SELECT ?m WHERE { $$ $$ ?m }", entityID,
829                         getUIConfig().getDenotedByProperty()).execTuples()
830                 .transform(URI.class, true, "m")) {
831             for (final URI mentionID : stream) {
832                 final String string = mentionID.stringValue();
833                 final int index = string.indexOf("#");
834                 if (index > 0) {
835                     final URI resourceID = Data.getValueFactory().createURI(
836                             string.substring(0, index));
837                     if (resourceIDs.elementSet().size() == maxResults
838                             && !resourceIDs.contains(resourceID)) {
839                         break;
840                     }
841                     resourceIDs.add(resourceID);
842                 }
843             }
844         }
845 
846         // Lookup the resources in the KS
847         final List<Record> resources;
848         resources = getSession().retrieve(KS.RESOURCE).ids(resourceIDs).exec().toList();
849         for (final Record resource : resources) {
850             resource.set(NUM_MENTIONS, resourceIDs.count(resource.getID()));
851         }
852         return resources;
853     }

896     private Stream<BindingSet> sendQuery(final String query, final Long timeout) throws Throwable {
897 
898         final String form = RDFUtil.detectSparqlForm(query);
899         if (form.equalsIgnoreCase("select")) {
900             return closeOnCompletion(getSession().sparql(query).timeout(timeout).execTuples());
901 
902         } else if (form.equalsIgnoreCase("construct") || form.equals("describe")) {
903             final List<String> variables = ImmutableList.of("subject", "predicate", "object");
904             final Function<Statement, BindingSet> transformer = new Function<Statement, BindingSet>() {
905 
906                 @Override
907                 public BindingSet apply(final Statement statement) {
908                     return new ListBindingSet(variables, statement.getSubject(),
909                             statement.getPredicate(), statement.getObject());
910                 }
911 
912             };
913             final Stream<BindingSet> stream = getSession().sparql(query).timeout(timeout)
914                     .execTriples().transform(transformer, 1);
915             stream.setProperty("variables", variables);
916             return closeOnCompletion(stream);
917 
918         } else {
919             final boolean result = getSession().sparql(query).timeout(timeout).execBoolean();
920             final List<String> variables = ImmutableList.of("result");
921             final BindingSet bindings = new ListBindingSet(variables,
922                     BooleanLiteralImpl.valueOf(result));
923             return closeOnCompletion(Stream.create(new BindingSet[] { bindings }).setProperty(
924                     "variables", variables));
925         }
926     }
