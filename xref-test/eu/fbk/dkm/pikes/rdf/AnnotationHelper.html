64      private static final String PARTICIPATION_REGEX = ""
65              + "SUB? (COORD CONJ?)* (PMOD (COORD CONJ?)*)? ((VC OPRD?)|(IM OPRD?))*";

67      public static void main(final String... args) {
68  
69          for (final String arg : args) {
70  
71              LOGGER.info("Processing {} ...", arg);
72  
73              final String lcaseName = arg.toLowerCase();
74              int index = lcaseName.lastIndexOf(".naf");
75              if (index < 0) {
76                  index = lcaseName.lastIndexOf(".xml");
77                  if (index < 0) {
78                      index = lcaseName.length();
79                  }
80              }
81              final String prefix = arg.substring(0, index);
82  
83              try (Reader reader = IO.utf8Reader(IO.read(arg))) {
84                  final KAFDocument document = KAFDocument.createFromStream(reader);
85                  for (int i = 1; i <= document.getNumSentences(); ++i) {
86                      final Writer writer = new StringWriter();
87                      final RDFWriter rdfWriter = Rio.createWriter(RDFFormat.TURTLE, writer);
88                      process(document, i, rdfWriter);
89                      String rdf = writer.toString();
90                      rdf = rdf.replace("\n\n", "\n");
91                      rdf = rdf.replace("# \n", "\n");
92                      Files.write(rdf, new File(prefix + "." + i + ".ttl"), Charsets.UTF_8);
93                  }
94              } catch (final Throwable ex) {
95                  LOGGER.error("Failed to process " + arg, ex);
96              }
97          }
98          LOGGER.info("Done");
99      }

101     public static void process(final KAFDocument document, final int sentence,
102             final RDFHandler handler) throws RDFHandlerException {
103 
104         final List<Term> terms = document.getSentenceTerms(sentence);
105         final String text = getText(terms);
106 
107         final String ns = document.getPublic().uri + "." + sentence + "#";
108         final Map<Term, URI> termURIs = getURIs(terms, ns);
109 
110         handler.startRDF();
111         handler.handleNamespace("rdfs", RDFS.NAMESPACE);
112         handler.handleNamespace("owl", OWL.NAMESPACE);
113         handler.handleNamespace("dbpedia", "http://dbpedia.org/resource/");
114         handler.handleNamespace("pb", NS_PB);
115         handler.handleNamespace("nb", NS_NB);
116         handler.handleNamespace("vn", NS_VN);
117         handler.handleNamespace("fn", NS_FN);
118         handler.handleNamespace("", ns);
119 
120         handler.handleComment("");
121         handler.handleComment("");
122         handler.handleComment("=== TEXT ===");
123         handler.handleComment("");
124 
125         handler.handleStatement(new StatementImpl(new URIImpl(ns), RDFS.LABEL, new LiteralImpl(
126                 "\n\t" + text)));
127 
128         handler.handleComment("");
129         handler.handleComment("");
130         handler.handleComment("=== COREFERENCE ===");
131 
132         for (final Coref coref : document.getCorefs()) {
133             final List<Span<Term>> spans = Lists.newArrayList();
134             for (final Span<Term> span : coref.getSpans()) {
135                 if (span.getFirstTarget().getSent() == sentence) {
136                     spans.add(span);
137                 }
138             }
139             if (spans.size() > 1) {
140                 final StringBuilder builder = new StringBuilder();
141                 int index = 0;
142                 for (final Span<Term> span : spans) {
143                     builder.append(index == 0 ? "" : "  ").append("span").append(++index)
144                             .append("='").append(getText(span.getTargets())).append("'");
145                 }
146                 handler.handleComment("");
147                 handler.handleComment(builder.toString());
148                 final List<URI> headURIs = Lists.newArrayList();
149                 for (final Span<Term> span : spans) {
150                     final Term head = NAFUtils.extractHead(document, span);
151                     headURIs.add(termURIs.get(head));
152                 }
153                 Collections.sort(headURIs, Statements.valueComparator());
154                 for (int i = 1; i < headURIs.size(); ++i) {
155                     handler.handleStatement(new StatementImpl(headURIs.get(0), OWL.SAMEAS,
156                             headURIs.get(i)));
157                 }
158             }
159         }
160 
161         handler.handleComment("");
162         handler.handleComment("");
163         handler.handleComment("=== LINKING ===");
164         handler.handleComment("");
165 
166         final Multimap<URI, URI> links = HashMultimap.create();
167         for (final LinkedEntity entity : document.getLinkedEntities()) {
168             if (entity.getWFs().getFirstTarget().getSent() == sentence) {
169                 final Span<Term> span = KAFDocument.newTermSpan(document.getTermsByWFs(entity
170                         .getWFs().getTargets()));
171                 final Term head = NAFUtils.extractHead(document, span);
172                 links.put(termURIs.get(head), new URIImpl(entity.getReference()));
173             }
174         }
175         for (final Entity entity : document.getEntities()) {
176             if (entity.getSpans().get(0).getFirstTarget().getSent() == sentence) {
177                 final Term head = NAFUtils.extractHead(document, entity.getSpans().get(0));
178                 for (final ExternalRef ref : entity.getExternalRefs()) {
179                     if (ref.getResource().toLowerCase().contains("spotlight")) {
180                         links.put(termURIs.get(head), new URIImpl(ref.getReference()));
181                     }
182                 }
183             }
184         }
185         for (final URI termURI : Ordering.from(Statements.valueComparator()).sortedCopy(
186                 links.keySet())) {
187             for (final URI linkURI : Ordering.from(Statements.valueComparator()).sortedCopy(
188                     links.get(termURI))) {
189                 handler.handleStatement(new StatementImpl(termURI, OWL.SAMEAS, linkURI));
190             }
191         }
192 
193         handler.handleComment("");
194         handler.handleComment("");
195         handler.handleComment("=== FRAMES ===");
196 
197         for (final Predicate pred : document.getPredicatesBySent(sentence)) {
198 
199             final Term predTerm = NAFUtils.extractHead(document, pred.getSpan());
200             final URI predURI = termURIs.get(predTerm);
201 
202             final StringBuilder builder = new StringBuilder();
203             builder.append("pred='").append(predTerm.getStr()).append("'");
204             for (final Role role : pred.getRoles()) {
205                 builder.append("  ").append(role.getSemRole()).append("='")
206                         .append(getText(role.getTerms())).append("'");
207             }
208             handler.handleComment("");
209             handler.handleComment(builder.toString());
210 
211             final List<URI> typeURIs = Lists.newArrayList();
212             for (final ExternalRef ref : pred.getExternalRefs()) {
213                 if (Strings.isNullOrEmpty(ref.getReference())) {
214                     continue;
215                 } else if (NAFUtils.RESOURCE_PROPBANK.equals(ref.getResource())) {
216                     typeURIs.add(new URIImpl(NS_PB + ref.getReference()));
217                 } else if (NAFUtils.RESOURCE_NOMBANK.equals(ref.getResource())) {
218                     typeURIs.add(new URIImpl(NS_NB + ref.getReference()));
219                 } else if (NAFUtils.RESOURCE_VERBNET.equals(ref.getResource())) {
220                     typeURIs.add(new URIImpl(NS_VN + ref.getReference()));
221                 } else if (NAFUtils.RESOURCE_FRAMENET.equals(ref.getResource())) {
222                     typeURIs.add(new URIImpl(NS_FN + ref.getReference()));
223                 }
224             }
225             Collections.sort(typeURIs, Statements.valueComparator());
226             for (final URI typeURI : typeURIs) {
227                 handler.handleStatement(new StatementImpl(predURI, RDF.TYPE, typeURI));
228             }
229 
230             for (final Role role : pred.getRoles()) {
231 
232                 final Set<URI> roleURIs = Sets.newHashSet();
233                 roleURIs.add(new URIImpl((predTerm.getMorphofeat().startsWith("VB") ? NS_PB
234                         : NS_NB) + role.getSemRole().toLowerCase()));
235                 for (final ExternalRef ref : role.getExternalRefs()) {
236                     String id = ref.getReference().toLowerCase();
237                     final int index = id.lastIndexOf('@');
238                     id = index < 0 ? id : id.substring(index + 1);
239                     if (Strings.isNullOrEmpty(id)) {
240                         continue;
241                     } else if (NAFUtils.RESOURCE_VERBNET.equals(ref.getResource())) {
242                         roleURIs.add(new URIImpl(NS_VN + id));
243                     } else if (NAFUtils.RESOURCE_FRAMENET.equals(ref.getResource())) {
244                         roleURIs.add(new URIImpl(NS_FN + id));
245                     }
246                 }
247 
248                 final List<URI> argURIs = Lists.newArrayList();
249                 final Term roleHead = NAFUtils.extractHead(document, role.getSpan());
250                 if (roleHead != null) {
251                     final Set<Term> argHeads = Sets.newHashSet();
252                     for (final Term term : document.getTermsByDepAncestors(
253                             Collections.singleton(roleHead), PARTICIPATION_REGEX)) {
254                         final String pos = term.getMorphofeat();
255                         if (pos.startsWith("NN") || pos.startsWith("VB") || pos.startsWith("JJ")
256                                 || pos.startsWith("RB") || pos.startsWith("PRP") || pos.startsWith("WP")) {
257                             argHeads.add(NAFUtils.syntacticToSRLHead(document, term));
258                         }
259                     }
260                     for (final Term argHead : argHeads) {
261                         argURIs.add(termURIs.get(argHead));
262                     }
263                     Collections.sort(argURIs, Statements.valueComparator());
264                 }
265 
266                 for (final URI roleURI : Ordering.from(Statements.valueComparator()).sortedCopy(
267                         roleURIs)) {
268                     for (final URI argURI : argURIs) {
269                         handler.handleStatement(new StatementImpl(predURI, roleURI, argURI));
270                     }
271                 }
272             }
273         }
274 
275         handler.endRDF();
276     }

278     private static Map<Term, URI> getURIs(final List<Term> terms, final String ns) {
279 
280         final List<String> termStrings = Lists.newArrayList();
281         for (final Term term : terms) {
282             String s = term.getStr().toLowerCase();
283             for (int i = 0; i < s.length(); ++i) {
284                 final char c = s.charAt(i);
285                 if (!(c >= 'a' && c <= 'z') && !(c >= 'A' && c <= 'Z') && !(c >= '0' && c <= '9')
286                         && c != '-' && c != '_') {
287                     s = s.substring(0, i);
288                     break;
289                 }
290             }
291             termStrings.add(s);
292         }
293 
294         final Map<Term, URI> uris = Maps.newHashMap();
295         for (int i = 0; i < terms.size(); ++i) {
296             final String is = termStrings.get(i);
297             int index = 0;
298             int count = 0;
299             for (int j = 0; j < terms.size(); ++j) {
300                 if (j == i) {
301                     index = count;
302                 }
303                 if (termStrings.get(j).equals(is)) {
304                     ++count;
305                 }
306             }
307             final String id = count <= 1 ? is : is + "_" + (index + 1);
308             uris.put(terms.get(i), new URIImpl(ns + id));
309         }
310         return uris;
311     }

313     private static String getText(final List<Term> terms) {
314         final StringBuilder builder = new StringBuilder();
315         boolean atBeginning = true;
316         for (final Term term : terms) {
317             for (final WF word : term.getWFs()) {
318                 final String s = word.getForm();
319                 final boolean punct = ",".equals(s) || ";".equals(s) || ".".equals(s)
320                         || ":".equals(s);
321                 builder.append(atBeginning || punct ? "" : " ");
322                 builder.append(word.getForm());
323                 atBeginning = false;
324             }
325         }
326         return builder.toString();
327     }
