354     public static void readConnectionFile() throws UnsupportedEncodingException, JAXBException,
355             IOException {
356         String resourceName = "populator-ks-connection.xml";
357         URL url = nafPopulator.class.getResource(resourceName);
358 
359         JAXBContext jc = JAXBContext.newInstance("eu.fbk.knowledgestore.populator.naf.connection");
360         Unmarshaller unmarshaller = jc.createUnmarshaller();
361         if (url != null) {
362             KnowledgestoreServer myFile = (KnowledgestoreServer) unmarshaller
363                     .unmarshal(new InputStreamReader(url.openStream(), "UTF-8"));
364 	    // read SERVER_URL from XML file unless previously defined with command line parameters
365 	    if (SERVER_URL.equals("")) {
366 		SERVER_URL = myFile.getUrl();
367 	    }
368             USERNAME = myFile.getUsername();
369             PASSWORD = myFile.getPassword();
370             checkSession();
371         } else {
372             System.err
373                     .println("Error: populator-ks-connection.xml.xml file not found!\nYou should first create the connection file to the KS.");
374         }
375 
376     }

378     static void checkSession() {
379 	logger.info("checkSession SERVER_URL |" + SERVER_URL + "|");
380         if (store == null) {
381             // Initialize a KnowledgeStore client
382             store = Client.builder(SERVER_URL).maxConnections(16).validateServer(false).build();
383         }
384         if (store != null && (session == null || session.isClosed())) {
385             // Acquire a session for a given username/password pair
386             session = store.newSession(USERNAME, PASSWORD);
387         }
388         if (store == null || session == null || session.isClosed()) {
389 	    String errMsg = "";
390 	    if (store == null) { errMsg = "null store"; }
391 	    else if (session == null) { errMsg = "null session"; }
392 	    else { errMsg = "closed session"; }
393             logger.error("checkSession with SERVER_URL " + SERVER_URL + " : " + errMsg);
394             System.exit(0);
395         }
396         try {
397             session.download(new URIImpl("http://localhost/test")).exec();
398         } catch (IllegalStateException e) {
399             e.printStackTrace();
400             logger.error("checkSession with SERVER_URL " + SERVER_URL + " : IllegalStateException");
401             System.exit(0);
402         } catch (OperationException e) {
403             e.printStackTrace();
404             logger.error("checkSession with SERVER_URL " + SERVER_URL + " : OperationException");
405             System.exit(0);
406         }
407
