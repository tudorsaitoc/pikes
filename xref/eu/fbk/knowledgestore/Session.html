25  /**
26   * A KnowledgeStore user session.
27   * <p>
28   * Interaction with the KnowledgeStore through its Java API occurs within a {@code Session}, which
29   * encapsulates user-specific configuration and, if authenticated, user credentials. A
30   * {@code Session} can be obtained via {@link KnowledgeStore#newSession()} (for anonymous user) or
31   * {@link KnowledgeStore#newSession(URI, String)} (authenticated user) and must be released with
32   * {@link #close()} to freed resources possibly allocated to the session.
33   * </p>
34   * <p>
35   * The identifier of the session's user, if any, is provided by method {@link #getUser()}, while
36   * the configured password is available using {@link #getPassword()}. Method
37   * {@link #getNamespaces()} returns a session-specific, modifiable prefix-to-namespace map that
38   * can be used for interacting with the KnowledgeStore; this map is initially filled with mappings
39   * from {@link Data#getNamespaceMap()}. Namespace mappings play a relevant role in the interaction
40   * with the KnowledgeStore, as they are used for encoding and decoding {@link XPath} expressions,
41   * merge {@link Criteria}s and SPARQL queries.
42   * </p>
43   * <p>
44   * Operations of the KnowledgeStore API can be invoked starting from a {@code Session} object and
45   * using <i>operation objects</i>. Every operation has its own {@link Operation} sub-class, an
46   * instance of which can be obtained by calling a method in {@code Session}; after an operation
47   * object is created, a number of common and operation-specific parameters can be set on this
48   * object to specify and control the execution of the operation, which is triggered by calling one
49   * of the {@code exec()} / {@code execXXX()} methods on the operation objects. More in details,
50   * the API operations in the following table are offered:
51   * </p>
52   * <blockquote>
53   * <table border="1">
54   * <tr>
55   * <th>{@code Session} method</th>
56   * <th>{@code Operation} sub-class</th>
57   * <th>Type</th>
58   * <th>Endpoint</th>
59   * <th>Description</th>
60   * </tr>
61   * <tr>
62   * <td>{@link #download()}</td>
63   * <td>{@link Download}</td>
64   * <td>read-only</td>
65   * <td>CRUD</td>
66   * <td>Downloads one or more resource representations.</td>
67   * </tr>
68   * <tr>
69   * <td>{@link #upload()}</td>
70   * <td>{@link Upload}</td>
71   * <td>read-write</td>
72   * <td>CRUD</td>
73   * <td>Uploads one or more resource representations.</td>
74   * </tr>
75   * <tr>
76   * <td>{@link #count(URI)}</td>
77   * <td>{@link Count}</td>
78   * <td>read-only</td>
79   * <td>CRUD</td>
80   * <td>Count the number of resources / mentions / entities / axioms that match an optional
81   * condition.</td>
82   * </tr>
83   * <tr>
84   * <td>{@link #retrieve(URI)}</td>
85   * <td>{@link Retrieve}</td>
86   * <td>read-only</td>
87   * <td>CRUD</td>
88   * <td>Retrieves selected properties of resources / mentions / entities / axioms that match an
89   * optional condition</td>
90   * </tr>
91   * <tr>
92   * <td>{@link #create(URI)}</td>
93   * <td>{@link Create}</td>
94   * <td>read-write</td>
95   * <td>CRUD</td>
96   * <td>Creates one or more new resources / mentions / entities / axioms.</td>
97   * </tr>
98   * <tr>
99   * <td>{@link #merge(URI)}</td>
100  * <td>{@link Merge}</td>
101  * <td>read-write</td>
102  * <td>CRUD</td>
103  * <td>Merges local record descriptions of resources / mentions / entities / axioms with
104  * descriptions stored in the KnowledgeStore, applying supplied merge criteria.</td>
105  * </tr>
106  * <tr>
107  * <td>{@link #update(URI)}</td>
108  * <td>{@link Update}</td>
109  * <td>read-write</td>
110  * <td>CRUD</td>
111  * <td>Updates one or more properties of all the resources / mentions / entities / axioms that
112  * match an optional condition.</td>
113  * </tr>
114  * <tr>
115  * <td>{@link #delete(URI)}</td>
116  * <td>{@link Delete}</td>
117  * <td>read-write</td>
118  * <td>CRUD</td>
119  * <td>Deletes all the resources / mentions / entities / axioms matching an optional condition.</td>
120  * </tr>
121  * <tr>
122  * <td>{@link #match()}</td>
123  * <td>{@link Match}</td>
124  * <td>read-only</td>
125  * <td>CRUD</td>
126  * <td>Returns all the &lt;resource, mention, entity, axioms&gt; combinations matching optional
127  * condition where the mention refers to the resource and entity and the axioms have been
128  * extracted from the mention.</td>
129  * </tr>
130  * <tr>
131  * <td>{@link #sparql(String, Object...)}</td>
132  * <td>{@link Sparql}</td>
133  * <td>read-only</td>
134  * <td>SPARQL</td>
135  * <td>Evaluates a SPARQL select / construct / describe / ask query on crystallized axiom data.</td>
136  * </tr>
137  * </table>
138  * </blockquote>
139  * <p>
140  * Note that operation marked as read-write may be forbidden to anonymous or non-privileged users,
141  * based on the security configuration of the KnowledgeStore. {@code Session} objects are thread
142  * safe. Object equality is used for comparing {@code Session} instances.
143  * </p>
144  */
145 public interface Session extends Closeable {
146 
147     /**
148      * Returns the username associated to this {@code Session}, if any.
149      * 
150      * @return the username, null if the {@code Session} is anonymous
151      * @throws IllegalStateException
152      *             if the {@code Session} has been closed
153      */
154     @Nullable
155     String getUsername() throws IllegalStateException;
156 
157     /**
158      * Returns the user password, if any.
159      * 
160      * @return the password
161      * @throws IllegalStateException
162      *             if the {@code Session} has been closed
163      */
164     @Nullable
165     String getPassword() throws IllegalStateException;
166 
167     /**
168      * Returns the modifiable namespace map of this {@code Session}. The map is backed by the
169      * default mappings of {@code Data#getNamespaceMap()}. New mappings are stored in the
170      * {@code Session} map and may override the default mappings. Removal of mappings is performed
171      * on the {@code Session} map, and may reactivate the default mapping as a consequence.
172      * 
173      * @return the modifiable namespace map of this {@code Session}
174      * @throws IllegalStateException
175      *             if the {@code Session} has been closed
176      */
177     Map<String, String> getNamespaces() throws IllegalStateException;
178 
179     /**
180      * Creates a new {@code Download} operation object.
181      * 
182      * @param resourceID
183      *            the ID of the resource whose representation should be downloaded
184      * @return the created {@code Download} operation object
185      * @throws IllegalStateException
186      *             if the {@code Session} has been closed
187      */
188     Download download(URI resourceID) throws IllegalStateException;
189 
190     /**
191      * Creates a new {@code Upload} operation object.
192      * 
193      * @param resourceID
194      *            the ID of the resource whose representation should be updated
195      * @return the created {@code Upload} operation object
196      * @throws IllegalStateException
197      *             if the {@code Session} has been closed
198      */
199     Upload upload(URI resourceID) throws IllegalStateException;
200 
201     /**
202      * Creates a new {@code Count} operation object for counting records of the type specified.
203      * 
204      * @param type
205      *            the URI of the type of record to count, either {@link KS#RESOURCE},
206      *            {@link KS#MENTION}, {@link KS#ENTITY} or {@link KS#AXIOM}
207      * @return the created {@code Count} operation object
208      * @throws IllegalStateException
209      *             if the {@code Session} has been closed
210      */
211     Count count(URI type) throws IllegalStateException;
212 
213     /**
214      * Creates a new {@code Retrieve} operation object for retrieving records of the type
215      * specified.
216      * 
217      * @param type
218      *            the URI of the type of record to retrieve, either {@link KS#RESOURCE},
219      *            {@link KS#MENTION}, {@link KS#ENTITY} or {@link KS#AXIOM}
220      * @return the created {@code Retrieve} operation object
221      * @throws IllegalStateException
222      *             if the {@code Session} has been closed
223      */
224     Retrieve retrieve(URI type) throws IllegalStateException;
225 
226     /**
227      * Creates a new {@code Create} operation object for storing new records of the type
228      * specified.
229      * 
230      * @param type
231      *            the URI of the type of record to create, either {@link KS#RESOURCE},
232      *            {@link KS#MENTION}, {@link KS#ENTITY} or {@link KS#AXIOM}
233      * @return the created {@code Create} operation object
234      * @throws IllegalStateException
235      *             if the {@code Session} has been closed
236      */
237     Create create(URI type) throws IllegalStateException;
238 
239     /**
240      * Creates a new {@code Merge} operation object for merging local record description of the
241      * type specified with descriptions stored in the KnowledgeStore.
242      * 
243      * @param type
244      *            the URI of the type of record to merge, either {@link KS#RESOURCE},
245      *            {@link KS#MENTION}, {@link KS#ENTITY} or {@link KS#AXIOM}
246      * @return the created {@code Merge} operation object
247      * @throws IllegalStateException
248      *             if the {@code Session} has been closed
249      */
250     Merge merge(URI type) throws IllegalStateException;
251 
252     /**
253      * Creates a new {@code Update} operation object for updating one or more properties of
254      * records of the type specified.
255      * 
256      * @param type
257      *            the URI of the type of record to update, either {@link KS#RESOURCE},
258      *            {@link KS#MENTION}, {@link KS#ENTITY} or {@link KS#AXIOM}
259      * @return the created {@code Update} operation object
260      * @throws IllegalStateException
261      *             if the {@code Session} has been closed
262      */
263     Update update(URI type) throws IllegalStateException;
264 
265     /**
266      * Creates a new {@code Delete} operation object for deleting records of the type specified.
267      * 
268      * @param type
269      *            the URI of the type of record to delete, either {@link KS#RESOURCE},
270      *            {@link KS#MENTION}, {@link KS#ENTITY} or {@link KS#AXIOM}
271      * @return the created {@code Delete} operation object
272      * @throws IllegalStateException
273      *             if the {@code Session} has been closed
274      */
275     Delete delete(URI type) throws IllegalStateException;
276 
277     /**
278      * Creates a new {@code Match} operation object for matching &lt;resource, mention, entity,
279      * axioms&gt; 4-tuples matching specific conditions.
280      * 
281      * @return the created {@code Match} operation object
282      * @throws IllegalStateException
283      *             if the {@code Session} has been closed
284      */
285     Match match() throws IllegalStateException;
286 
287     /**
288      * Creates a new {@code Sparql} operation object for evaluating a SPARQL query on crystallized
289      * axiom data.
290      * 
291      * @param expression
292      *            the SPARQL query expression, not null
293      * @param arguments
294      *            arguments to be injected in {@code $$} placeholders in the query expression; can
295      *            be {@code URI}s, {@code Literal}s or scalar values that can be converted to
296      *            {@code Literal}s (e.g., strings, integers)
297      * @return the created {@code Sparql} operation object
298      * @throws IllegalStateException
299      *             if the {@code Session} has been closed
300      */
301     Sparql sparql(String expression, Object... arguments) throws IllegalStateException;
302 
303     /**
304      * Tests whether this {@code Session} instance has been closed.
305      * 
306      * @return true, if this {@code Session} instance has been closed
307      */
308     boolean isClosed();
309 
310     /**
311      * {@inheritDoc} Closes the {@code Session} instance, releasing any resource possibly
312      * allocated. Calling this method additional times has no effect. After this method is called,
313      * calls to other method of the {@code Session} interface will result in
314      * {@link IllegalStateException}s to be thrown.
315      */
316     @Override
317     void close();
318 
319 }
