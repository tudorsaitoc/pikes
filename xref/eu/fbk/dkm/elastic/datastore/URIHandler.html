52  public class URIHandler {
53      private static final Logger LOGGER = LoggerFactory.getLogger(URIHandler.class);
54      private static final String indexName = "urimappingindex";
55      private static final String extendedURIPropName = "ex";
56      private static final TimeValue scrollTimeOut = TimeValue.timeValueMinutes(1);
57      private final HashSet<URI> strongCompress;
58      private static final char URI_ESCAPE = (char)1;
59      private final String[] types;
60      private final Client client;
61      private final AtomicInteger counter;
62      private final BiMap<String, URI> mapper; //<compressed, extended>, compressed is a String rapresentation of a Integer.
63        
64      public URIHandler(Set<URI> weakCompression, HashSet<URI> strongCompression, Client client) throws IOException{
65          
66          this.strongCompress = strongCompression;
67          this.client = client;
68          counter = new AtomicInteger(0);
69          types = new String[2];
70          types[0] = "s"; //strongCompressed URIs.
71          types[1] = "w"; //weakCompressed URIs.
72          BiMap<String, URI> map = HashBiMap.create();
73          mapper = Maps.synchronizedBiMap(map);
74          initMap(weakCompression);
75          client.admin().indices().prepareOptimize(indexName).setMaxNumSegments(1).setFlush(true).execute().actionGet();
76      }
77      
78      
79  //methods called only from the contructor
80      /**
81       * check if the index exists if not the index is created. 
82       * The content of the index is loaded into the mapper.
83       * The set of weakCompression nameSpaces is added.
84       * @param nameSpaces
85       * @throws IOException 
86       */
87      private void initMap(Set<URI> nameSpaces) throws IOException{
88          if(isIndexExists()){ //if the index exists load the content.
89              LOGGER.debug("index: " + indexName + " already exists");
90              loadMappingFromIndex();            
91          }else{ //create it.
92              createIndex();
93              LOGGER.debug("index: " + indexName + " created");
94          }
95          addWeakCompressions(nameSpaces);
96      }
97      
98      /**
99       * loads the content of the index in the Map.
100      */
101     private void loadMappingFromIndex(){
102         SearchResponse response = client.prepareSearch(indexName).setTypes(types)
103                 .setQuery(QueryBuilders.constantScoreQuery(FilterBuilders.matchAllFilter()))
104                 .setScroll(scrollTimeOut).execute().actionGet();
105         while(true){ //will stop when the scroll doesn't return any hits.
106             for(SearchHit hit : response.getHits().getHits()){
107                 if(hit.isSourceEmpty())
108                     throw new UnsupportedOperationException("deserialization of projected object not supported, URIHandler");
109                 //else
110                 String compressedURI = hit.getId();
111                 Map<String, Object> document = hit.getSource();
112                 URI extendedURI = new URIImpl((String)document.get(extendedURIPropName));
113                 int uriInteger = Integer.parseInt(compressedURI);
114                 if(uriInteger > counter.get()) counter.set(uriInteger);
115                 mapper.put(compressedURI, extendedURI); //add the couple to the map.
116             }
117             response = client.prepareSearchScroll(response.getScrollId()).setScroll(scrollTimeOut).execute().actionGet();
118             if (response.getHits().getHits().length == 0){
119                 break;
120             }
121         }
122         counter.incrementAndGet(); //counter is the next id available.
123         
124         LOGGER.debug("compression: " + mapper.toString());
125     }

127     /**
128      * creates the index with name = indexName, the types with the names types and the mapping.
129      * @throws IOException 
130      */
131     private void createIndex() throws IOException{
132         CreateIndexRequestBuilder createRequest = client.admin().indices().prepareCreate(indexName)
133                 .setSettings(ImmutableSettings.settingsBuilder().put("index.refresh_interval", -1).put().build());
134         //set the mapping for the types.
135         for(String typeName : types){
136             XContentBuilder mapping = jsonBuilder().startObject()
137                     .field(typeName).startObject()
138                     .field("dynamic", "strict")
139                     .field("properties").startObject()
140                     .field("_timestamp").startObject().field("enabled", false).endObject()
141                 .field("_all").startObject().field("enabled", false).endObject()
142                     .field("norms").startObject().field("enabled", false).endObject()
143                     
144                     .field(extendedURIPropName).startObject()
145                     .field("type", "string")
146                     .field("index", "no").endObject()
147                     
148                     .endObject()
149                     .endObject()
150                     .endObject();
151            
152             createRequest.addMapping(typeName, mapping);
153             LOGGER.trace("uriHandler index: " + indexName + "." + typeName + " mapping: " + mapping.string());
154         }
155         CreateIndexResponse createResponse = createRequest.execute().actionGet();
156         if(!createResponse.isAcknowledged()){
157              throw new RuntimeException("can not create the index: " + indexName); 
158         }
159     }

161     /**
162      * adds to the index the missing weakCompressions.
163      * @param nameSpaces
164      * @throws IOException 
165      */
166     private void addWeakCompressions(Set<URI> nameSpaces) throws IOException{
167         for(URI uri : nameSpaces){
168             if(!mapper.containsValue(uri)){ //if the value is not already in the map (and in the index)
169                 store(uri, types[1]); //store the mapping as weak compression. 
170             }
171         }
172     }

178     /**
179      * adds a mapping of the specified type (strong or weak) in the index and in the map.
180      * the called must care about syncronization.
181      * @param extendedUri the uri to compress.
182      * @param type must be in the array types.
183      * @return the id of the new mapping. (compressed string that maps the extendedUri)
184      * @throws IOException 
185      */
186     private String store(URI extendedUri, String type) throws IOException{
187         String extended = extendedUri.toString();
188         XContentBuilder source = jsonBuilder().startObject().field(extendedURIPropName, extended).endObject();
189         String id;
190         synchronized(this){
191             id = Integer.toString(counter.get());
192             
193             IndexRequest indexReq = new IndexRequest(indexName, type, id).source(source);
194             this.waitYellowStatus();
195             IndexResponse indexResponse = client.index(indexReq).actionGet();
196             
197             if(!indexResponse.isCreated())
198                 throw new IllegalStateException("failed to add the weakCompression of " + source.string() + " in: " + indexName + "." + type);
199             this.waitYellowStatus();
200             FlushResponse flushResponse = client.admin().indices().prepareFlush(indexName).execute().actionGet();
201             
202             if(flushResponse.getFailedShards() != 0){ //if there are failures.
203                 String errorMessage = "flush of the idex: " + indexName + " failed:\n";
204                 for(ShardOperationFailedException failure : flushResponse.getShardFailures()){
205                     errorMessage += "\t shardId: " + failure.shardId() + " ; reason: " + failure.reason() + "\n";
206                 }
207                 throw new IllegalStateException(errorMessage); //explode
208             }
209             
210             mapper.put(Integer.toString(counter.get()), extendedUri); //add to the map
211             counter.incrementAndGet(); //increment the id.
212         }
213         return id;
214     }

216     /**
217      * 
218      * @param extended the uri to check if can be strong compressed
219      * @return the strong compressed string (escaped) or null if the string must not be strong compressed.
220      * @throws IOException 
221      */
222     private synchronized String strongCompression(URI extended) throws IOException{
223        // if(extended == null) return null;
224         String compressedNotEscaped = mapper.inverse().get(extended);
225         if(compressedNotEscaped != null)  //if there is already a strong compression in the map.
226             return String.valueOf(URI_ESCAPE) + compressedNotEscaped;
227         URI namespace = new URIImpl(extended.getNamespace());
228         if(!strongCompress.contains(namespace)) //if this uri has not to be strong compressed. 
229             return null;
230         
231         //the string must be strong compressed and is not already in the map (and in the index)
232         
233         //add the strong compression
234 
235         LOGGER.debug(extended.toString() + " --S--> " + compressedNotEscaped);
236 
237         compressedNotEscaped =  store(extended, types[0]);
238         return String.valueOf(URI_ESCAPE) + compressedNotEscaped;
239     }

241     /**
242      * encodes the specified uri using the mapper. (opposite of decode)
243      * @param uri
244      * @return the encoded rapresentation of the uri. 
245      * @throws java.io.IOException 
246      */
247     public String encode(URI uri) throws IOException{
248         LOGGER.debug("encoding of: " + uri);
249         //check if the uri can be strong compressed
250         String compressed = strongCompression(uri);
251         if(compressed != null){ //if the string has been strongCompressed and escaped.
252            // LOGGER.debug("compressed as: " + compressed);
253             return compressed; 
254         }
255         
256         
257         //if in the map there is no strong compression for the uri, check for weak compression
258         URI nameSpace = new URIImpl(uri.getNamespace());
259         
260         String encodedNotEscape = mapper.inverse().get(nameSpace);
261         
262         if(encodedNotEscape != null){ //if there is a weak compression
263             compressed = String.valueOf(URI_ESCAPE).concat(encodedNotEscape.concat(String.valueOf(URI_ESCAPE)).concat(uri.getLocalName()));
264             LOGGER.debug("compressed as: " + compressed);
265             return compressed;
266         }
267         //if no compression can be done.
268         compressed = String.valueOf(URI_ESCAPE).concat(uri.toString());
269         LOGGER.debug("compressed as: " + compressed);
270         return compressed;
271     }

273     /**
274      * return the extended uri rapresented by the compressed string. (opposite of encode)
275      * @param compressed
276      * @return 
277      */
278     public URI decode(String compressed){
279         LOGGER.trace("decompression of: " + compressed);
280            //if the string is not a URI
281         if(compressed.length() < 2 || !(compressed.charAt(0) == URI_ESCAPE)){
282             throw new IllegalArgumentException(compressed + " is not a valid URI for decoding. Expected <$escape><number> or <$escape><number><$escape><String> or <$escape><URI>");
283         }
284         compressed = compressed.substring(1); //remove the first escape character.
285         //check if it's strong or weak compressed
286        
287         if(Character.isDigit(compressed.charAt(0))){
288             int pos = compressed.indexOf(Character.toString(URI_ESCAPE), 1);
289             if(pos == -1){ //strong compressed
290                 URI res = mapper.get(compressed);
291                 LOGGER.trace("decompressed as: " + res);
292                 return res;
293             }
294             //it's weak compressed
295             String number = compressed.substring(0, pos);
296             String nameSpaceStr = mapper.get(number).toString();
297             String localNameStr = compressed.substring(pos+1); //discard the escape char.
298             URI res = new URIImpl(nameSpaceStr + localNameStr);
299             LOGGER.trace("decompressed as: " + res);
300             return res;
301             }
302         //if it's not compressed
303         URI res = new URIImpl(compressed);
304         LOGGER.trace("decompressed as: " + res);
305         return res;
306     }
307     
308     public boolean isEncodedUri(String str){
309         return str!= null && str.length()>1 && str.charAt(0) == URI_ESCAPE;
310     }
311      
312     private void waitYellowStatus(){
313         client.admin().cluster().prepareHealth().setWaitForYellowStatus().execute().actionGet();
314     }
315     private boolean isIndexExists(){
316         this.waitYellowStatus();
317         IndicesExistsRequest request = new IndicesExistsRequest(indexName);
318         IndicesExistsResponse response = client.admin().indices().exists(request).actionGet();
319         return response.isExists();
320     }
