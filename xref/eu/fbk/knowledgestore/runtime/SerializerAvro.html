81      public byte[] compressURI(final URI uri) {
82          Preconditions.checkNotNull(uri);
83          try {
84              final ByteArrayOutputStream stream = new ByteArrayOutputStream();
85              final Encoder encoder = EncoderFactory.get().directBinaryEncoder(stream, null);
86              final DatumWriter<Object> writer = new GenericDatumWriter<Object>(
87                      Schemas.COMPRESSED_IDENTIFIER);
88              this.dictionary.keyFor(uri); // ensure a compressed version of URI is available
89              final Object generic = encodeIdentifier(uri);
90              writer.write(generic, encoder);
91              return stream.toByteArray();
92  
93          } catch (final IOException ex) {
94              throw new Error("Unexpected exception (!): " + ex.getMessage(), ex);
95          }
96      }

98      public URI expandURI(final byte[] bytes) {
99          Preconditions.checkNotNull(bytes);
100         try {
101             final InputStream stream = new ByteArrayInputStream(bytes);
102             final Decoder decoder = DecoderFactory.get().directBinaryDecoder(stream, null);
103             final DatumReader<Object> reader = new GenericDatumReader<Object>(
104                     Schemas.COMPRESSED_IDENTIFIER);
105             final Object generic = reader.read(null, decoder);
106             return (URI) decodeNode(generic);
107 
108         } catch (final IOException ex) {
109             throw new Error("Unexpected exception (!): " + ex.getMessage(), ex);
110         }
111     }

182     private List<Object> decodeNodes(final Object generic) {
183         if (generic instanceof Iterable<?>) {
184             final Iterable<?> iterable = (Iterable<?>) generic;
185             final int size = Iterables.size(iterable);
186             final List<Object> nodes = Lists.<Object>newArrayListWithCapacity(size);
187             for (final Object element : iterable) {
188                 nodes.add(decodeNode(element));
189             }
190             return nodes;
191         }
192         Preconditions.checkNotNull(generic);
193         return ImmutableList.of(decodeNode(generic));
194     }
195 
196     private Object decodeNode(final Object generic) {
197         if (generic instanceof GenericRecord) {
198             final GenericRecord record = (GenericRecord) generic;
199             final Schema schema = record.getSchema();
200             if (schema.equals(Schemas.RECORD)) {
201                 return decodeRecord(record, null);
202             } else if (schema.equals(Schemas.PLAIN_IDENTIFIER)
203                     || schema.equals(Schemas.COMPRESSED_IDENTIFIER)) {
204                 return decodeIdentifier(record);
205             } else if (schema.equals(Schemas.STATEMENT)) {
206                 return decodeStatement(record);
207             }
208         }
209         return decodeLiteral(generic);
210     }
211 
212     @SuppressWarnings("unchecked")
213     private Record decodeRecord(final GenericRecord generic,
214             @Nullable final Set<URI> propertiesToDecode) {
215         final Record record = Record.create();
216         final GenericRecord encodedID = (GenericRecord) generic.get(0);
217         if (encodedID != null) {
218             record.setID((URI) decodeIdentifier(encodedID));
219         }
220         for (final GenericRecord prop : (Iterable<GenericRecord>) generic.get(1)) {
221             final URI property = (URI) decodeIdentifier((GenericRecord) prop.get(0));
222             final List<Object> values = decodeNodes(prop.get(1));
223             if (propertiesToDecode == null || propertiesToDecode.contains(property)) {
224                 record.set(property, values);
225             }
226         }
227         return record;
228     }

230     private Value decodeValue(final Object generic) {
231         if (generic instanceof GenericRecord) {
232             final GenericRecord record = (GenericRecord) generic;
233             final Schema schema = record.getSchema();
234             if (schema.equals(Schemas.COMPRESSED_IDENTIFIER)
235                     || schema.equals(Schemas.PLAIN_IDENTIFIER)) {
236                 return decodeIdentifier(record);
237             }
238         }
239         return decodeLiteral(generic);
240     }

242     private Resource decodeIdentifier(final GenericRecord record) {
243         final Schema schema = record.getSchema();
244         if (schema.equals(Schemas.COMPRESSED_IDENTIFIER)) {
245             try {
246                 return this.dictionary.objectFor((Integer) record.get(0));
247             } catch (final IOException ex) {
248                 throw new IllegalStateException("Cannot access dictionary: " + ex.getMessage(), ex);
249             }
250         } else if (schema.equals(Schemas.PLAIN_IDENTIFIER)) {
251             final String string = record.get(0).toString();
252             if (string.startsWith("_:")) {
253                 return this.factory.createBNode(string.substring(2));
254             } else {
255                 return this.factory.createURI(string);
256             }
257         }
258         throw new IllegalArgumentException("Unsupported encoded identifier: " + record);
259     }

261     private Literal decodeLiteral(final Object generic) {
262         if (generic instanceof GenericRecord) {
263             final GenericRecord record = (GenericRecord) generic;
264             final Schema schema = record.getSchema();
265             if (schema.equals(Schemas.STRING_LANG)) {
266                 final String label = record.get(0).toString(); // Utf8 class used
267                 final Object language = record.get(1);
268                 return this.factory.createLiteral(label, language.toString());
269             } else if (schema.equals(Schemas.SHORT)) {
270                 return this.factory.createLiteral(((Integer) record.get(0)).shortValue());
271             } else if (schema.equals(Schemas.BYTE)) {
272                 return this.factory.createLiteral(((Integer) record.get(0)).byteValue());
273             } else if (schema.equals(Schemas.BIGINTEGER)) {
274                 return this.factory.createLiteral(record.get(0).toString(), XMLSchema.INTEGER);
275             } else if (schema.equals(Schemas.BIGDECIMAL)) {
276                 return this.factory.createLiteral(record.get(0).toString(), XMLSchema.DECIMAL);
277             } else if (schema.equals(Schemas.CALENDAR)) {
278                 final int tz = (Integer) record.get(0);
279                 final GregorianCalendar calendar = new GregorianCalendar();
280                 calendar.setTimeInMillis((Long) record.get(1));
281                 calendar.setTimeZone(TimeZone.getTimeZone(String.format("GMT%s%02d:%02d",
282                         tz >= 0 ? "+" : "-", Math.abs(tz) / 60, Math.abs(tz) % 60)));
283                 return this.factory.createLiteral(this.datatypeFactory
284                         .newXMLGregorianCalendar(calendar));
285             }
286         } else if (generic instanceof CharSequence) {
287             return this.factory.createLiteral(generic.toString()); // Utf8 class used
288         } else if (generic instanceof Boolean) {
289             return this.factory.createLiteral((Boolean) generic);
290         } else if (generic instanceof Long) {
291             return this.factory.createLiteral((Long) generic);
292         } else if (generic instanceof Integer) {
293             return this.factory.createLiteral((Integer) generic);
294         } else if (generic instanceof Double) {
295             return this.factory.createLiteral((Double) generic);
296         } else if (generic instanceof Float) {
297             return this.factory.createLiteral((Float) generic);
298         }
299         Preconditions.checkNotNull(generic);
300         throw new IllegalArgumentException("Unsupported generic data: " + generic);
301     }

303     private Statement decodeStatement(final GenericRecord record) {
304         final Resource subj = decodeIdentifier((GenericRecord) record.get(0));
305         final URI pred = (URI) decodeIdentifier((GenericRecord) record.get(1));
306         final Value obj = decodeValue(record.get(2));
307         final Resource ctx = decodeIdentifier((GenericRecord) record.get(3));
308         if (ctx == null) {
309             return this.factory.createStatement(subj, pred, obj);
310         } else {
311             return this.factory.createStatement(subj, pred, obj, ctx);
312         }
313     }

315     private Object encodeNodes(final Iterable<? extends Object> nodes) {
316         final int size = Iterables.size(nodes);
317         if (size == 1) {
318             return encodeNode(Iterables.get(nodes, 0));
319         }
320         final List<Object> list = Lists.<Object>newArrayListWithCapacity(size);
321         for (final Object node : nodes) {
322             list.add(encodeNode(node));
323         }
324         return list;
325     }
326 
327     private Object encodeNode(final Object node) {
328         if (node instanceof Record) {
329             return encodeRecord((Record) node, null);
330         } else if (node instanceof Literal) {
331             return encodeLiteral((Literal) node);
332         } else if (node instanceof Resource) {
333             return encodeIdentifier((Resource) node);
334         } else if (node instanceof Statement) {
335             return encodeStatement((Statement) node);
336         }
337         Preconditions.checkNotNull(node);
338         throw new IllegalArgumentException("Unsupported node: " + node);
339     }

341     private Object encodeRecord(final Record record, @Nullable final Set<URI> propertiesToEncode) {
342         final URI id = record.getID();
343         final Object encodedID = id == null ? null : encodeIdentifier(id);
344         final List<Object> props = Lists.newArrayList();
345         for (final URI property : record.getProperties()) {
346             if (propertiesToEncode == null || propertiesToEncode.contains(property)) {
347                 ensureInDictionary(property);
348                 final List<? extends Object> nodes = record.get(property);
349                 if (property.equals(RDF.TYPE)) {
350                     for (final Object value : nodes) {
351                         if (value instanceof URI) {
352                             ensureInDictionary((URI) value);
353                         }
354                     }
355                 }
356                 final GenericData.Record prop = new GenericData.Record(Schemas.PROPERTY);
357                 prop.put("propertyURI", encodeIdentifier(property));
358                 prop.put("propertyValue", encodeNodes(nodes));
359                 props.add(prop);
360             }
361         }
362         return SerializerAvro.newGenericRecord(Schemas.RECORD, encodedID, props);
363     }

365     private Object encodeValue(final Value value) {
366         if (value instanceof Literal) {
367             return encodeLiteral((Literal) value);
368         } else if (value instanceof Resource) {
369             return encodeIdentifier((Resource) value);
370         } else {
371             throw new IllegalArgumentException("Unsupported value: " + value);
372         }
373     }
374 
375     private Object encodeIdentifier(final Resource identifier) {
376         if (identifier instanceof URI) {
377             try {
378                 final Integer key = this.dictionary.keyFor((URI) identifier, false);
379                 if (key != null) {
380                     return SerializerAvro.newGenericRecord(Schemas.COMPRESSED_IDENTIFIER, key);
381                 }
382             } catch (final IOException ex) {
383                 throw new IllegalStateException("Cannot access dictionary: " + ex.getMessage(), ex);
384             }
385         }
386         final String id = identifier instanceof BNode ? "_:" + ((BNode) identifier).getID()
387                 : identifier.stringValue();
388         return SerializerAvro.newGenericRecord(Schemas.PLAIN_IDENTIFIER, id);
389     }

391     private Object encodeLiteral(final Literal literal) {
392         final URI datatype = literal.getDatatype();
393         if (datatype == null || datatype.equals(XMLSchema.STRING)) {
394             final String language = literal.getLanguage();
395             if (language == null) {
396                 return literal.getLabel();
397             } else {
398                 return SerializerAvro.newGenericRecord(Schemas.STRING_LANG, literal.getLabel(),
399                         language);
400             }
401         } else if (datatype.equals(XMLSchema.BOOLEAN)) {
402             return literal.booleanValue();
403         } else if (datatype.equals(XMLSchema.LONG)) {
404             return literal.longValue();
405         } else if (datatype.equals(XMLSchema.INT)) {
406             return literal.intValue();
407         } else if (datatype.equals(XMLSchema.DOUBLE)) {
408             return literal.doubleValue();
409         } else if (datatype.equals(XMLSchema.FLOAT)) {
410             return literal.floatValue();
411         } else if (datatype.equals(XMLSchema.SHORT)) {
412             return SerializerAvro.newGenericRecord(Schemas.SHORT, literal.intValue());
413         } else if (datatype.equals(XMLSchema.BYTE)) {
414             return SerializerAvro.newGenericRecord(Schemas.BYTE, literal.intValue());
415         } else if (datatype.equals(XMLSchema.INTEGER)) {
416             return SerializerAvro.newGenericRecord(Schemas.BIGINTEGER, literal.stringValue());
417         } else if (datatype.equals(XMLSchema.DECIMAL)) {
418             return SerializerAvro.newGenericRecord(Schemas.BIGDECIMAL, literal.stringValue());
419         } else if (datatype.equals(XMLSchema.DATETIME)) {
420             final XMLGregorianCalendar calendar = literal.calendarValue();
421             return SerializerAvro.newGenericRecord(Schemas.CALENDAR, calendar.getTimezone(),
422                     calendar.toGregorianCalendar().getTimeInMillis());
423         }
424         throw new IllegalArgumentException("Unsupported literal: " + literal);
425     }

427     private Object encodeStatement(final Statement statement) {
428         return SerializerAvro.newGenericRecord(Schemas.STATEMENT,
429                 encodeIdentifier(statement.getSubject()),
430                 encodeIdentifier(statement.getPredicate()), //
431                 encodeValue(statement.getObject()), //
432                 encodeIdentifier(statement.getContext()));
433     }

435     private URI ensureInDictionary(final URI uri) {
436         try {
437             this.dictionary.keyFor(uri);
438             return uri;
439         } catch (final IOException ex) {
440             throw new IllegalStateException("Cannot access dictionary: " + ex.getMessage(), ex);
441         }
442     }

444     private static GenericData.Record newGenericRecord(final Schema schema,
445             final Object... fieldValues) {
446 
447         final GenericData.Record record = new GenericData.Record(schema);
448         for (int i = 0; i < fieldValues.length; ++i) {
449             record.put(i, fieldValues[i]);
450         }
451         return record;
452     }
