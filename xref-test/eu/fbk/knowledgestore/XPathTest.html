183     public static Record getMockResource() {
184         final GregorianCalendar calendar = new GregorianCalendar();
185         calendar.set(2013, 9, 23);
186 
187         final Record rep = Record.create();
188         rep.setID(new URIImpl("ks:r15_rep"));
189         rep.set(RDF.TYPE, KS.REPRESENTATION);
190         rep.set(NFO.FILE_NAME, "r15.txt");
191         rep.set(NFO.FILE_SIZE, 1533L);
192         rep.set(NFO.FILE_CREATED, new Date());
193         rep.set(NIE.MIME_TYPE, "text/plain");
194         rep.set(DCTERMS.CREATOR, "John", "Steve", "Mark");
195         rep.set(DCTERMS.ISSUED, 1000, 1005, 1007);
196 
197         final Record resource = Record.create();
198         resource.setID(new URIImpl("ks:r15"));
199         resource.set(RDF.TYPE, KS.RESOURCE);
200         resource.set(DCTERMS.TITLE, "This is the news title");
201         resource.set(DCTERMS.ISSUED, calendar);
202         resource.set(NIE.IS_STORED_AS, rep);
203         return resource;
204     }
