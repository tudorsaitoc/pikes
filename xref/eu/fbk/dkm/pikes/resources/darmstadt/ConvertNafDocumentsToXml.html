4   import java.nio.file.Files;
5   import java.util.ArrayList;
6   import java.util.HashMap;
7   import java.util.List;
8   import java.util.Set;
9   
10  import javax.xml.parsers.DocumentBuilder;

12  import javax.xml.transform.OutputKeys;
13  import javax.xml.transform.Transformer;

15  import javax.xml.transform.dom.DOMSource;
16  import javax.xml.transform.stream.StreamResult;

18  import com.google.common.collect.Sets;
19  
20  import org.openrdf.model.impl.URIImpl;
21  import org.slf4j.LoggerFactory;
22  import org.w3c.dom.Document;
23  import org.w3c.dom.Element;
24  
25  import ixa.kaflib.KAFDocument;
26  import ixa.kaflib.Opinion;
27

34  
35  public class ConvertNafDocumentsToXml {
36  
37  	private static final org.slf4j.Logger LOGGER = LoggerFactory.getLogger(ConvertNafDocumentsToXml.class);
38  
39  	public static void main(String[] args) {
40  		try {
41  			final CommandLine cmd = CommandLine
42  					.parser()
43  					.withName("yamcha-extractor")
44  					.withHeader("Check ESWC dataset with Darmstadt")
45  					.withOption("i", "input-folder", "the folder of the NAF corpus", "DIR", CommandLine.Type.DIRECTORY_EXISTING, true, false, true)
46  					.withOption("o", "output-file", "output file", "FILE", CommandLine.Type.FILE, true, false, true)
47  					.withOption("l", "label", "opinion label", "STRING", CommandLine.Type.STRING, true, false, true)
48  					.withOption("n", "numeric", "use numeric values for IDs")
49  					.withOption(null, "list", "use list of file to sort", "FILE", CommandLine.Type.FILE_EXISTING, true, false, false)
50  					.withLogger(LoggerFactory.getLogger("eu.fbk")).parse(args);
51  
52  			File inputFolder = cmd.getOptionValue("input-folder", File.class);
53  			File outputFile = cmd.getOptionValue("output-file", File.class);
54  			Set<String> labels = Sets.newHashSet(cmd.getOptionValue("label", String.class, "").split(","));
55  			
56  			File list = cmd.getOptionValue("list", File.class);
57  
58  			boolean useNumeric = cmd.hasOption("numeric");
59  
60  			DocumentBuilderFactory docFactory = DocumentBuilderFactory.newInstance();
61  			DocumentBuilder docBuilder = docFactory.newDocumentBuilder();
62  
63  			Document doc = docBuilder.newDocument();
64  			Element rootElement = doc.createElement("Sentences");
65  			doc.appendChild(rootElement);
66  
67  			int id = 0;
68  
69  			Iterable<KAFDocument> corpus = Corpus.create(false, inputFolder);
70  
71  			if (list != null) {
72  				LOGGER.info("Load file list from {}", list.getAbsolutePath());
73  				ArrayList<KAFDocument> files = new ArrayList<>();
74  				List<String> fileList = Files.readAllLines(list.toPath());
75  				for (String fileName : fileList) {
76  					fileName = fileName.trim();
77  					if (fileName.length() == 0) {
78  						continue;
79  					}
80  					String documentFileName = inputFolder + File.separator + fileName;
81  					files.add(KAFDocument.createFromFile(new File(documentFileName)));
82  				}
83  				corpus = files;
84  			}
85  
86  			int fileNum = 0;
87  			for (KAFDocument document : corpus) {
88  				fileNum++;
89  				LOGGER.info("File {}", document.getPublic().uri);
90  				Element sentenceElement = doc.createElement("sentence");
91  
92  				if (useNumeric) {
93  					sentenceElement.setAttribute("id", "" + id++);
94  				}
95  				else {
96  					URIImpl uri = new URIImpl(document.getPublic().uri);
97  					sentenceElement.setAttribute("id", uri.getLocalName());
98  				}
99  
100 				rootElement.appendChild(sentenceElement);
101 				Element textElement = doc.createElement("text");
102 				textElement.appendChild(doc.createTextNode(document.getRawText()));
103 				sentenceElement.appendChild(textElement);
104 
105 				for (Opinion opinion : document.getOpinions()) {
106 
107 				    boolean matches = false;
108 				    for (String l : labels) {
109 				        if (opinion.getLabel().contains(l)) {
110 				            matches = true;
111 				            break;
112 				        }
113 				    }
114 				    if (!matches) {
115 				        continue;
116 				    }
117 				    
118 					String expression = null;
119 					if (opinion.getOpinionExpression() == null) {
120 						continue;
121 					}
122 
123 					HashMap<String, Integer> indexes = new HashMap<>();
124 					indexes.put("holder-start", -1);
125 					indexes.put("holder-end", -1);
126 					indexes.put("target-start", -1);
127 					indexes.put("target-end", -1);
128 
129 					expression = opinion.getExpressionSpan().getStr();
130 					indexes.put("expression-start", opinion.getExpressionSpan().getTargets().get(0).getOffset());
131 					indexes.put("expression-end", opinion.getExpressionSpan().getTargets().get(opinion.getExpressionSpan().getTargets().size() - 1).getOffset() +
132 							opinion.getExpressionSpan().getTargets().get(opinion.getExpressionSpan().getTargets().size() - 1).getLength());
133 
134 					String holder = null;
135 					if (opinion.getOpinionHolder() != null && !opinion.getOpinionHolder().getTerms().isEmpty()) {
136 						holder = opinion.getHolderSpan().getStr();
137 						indexes.put("holder-start", opinion.getHolderSpan().getTargets().get(0).getOffset());
138 						indexes.put("holder-end", opinion.getHolderSpan().getTargets().get(opinion.getHolderSpan().getTargets().size() - 1).getOffset() +
139 								opinion.getHolderSpan().getTargets().get(opinion.getHolderSpan().getTargets().size() - 1).getLength());
140 					}
141 					else {
142 						holder = "null";
143 					}
144 
145 					String target = null;
146 					if (opinion.getOpinionTarget() != null && !opinion.getOpinionTarget().getTerms().isEmpty()) {
147 						target = opinion.getTargetSpan().getStr();
148 						indexes.put("target-start", opinion.getTargetSpan().getTargets().get(0).getOffset());
149 						indexes.put("target-end", opinion.getTargetSpan().getTargets().get(opinion.getTargetSpan().getTargets().size() - 1).getOffset() +
150 								opinion.getTargetSpan().getTargets().get(opinion.getTargetSpan().getTargets().size() - 1).getLength());
151 					}
152 					else {
153 						target = "null";
154 					}
155 
156 					Element frameElement = doc.createElement("frame");
157 
158 					Element holderElement = doc.createElement("holder");
159 					holderElement.setAttribute("value", holder);
160 					holderElement.setAttribute("start", Integer.toString(indexes.get("holder-start")));
161 					holderElement.setAttribute("end", Integer.toString(indexes.get("holder-end")));
162 
163 					Element topicElement = doc.createElement("topic");
164 					topicElement.setAttribute("value", target);
165 					topicElement.setAttribute("start", Integer.toString(indexes.get("target-start")));
166 					topicElement.setAttribute("end", Integer.toString(indexes.get("target-end")));
167 
168 					Element opinionElement = doc.createElement("opinion");
169 					opinionElement.setAttribute("value", expression);
170 					opinionElement.setAttribute("start", Integer.toString(indexes.get("expression-start")));
171 					opinionElement.setAttribute("end", Integer.toString(indexes.get("expression-end")));
172 					Element polarityElement = doc.createElement("polarity");
173 					polarityElement.appendChild(doc.createTextNode(opinion.getPolarity() != null ? normalizePolarity(opinion.getPolarity()) : "neutral"));
174 					opinionElement.appendChild(polarityElement);
175 
176 					frameElement.appendChild(holderElement);
177 					frameElement.appendChild(topicElement);
178 					frameElement.appendChild(opinionElement);
179 					sentenceElement.appendChild(frameElement);
180 				}
181 			}
182 
183 			LOGGER.info("Read {} files", fileNum);
184 
185 			TransformerFactory transformerFactory = TransformerFactory.newInstance();
186 			Transformer transformer = transformerFactory.newTransformer();
187 			transformer.setOutputProperty(OutputKeys.INDENT, "yes");
188 			transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "4");
189 			transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
190 
191 			DOMSource source = new DOMSource(doc);
192 
193 			StreamResult result = new StreamResult(outputFile);
194 //			StreamResult result = new StreamResult(System.out);
195 
196 			transformer.transform(source, result);
197 
198 		} catch (final Throwable ex) {
199 			CommandLine.fail(ex);
200 		}
201 	}
202 	
203 	private static String normalizePolarity( String polarity) {
204         String p = polarity.toLowerCase();
205         if (p.contains("pos")) {
206             return "positive";
207         } else if (p.contains("neg")) {
208             return "negative";
209         } else {
210             return "neutral";
211         }
212     }
