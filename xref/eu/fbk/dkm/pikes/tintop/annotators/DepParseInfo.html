16  public class DepParseInfo {
17  
18      private HashMap<Integer, Integer> depParents = new HashMap<>();
19      private HashMap<Integer, String> depLabels = new HashMap<>();
20  
21      private static String space(int width) {
22          StringBuilder b = new StringBuilder();
23          for (int i = 0; i < width; i++) {
24              b.append(' ');
25          }
26          return b.toString();
27      }
28  
29      // helper for toString()
30      private static void recToString(IndexedWord curr, StringBuilder sb, int offset, Set<IndexedWord> used,
31              SemanticGraph dependencies, HashMap<Integer, String> depLabels, HashMap<Integer, Integer> depParents) {
32          used.add(curr);
33          List<SemanticGraphEdge> edges = dependencies.outgoingEdgeList(curr);
34          Collections.sort(edges);
35          for (SemanticGraphEdge edge : edges) {
36              IndexedWord target = edge.getTarget();
37  
38              depParents.put(target.index(), edge.getSource().index());
39              depLabels.put(target.index(), edge.getRelation().toString());
40  
41              sb.append(space(2 * offset)).append("-> ").append(target).append(" (").append(edge.getRelation()).append(")\n");
42  
43              if (!used.contains(target)) { // recurse
44                  recToString(target, sb, offset + 1, used, dependencies, depLabels, depParents);
45              }
46          }
47      }
48  
49      public DepParseInfo(HashMap<Integer, Integer> depParents,
50              HashMap<Integer, String> depLabels) {
51          this.depParents = depParents;
52          this.depLabels = depLabels;
53      }
54  
55      public DepParseInfo(SemanticGraph dependencies) {
56          Collection<IndexedWord> rootNodes = dependencies.getRoots();
57          if (rootNodes.isEmpty()) {
58              // Shouldn't happen, but return something!
59              return;
60          }
61  
62          StringBuilder sb = new StringBuilder();
63          Set<IndexedWord> used = Generics.newHashSet();
64          for (IndexedWord root : rootNodes) {
65              depParents.put(root.index(), 0);
66              depLabels.put(root.index(), "root");
67              sb.append("-> ").append(root).append(" (root)\n");
68              recToString(root, sb, 1, used, dependencies, depLabels, depParents);
69          }
70  
71          Set<IndexedWord> nodes = Generics.newHashSet(dependencies.vertexSet());
72          nodes.removeAll(used);
73          while (!nodes.isEmpty()) {
74              IndexedWord node = nodes.iterator().next();
75              sb.append(node).append("\n");
76              recToString(node, sb, 1, used, dependencies, depLabels, depParents);
77              nodes.removeAll(used);
78          }
79  
80      }
81  
82      public HashMap<Integer, Integer> getDepParents() {
83          return depParents;
84      }
85  
86      public HashMap<Integer, String> getDepLabels() {
87          return depLabels;
88      }
89  
90      @Override public String toString() {
91          return "DepParseInfo{" +
92                  "depParents=" + depParents +
93                  ", depLabels=" + depLabels +
94                  '}';
95      }
96  }
