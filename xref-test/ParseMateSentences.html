2   import com.google.common.io.Files;
3   import edu.stanford.nlp.ling.CoreAnnotations;

11  import is2fbk.data.SentenceData09;

15  import org.w3c.dom.Document;
16  import org.w3c.dom.Element;
17  import org.w3c.dom.Node;
18  import org.w3c.dom.NodeList;

20  import se.lth.cs.srl.corpus.Sentence;
21  import se.lth.cs.srl.languages.Language;
22  import se.lth.cs.srl.pipeline.Pipeline;
23  
24  import javax.xml.parsers.DocumentBuilder;
25  import javax.xml.parsers.DocumentBuilderFactory;
26  import java.io.File;
27  import java.io.IOException;
28  import java.util.ArrayList;
29  import java.util.HashSet;
30  import java.util.List;
31  import java.util.Properties;
32  import java.util.zip.ZipFile;

37  
38  public class ParseMateSentences {
39  
40      private static final org.slf4j.Logger LOGGER = LoggerFactory.getLogger(ParseMateSentences.class);
41  
42      public static void main(String[] args) {
43  
44          try {
45  
46              final CommandLine cmd = CommandLine
47                      .parser()
48                      .withName("parse-fn-with-mate")
49                      .withHeader("Parse FrameNet example sentences with Stanford CoreNLP and Mate SRL parser")
50                      .withOption("f", "fn-folder", "FrameNet LU folder", "FOLDER", CommandLine.Type.DIRECTORY_EXISTING,
51                              true, false, true)
52                      .withOption("a", "anna-model", "Anna model", "FILE", CommandLine.Type.FILE_EXISTING, true, false,
53                              true)
54                      .withOption("m", "mate-model", "Mate SRL model", "FILE", CommandLine.Type.FILE_EXISTING, true,
55                              false, true)
56                      .withOption("o", "output", "Output folder", "FOLDER", CommandLine.Type.DIRECTORY,
57                              true, false, true)
58                      .withLogger(LoggerFactory.getLogger("eu.fbk")) //
59                      .parse(args);
60  
61  //            String fnFolder = "/Users/alessio/Desktop/lu";
62  //            String mateModelParser = "/Users/alessio/Desktop/srl-20130917/models/retrain-anna-20140819.model";
63  //            String mateModelSrl = "/Users/alessio/Desktop/srl-20130917/models/retrain-srl-20140818.model";
64              String fnFolder = cmd.getOptionValue("fn-folder", String.class);
65              String mateModelParser = cmd.getOptionValue("anna-model", String.class);
66              String mateModelSrl = cmd.getOptionValue("mate-model", String.class);
67  
68              File outputDir = cmd.getOptionValue("output", File.class);
69              if (!outputDir.exists()) {
70                  outputDir.mkdirs();
71              }
72              if (!outputDir.isDirectory()) {
73                  LOGGER.error("{} is not a directory", outputDir.getAbsolutePath());
74                  System.exit(1);
75              }
76  
77              Properties props = new Properties();
78              props.setProperty("annotators", "tokenize, ssplit, pos, lemma");
79              props.setProperty("tokenize.whitespace", "true");
80              props.setProperty("ssplit.eolonly", "true");
81  
82              LOGGER.info("Loading Stanford CoreNLP");
83              StanfordCoreNLP pipeline = new StanfordCoreNLP(props);
84  
85              LOGGER.info("Loading Anna Parser");
86              String[] arrayOfString = { "-model", mateModelParser };
87              Options localOptions = new Options(arrayOfString);
88              Parser mateParser = new Parser(localOptions);
89  
90              LOGGER.info("Loading Mate Srl");
91              ZipFile zipFile;
92              zipFile = new ZipFile(mateModelSrl);
93              SemanticRoleLabeler mateSrl = Pipeline.fromZipFile(zipFile);
94              zipFile.close();
95              Language.setLanguage(Language.L.valueOf("eng"));
96  
97              DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
98              DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
99  
100             File nafFolderFile = new File(fnFolder);
101             if (!nafFolderFile.exists()) {
102                 throw new IOException();
103             }
104             if (!nafFolderFile.isDirectory()) {
105                 throw new IOException();
106             }
107 
108             HashSet<String> allSentences = new HashSet<>();
109             int count = 0;
110 
111             long begin = System.currentTimeMillis();
112 
113             File[] listOfFiles = nafFolderFile.listFiles();
114             for (int i = 0; i < listOfFiles.length; i++) {
115                 File file = listOfFiles[i];
116                 if (file.isFile()) {
117 
118                     if (!file.getName().endsWith(".xml")) {
119                         continue;
120                     }
121 
122                     LOGGER.debug("### File: {}", file.getName());
123 
124                     StringBuilder stringBuilder = new StringBuilder();
125 
126                     Document doc = dBuilder.parse(file);
127                     doc.getDocumentElement().normalize();
128 
129                     NodeList nList;
130                     nList = doc.getElementsByTagName("sentence");
131                     for (int temp = 0; temp < nList.getLength(); temp++) {
132                         Node nNode = nList.item(temp);
133                         if (nNode.getNodeType() == Node.ELEMENT_NODE) {
134                             Element eElement = (Element) nNode;
135                             allSentences.add(eElement.getAttribute("ID"));
136                             count++;
137 
138                             NodeList labelList = eElement.getElementsByTagName("text");
139                             for (int j = 0; j < labelList.getLength(); j++) {
140                                 Node labelNode = labelList.item(j);
141                                 if (labelNode.getNodeType() == Node.ELEMENT_NODE) {
142                                     Element labelElement = (Element) labelNode;
143                                     String text = labelElement.getTextContent();
144 
145                                     LOGGER.debug("Text: {}", text);
146 
147                                     try {
148 
149                                         Annotation s = new Annotation(text);
150                                         pipeline.annotate(s);
151 
152                                         List<CoreMap> sents = s.get(CoreAnnotations.SentencesAnnotation.class);
153                                         for (CoreMap thisSent : sents) {
154                                             List<String> forms = new ArrayList<>();
155                                             List<String> poss = new ArrayList<>();
156                                             List<String> lemmas = new ArrayList<>();
157 
158                                             forms.add("<root>");
159                                             poss.add("<root>");
160                                             lemmas.add("<root>");
161 
162                                             List<CoreLabel> tokens = thisSent
163                                                     .get(CoreAnnotations.TokensAnnotation.class);
164                                             for (CoreLabel token : tokens) {
165                                                 String form = token.get(CoreAnnotations.TextAnnotation.class);
166                                                 String pos = token.get(CoreAnnotations.PartOfSpeechAnnotation.class);
167                                                 String lemma = token.get(CoreAnnotations.LemmaAnnotation.class);
168 
169                                                 form = AnnotatorUtils.codeToParenthesis(form);
170                                                 lemma = AnnotatorUtils.codeToParenthesis(lemma);
171                                                 pos = AnnotatorUtils.codeToParenthesis(pos);
172 
173                                                 forms.add(form);
174                                                 poss.add(pos);
175                                                 lemmas.add(lemma);
176                                             }
177 
178                                             SentenceData09 localSentenceData091 = new SentenceData09();
179                                             localSentenceData091.init(forms.toArray(new String[forms.size()]));
180                                             localSentenceData091.setPPos(poss.toArray(new String[poss.size()]));
181 
182                                             // Anna
183                                             SentenceData09 localSentenceData092 = mateParser
184                                                     .apply(localSentenceData091);
185                                             Sentence mateSentence = AnnotationPipeline
186                                                     .createSentenceFromAnna33(localSentenceData092,
187                                                             lemmas);
188 
189                                             // Mate
190                                             mateSrl.parseSentence(mateSentence);
191 
192                                             stringBuilder.append(mateSentence.toString());
193                                             stringBuilder.append("\n\n");
194                                         }
195                                     } catch (Throwable t) {
196                                         t.printStackTrace();
197                                     }
198                                 }
199                             }
200                         }
201                     }
202 
203                     File outputFile = new File(outputDir.getAbsolutePath() + File.separator + file.getName() + ".conll");
204                     Files.write(stringBuilder.toString(), outputFile, Charsets.UTF_8);
205                 }
206             }
207 
208             long end = System.currentTimeMillis();
209 
210             LOGGER.info("{} sentence(s) in {} millisecond(s)", count, (end - begin));
211 
212         } catch (Exception ex) {
213             CommandLine.fail(ex);
214         }
215     }
216 
217 }
