132     public static statistics readFile(String filepath, String disabled_Items, processNAFVariables vars) throws JAXBException,
133             IOException {
134     	vars.storePartialInforInCaseOfError = true;
135     	vars.filePath = new File(filepath);
136         logDebug("Start working with (" + vars.filePath.getName() + ")",vars);
137         String disabledItems = "";// Default empty so generate all layers of data
138         if (disabled_Items != null
139                 && (disabled_Items.matches("(?i)Entity") || disabled_Items.contains("(?i)Mention") || disabled_Items.contains("(?i)Resource"))) {
140             disabledItems = disabled_Items;
141             logDebug("Disable layer: " + disabledItems,vars);
142         }
143         readNAFFile(vars.filePath,vars);
144         for (Object obj : vars.doc
145                 .getNafHeaderOrRawOrTextOrTermsOrDepsOrChunksOrEntitiesOrCoreferencesOrConstituencyOrTimeExpressionsOrFactualitylayerOrTunitsOrLocationsOrDates()) {
146             if (obj instanceof NafHeader) {
147                 getNAFHEADERMentions((NafHeader) obj,vars);
148             } else if (obj instanceof Raw) {
149                 // raw text
150             	vars.rawText = ((Raw) obj).getvalue();
151             } else if (obj instanceof Text) {
152             	vars.globalText = (Text) obj;
153             } else if (obj instanceof Entities) {
154 		getEntitiesMentions((Entities) obj, disabledItems,vars);
155             } 
156 	    else if (obj instanceof Coreferences) {
157                 getCoreferencesMentions((Coreferences) obj,vars);
158             } 
159 	    else if (obj instanceof TimeExpressions) {
160                 getTimeExpressionsMentions((TimeExpressions) obj,vars);
161             } 
162 	    else if (obj instanceof Factualitylayer) {
163                 getFactualityMentions((Factualitylayer) obj,vars);
164             }
165 	    else if (obj instanceof Terms) {
166 	    	vars.globalTerms = (Terms) obj;
167             } else {
168                 // logError("Error:Uncatchable Object:"+obj);
169             }
170         }
171 	getSRLMentions(vars);
172 
173 	// logDebug("ROL1 before dumpStack()") ; Thread.currentThread().dumpStack();
174 
175 	fixMentions(vars);
176 
177         logDebug("End of NAF populating.",vars);
178         statistics st = new statistics();
179         st.setObjectMention((vars.corefMention2+vars.entityMen2));
180         st.setPER(vars.PER);
181         st.setORG(vars.ORG);
182         st.setLOC(vars.LOC);
183         st.setFin(vars.fin);
184         st.setMix(vars.mix);
185         st.setPRO(vars.PRO);
186         st.setNo_mapping(vars.no_mapping);
187         st.setTimeMention(vars.timeMention2);
188         st.setEventMention((vars.factualityMentions2 + vars.srlMention2));
189         st.setParticipationMention(vars.rolewithEntity2);
190         st.setEntity(vars.entityMen);
191         st.setCoref(vars.corefMention);
192         st.setCorefMentionEvent(vars.corefMentionEvent);
193         st.setCorefMentionNotEvent(vars.corefMentionNotEvent);
194         st.setFactuality(vars.factualityMentions);
195         st.setRole(vars.roleMentions);
196         st.setRolewithEntity(vars.rolewithEntity);
197         st.setRolewithoutEntity(vars.rolewithoutEntity);
198         st.setSrl(vars.srlMention);
199         st.setTimex(vars.timeMention);
200         
201         logDebug(st.getStats(),vars);
202         return st;

417 			
418 			
419 		if(modeactive){

421 				switch(exRObj.getReftype()){
422 				case "en": exrEn.addLast(exRObj); break;

425 				case "nl": exrNl.addLast(exRObj); break;

428 				
429 				if(exRObj.getReference().contains("://dbpedia.org")){
430 					exrEn.addLast(exRObj);
431 				}else if(exRObj.getReference().contains("://es.dbpedia.org")){
432 					exrEs.addLast(exRObj);
433 				}else if(exRObj.getReference().contains("://it.dbpedia.org")){
434 					exrIt.addLast(exRObj);
435 				}else {
436 					exrNl.addLast(exRObj);
437 				}
438 			}else{
439 				logWarn("Every entity must contain a 'references' element with type or DBpedia reference: not possible to add ExternalRef to null.",vars);
440 			}
441 		}
442 		
443 		if(exRObj!=null&&exRObj.getExternalRef()!=null && exRObj.getExternalRef().size()>0){
444 			for(ExternalRef ff :exRObj.getExternalRef()){
445 			getAllLayersOfExternalReferences( modeactive, exrEn,  ff, exrEs, exrIt,  exrNl,vars);
446 			}
447 		}else
448 			return;
449 	}

451 	private static String getHighConfidenceReferenceValue(
452 			LinkedList<ExternalRef> exrl) {
453     	ExternalRef current=null;
454     	Double dt=0.0;
455 		for(ExternalRef tmp:exrl){
456 			Double co=Double.parseDouble(tmp.getConfidence());
457 			if(current==null){
458 				current = tmp;
459 				dt=co;
460 			}else{
461 				if(co>dt){
462 					current = tmp;
463 					dt=co;
464 				}
465 			}
466 		}
467 		return current.getReference();
468 	}
