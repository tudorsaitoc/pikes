210     private String printResult(List<Record> result){
211         String log = "";
212         if(result == null){
213             log = "null";
214         }else{
215             if(result.isEmpty())
216                 log = "list of 0 Records";
217             else
218                 for(Record rec : result){
219                     if(rec != null) log += rec.toString(Data.getNamespaceMap(), true);
220                     else log += "null record;\n";
221                 }
222         }
223         log += "\n\n\n";
224         return log;       
225     }

228     private void startES(String configPath){
229         dataStore = new DataStoreElastic(configPath);
230         try {
231             dataStore.init();
232         } catch (IOException | IllegalStateException ex) {
233            LOGGER.debug("error in the init");
234         }
235         
236         try {
237             instance = (DataTransactionElastic) dataStore.begin(true);
238         } catch (IOException | IllegalStateException ex) {
239            LOGGER.debug("error in the begin");
240         }
241     } 
242     
243     private void closeES(){
244          try{
245             instance.end(true);
246         }catch (IOException | IllegalStateException ex) {
247            LOGGER.debug("error in the end");
248         }        
249         dataStore.close();
250     }
